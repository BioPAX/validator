3   /**
4    * A functional object (strategy)
5    * interface that contain the only method, which
6    * is called by the validator framework to get an
7    * identifier or name of a model element or parser
8    * and report an error about it.
9    *
10   * @author rodche
11   */
12  public interface Identifier {
13  
14    /**
15     * Gets the id or name of a model element or
16     * other object that matters during a validation
17     * and worth mentioning in a error message.
18     *
19     * This is called by the Validator framework to get a
20     * domain specific identifier or name of a model element
21     * to report an error about.
22     *
23     * This interface is required, because a domain specific 'get id'
24     * method, if any, is not known to the core validation framework
25     * in advance, and toString method may not be satisfactory.
26     *
27     * @param obj a model element or, e.g., stream
28     * @return id
29     */
30    String identify(Object obj);
31  
32  }
