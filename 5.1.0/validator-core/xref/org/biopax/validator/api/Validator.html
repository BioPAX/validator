1   package org.biopax.validator.api;
2   
3   
4   import java.io.InputStream;
5   import java.util.Collection;
6   import java.util.Set;
7   
8   import org.biopax.validator.api.beans.Validation;
9   
10  /**
11   * Object model validator.
12   *
13   * @author rodche
14   */
15  public interface Validator {
16  
17    /**
18     * Gets all the active validation rules.
19     *
20     * @return available (loaded) validation rules
21     */
22    Set<Rule<?>> getRules();
23  
24  
25    /**
26     * Gets all the registered validation results (where error cases get stored).
27     *
28     * @return results
29     */
30    Collection<Validation> getResults();
31  
32  
33    /**
34     * Associates a validation report with the data and creates in-memory model.
35     *
36     * The validation result is object where all the problems
37     * related to this data instance are collected.
38     *
39     * @param validation validation result object
40     * @param inputStream data input stream
41     */
42    void importModel(Validation validation, InputStream inputStream);
43  
44    /**
45     * Associates a model, element, or stream with the validation result.
46     *
47     * Some errors may occur during the data import and parsing before the model is created,
48     * e.g., in XML headers, software bugs. We want to report these problems as well.
49     *
50     * @param element object (e.g., existing Model)
51     * @param validation results object
52     */
53    void associate(Object element, Validation validation);
54  
55  
56    /**
57     * Associates, e.g., a property value with validation results
58     * by using the 'parent' element to find it. This allows for
59     * errors to be reported while model is being read,
60     * maybe before the child element is added to the model.
61     *
62     * @param parent model element
63     * @param child value
64     */
65    void indirectlyAssociate(Object parent, Object child);
66  
67  
68    /**
69     * Given the object, finds
70     * the corresponding validations
71     * (that it's been associated with).
72     *
73     * @param obj a model, element or even I/O handler
74     * @return validation results
75     */
76    Collection<Validation> findValidation(Object obj);
77  
78  
79    /**
80     * Post-model validation.
81     *
82     * The result also includes other
83     * problems that might occur during
84     * the model initialization and creation.
85     *
86     * @param validation results object
87     */
88    void validate(Validation validation);
89  
90  
91    /**
92     * Adds the validation error (with proper attributes)
93     * to registered validation objects associated with the object.
94     *
95     * @param obj object associated with a validation result (can be even stream, during import, but usually is a model element)
96     * @param errorCode error code
97     * @param reportedBy class name of a validation rule or a name of another validating class or method (AOP aspect method, jointpoint).
98     * @param isFixed if true, - find and set the attribute
99     * @param args additional message parameters (details)
100    */
101   void report(Object obj, String errorCode, String reportedBy, boolean isFixed, Object... args);
102 
103 }
