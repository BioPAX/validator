38      public Set<OntologyTermI> getValidTerms( String accession, boolean allowChildren, boolean useTerm ) {
39          Set<OntologyTermI> collectedTerms = new HashSet<>();
40  
41          final OntologyTermI term = getTermForAccession( accession );
42          if ( term != null ) {
43              if ( useTerm ) {
44                  collectedTerms.add( term );
45              }
46  
47              if ( allowChildren ) {
48                  collectedTerms.addAll( getAllChildren( term ) );
49              }
50          }
51  
52          return collectedTerms;
53      }

86      /**
87       * Add a new Term in the pool. It will be indexed by its ID.
88       *
89       * @param term the OntologyTerm to add in that OntologyAccess.
90       */
91      public void addTerm( OntologyTermI term ) {
92          ontologyTerms.add( term );
93          String id = term.getTermAccession();
94          if ( id2ontologyTerm.containsKey( id ) ) {
95              OntologyTermI old = id2ontologyTerm.get( id );
96              log.error( "WARNING: 2 Objects have the same ID (" + id
97                + "), the old one is being replaced. old: "
98                + old.getPreferredName() + " new: " + term.getPreferredName() );
99          }
100         id2ontologyTerm.put( id, term );
101         flushRootsCache();
102     }
103 
104     /**
105      * Create a relashionship parent to child between two OntologyTerm.
106      *
107      * @param parentId The parent term.
108      * @param childId  The child term.
109      */
110     public void addLink(String parentId, String childId) {
111         OntologyTermI child = id2ontologyTerm.get(childId);
112         OntologyTermI parent = id2ontologyTerm.get(parentId);
113         if (child == null || parent == null) {
114             throw new NullPointerException("You must give a non null " +
115               "child/parent for addLink method!");
116         } else {
117             if (!children.containsKey(parent)) {
118                 children.put(parent, new HashSet<>());
119             }
120             if (!parents.containsKey(child)) {
121                 parents.put(child, new HashSet<>());
122             }
123             children.get(parent).add(child);
124             parents.get(child).add(parent);
125             flushRootsCache();
126         }
127     }
128 
129     /**
130      * Remove the Root cache from memory.<br/> That method should be called
131      * every time the collection of OntologyTerm is altered.
132      */
133     private void flushRootsCache() {
134         if ( roots != null ) {
135             // flush roots cache
136             roots.clear();
137             roots = null;
138         }
139     }

164     /**
165      * Get the Root terms of the ontology. The way to get it is as follow: pick a term at random, and go to his highest
166      * parent.
167      *
168      * @return a collection of Root term.
169      */
170     public Collection<OntologyTermI> getRoots() {
171         if ( roots != null ) {
172             return roots;
173         }
174 
175         // it wasn't precalculated, then do it here...
176         roots = new HashSet<>();
177 
178         for ( Iterator iterator = ontologyTerms.iterator(); iterator.hasNext(); ) {
179             OntologyTermI ontologyTerm = ( OntologyTermI ) iterator.next();
180             if ( !hasParent( ontologyTerm ) ) {
181                 roots.add( ontologyTerm );
182             }
183         }
184 
185         if ( roots.isEmpty() ) {
186             return Collections.emptyList();
187         }
188 
189         return roots;
190     }

223     public Set<OntologyTermI> getDirectParents( OntologyTermI term ) {
224         final Set<OntologyTermI> directParents = parents.get( term );
225         if ( directParents == null ) {
226             return Collections.emptySet();
227         } else {
228             return directParents;
229         }

232     public Set<OntologyTermI> getDirectChildren( OntologyTermI term ) {
233         final Set<OntologyTermI> directChildren = children.get( term );
234         if ( directChildren == null ) {
235             return Collections.emptySet();
236         } else {
237             return directChildren;
238         }
239     }

241     public Set<OntologyTermI> getAllParents( OntologyTermI term ) {
242         Set<OntologyTermI> parents = new HashSet<>();
243         getAllParents( term, parents );
244         return parents;
245     }
246 
247     private void getAllParents( OntologyTermI term, Set<OntologyTermI> parents ) {
248         final Collection<OntologyTermI> directParents = getDirectParents( term );
249         parents.addAll( directParents );
250         for ( OntologyTermI parent : directParents ) {
251             getAllParents( parent, parents );
252         }
253     }
254 
255     public Set<OntologyTermI> getAllChildren( OntologyTermI term ) {
256         Set<OntologyTermI> children = new HashSet<>();
257         getAllChildren( term, children );
258         return children;
259     }
260 
261     private void getAllChildren( OntologyTermI term, Set<OntologyTermI> children ) {
262         getAllChildren( "", term, children, new HashSet(512) );
263     }
264 
265     private void getAllChildren( String prefix, OntologyTermI term, Set<OntologyTermI> children,
266                                  Set<String> traversedChildren )
267     {
268         if( traversedChildren.contains( term.getTermAccession() ) ) {
269             return;
270         }
271         final Collection<OntologyTermI> directChildren = getDirectChildren( term );
272         traversedChildren.add( term.getTermAccession() );
273         children.addAll( directChildren );
274         for ( OntologyTermI child : directChildren ) {
275             getAllChildren( prefix+"     ", child, children, traversedChildren );
276         }
277     }

281     public void print() {
282         log.info( ontologyTerms.size() + " terms to display." );
283         final Collection<OntologyTermI> roots = getRoots();
284         log.info( this.roots.size() + " root(s) found." );
285         for ( OntologyTermI root : roots ) {
286             print( root );
287         }
288     }
289 
290     private void print( OntologyTermI term, String indent ) {
291         log.info( indent + term.getTermAccession() + "   " + term.getPreferredName() );
292         for ( OntologyTermI child : getDirectChildren( term ) ) {
293             print( child, indent + "  " );
294         }
295     }
296 
297     public void print( OntologyTermI term ) {
298         print( term, "" );
299     }
