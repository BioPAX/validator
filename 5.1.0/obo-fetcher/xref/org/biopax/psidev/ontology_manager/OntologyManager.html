75    /**
76    * Collect all available names in the given collection of OntologyAccess terms.
77    * @param terms the terms for which we want the names.
78    * @return a non null collection of names.
79    */
80    //if min. java is 10, we'd replace 'static' with 'default'
81    static Collection<String> getTermNames(Collection<OntologyTermI> terms) {
82      if ( terms == null ) {
83        return Collections.emptyList();
84      }
85      Collection<String> names = new ArrayList<>( terms.size() );
86      for ( OntologyTermI term : terms ) {
87        names.add( term.getPreferredName() );
88        names.addAll( term.getNameSynonyms() );
89      }
90      return names;
91    }
