3   import org.biopax.validator.api.beans.Validation;
4   
5   
6   /**
7    * A validation rule.
8    *
9    * @param <T> a type of a model object or a data parser, stream, etc.
10   *
11   * @author rodche
12   */
13  public interface Rule<T> {
14  
15    /**
16     * Validates the object.
17     *
18     * @param validation the object where the model, validation settings and errors are stored
19     * @param thing      to validate
20     */
21    void check(Validation validation, T thing);
22  
23    /**
24     * Can check it?
25     *
26     * @param thing an object to validate
27     * @return True when this rule can check the object.
28     */
29    boolean canCheck(Object thing);
30  
31  
32    /**
33     * Saves the error or warning that occurred or was fixed.
34     * Call this method from a validation rule implementation
35     * every time after a problem is found and/or fixed.
36     *
37     * @param validation the object where the model, validation settings and errors are stored
38     * @param object     that is invalid or caused the error
39     * @param code       error code, e.g., 'illegal.value'
40     * @param setFixed   true/false - whether the issue was auto-fixed or not
41     * @param args       extra parameters for the error message template
42     */
43    void error(Validation validation, Object object, String code, boolean setFixed, Object... args);
44  
45  }
