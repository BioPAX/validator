4   import org.biopax.paxtools.model.level3.Evidence;
5   import org.biopax.paxtools.model.level3.EvidenceCodeVocabulary;
6   import org.biopax.paxtools.model.level3.ExperimentalForm;
7   import org.biopax.paxtools.model.level3.Score;
8   import org.biopax.validator.AbstractRule;
9   import org.biopax.validator.api.beans.Validation;
10  import org.springframework.stereotype.Component;
11  
12  /**
13   * Checks:
14   * Evidence properties and cardinality constraints.
15   * 
16   * This is not a trivial "cardinality and range rule" that
17   * cannot simply extend the CardinalityAndRangeRule class.
18   * 
19   * @author rodche
20   */
21  @Component
22  public class EvidencePropertiesUsageRule extends AbstractRule<Evidence> {
23  
24      public void check(final Validation validation, Evidence ev) {
25  
26  		// check it has AT LEAST one of the things
27  		if ((ev.getEvidenceCode() == null || ev.getEvidenceCode().isEmpty())
28  				&& (ev.getExperimentalForm() == null || ev.getExperimentalForm().isEmpty())
29  				&& (ev.getConfidence() == null || ev.getConfidence().isEmpty())) {
30  			error(validation, ev,
31  					"min.cardinality.violated", false, "'evidenceCode' or 'confidence' or 'experimantalForm'", 1);
32  		} else {
33  
34  			// evidenceCode range
35  			if (ev.getEvidenceCode() != null) {
36  				for (Object cv : ev.getEvidenceCode()) {
37  					if (!EvidenceCodeVocabulary.class.isInstance(cv)) {
38  						error(validation, ev, "range.violated", 
39  								false, "evidenceCode",
40  								cv, cv.getClass().getSimpleName(), "EvidenceCodeVocabulary");
41  					}
42  				}
43  			}
44  
45  			// experimentalForm
46  			if (ev.getExperimentalForm() != null) {
47  				for (Object cv : ev.getExperimentalForm()) {
48  					if (!ExperimentalForm.class.isInstance(cv)) {
49  						error(validation, ev, "range.violated", 
50  								false, "experimentalForm", cv, cv.getClass().getSimpleName(), "ExperimentalForm");
51  					}
52  				}
53  			}
54  
55  			// confidence
56  			if (ev.getConfidence() != null) {
57  				for (Object cv : ev.getConfidence()) {
58  					if (!Score.class.isInstance(cv)) {
59  						error(validation, ev, "range.violated", 
60  								false, "confidence", cv, cv.getClass().getSimpleName(), "Score");
61  					}
62  				}
63  			}
64  		}
65  	}
66  
67  	public boolean canCheck(Object thing) {
68  		return thing instanceof Evidence;
69  	}
70  
71  }
