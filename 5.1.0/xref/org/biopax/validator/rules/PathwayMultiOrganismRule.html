18  /**
19   * Warn if a pathway and its component have different (not null) 'organism' values.
20   * What to do? (ignore, delete this value, or override nested organism properties with pathway's value)
21   *
22   * @author rodche
23   */
24  @Component
25  public class PathwayMultiOrganismRule extends AbstractRule<Pathway> {
26    private final static Filter<PropertyEditor> filter = new Filter<PropertyEditor>() {
27  
28      public boolean filter(PropertyEditor editor) {
29        //skip for: nextStep (those can be reached via pathwayOrder or pathwayComponent) and evidence (multi-org. is normal there)
30        return !("nextStep".equals(editor.getProperty()) || "evidence".equals(editor.getProperty()));
31      }
32    };
33  
34    public void check(final Validation validation, final Pathway pathway) {
35      Fetcher fetcher = new Fetcher(SimpleEditorMap.L3, Fetcher.evidenceFilter, Fetcher.nextStepFilter);
36      fetcher.setSkipSubPathways(true);
37      final Set<BioSource> organisms = fetcher.fetch(pathway, BioSource.class);
38      //collect taxonomy IDs (from the BioSource objects that have valid xrefs)
39      PathAccessor accessor = new PathAccessor("BioSource/xref:UnificationXref/id");
40      Set<String> taxIds = accessor.getValueFromBeans(organisms);
41      if (taxIds.size() > 1) {
42        accessor = new PathAccessor("BioSource/name");
43        Set<String> names = new HashSet<String>();
44        for(Object name  : accessor.getValueFromBeans(organisms))
45         names.add(String.valueOf(name).toLowerCase());
46  
47        error(validation, pathway, "multi.organism.pathway", false, taxIds, names);
48      }
49    }
50  
51    public boolean canCheck(Object thing) {
52      return thing instanceof Pathway;
53    }
54  
55  }
