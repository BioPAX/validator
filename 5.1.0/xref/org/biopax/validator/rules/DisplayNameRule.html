7   import org.biopax.validator.api.beans.Validation;
8   import org.springframework.stereotype.Component;

10  /**
11   * displayName length shouldn't exceed the specified limit.
12   * @author rodche
13   */
14  @Component
15  public class DisplayNameRule extends AbstractRule<Named> {
16      public static final int MAX_DISPLAYNAME_LEN = 25;
17  
18  	public boolean canCheck(Object thing) {
19  		return (thing instanceof Named); 
20  	}
21      
22      public void check(final Validation validation, Named named) 
23      {
24      	boolean fixed = false;
25      	
26      	if (named.getDisplayName() == null) {
27      		if(validation.isFix()) {
28      			// use the standardName if present
29  				if (named.getStandardName() != null) {
30  					named.setDisplayName(named.getStandardName());
31  					fixed = true;
32  				} // otherwise, use the shortest name, if anything...
33  				else if (!named.getName().isEmpty()) {
34  					String dsp = named.getName().iterator().next();
35  					for (String name : named.getName()) {
36  						if (name.length() < dsp.length())
37  							dsp = name;
38  					}
39  					named.setDisplayName(dsp);
40  					fixed = true;
41  				}
42  			}
43      		// report
44  			error(validation, named, "no.display.name", fixed && validation.isFix());
45  		} 
46      	
47      	// check max. length
48      	String name = named.getDisplayName();
49      	if (name != null) { // if existed or was added above
50          	Integer max = (named instanceof Provenance) ? 50 : MAX_DISPLAYNAME_LEN;
51          	if (name.length() > max)
52  				error(validation, named, "too.long.display.name", false
53  					, name 
54  						+ ((fixed) ? "(auto-created form other names!)" : ""), name.length(), max);
55  		}
56      }
57  
58  }
