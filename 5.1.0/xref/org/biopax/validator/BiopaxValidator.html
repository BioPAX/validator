74  	public void validate(final Validation validation) {
75  		assert(validation != null);
76  
77  		if (validation == null || validation.getModel() == null) {
78  			throw new ValidatorException(
79  				"Failed: no BioPAX model to validate " +
80  					"(have you successfully imported or created one already?)");
81  		}
82  
83  		// register the validation (if not done already)
84  		if (!getResults().contains(validation)) {
85  			getResults().add(validation);
86  		}
87  
88  
89  		// break if max.errors exceeded (- reported by AOP interceptors, while parsing a file, or - in previous runs)
90  		if (validation.isMaxErrorsSet()
91  			&& validation.getNotFixedErrors() > validation.getMaxErrors()) {
92  			log.info("Errors limit (" + validation.getMaxErrors() + ") is exceeded; exitting...");
93  			return;
94  		}
95  
96  		Model model = (Model) validation.getModel();
97  
98  		log.debug("validating model: " + model + " that has "
99  			+ model.getObjects().size() + " objects");
100 
101 		if(model.getLevel() != BioPAXLevel.L3) {
102 			model = (new LevelUpgrader()).filter(model);
103 			validation.setModel(model);
104 			log.info("Upgraded to BioPAX Level3 model: " + validation.getDescription());
105 		}
106 
107 		assert(model != null && model.getLevel() == BioPAXLevel.L3);
108 
109 		// Check/fix Rule<? extends BioPAXElement> rules concurrently (low risk of getting CMEx), 
110 		// because they normally do minor changes and simply cannot add/remove
111 		// elements in the Model (though, can alter a property of any biopax object)
112 		ExecutorService exec = Executors.newFixedThreadPool(100);
113 
114 		// First, check/fix individual objects
115 		// (no need to copy; these rules cannot add/remove objects in model)
116 		for (BioPAXElement el : model.getObjects())
117 		{
118 			// rules can check/fix specific elements
119 //			for (Rule rule : rules) {				
120 //				Behavior behavior = utils.getRuleBehavior(rule.getClass().getName(), validation.getProfile());    	
121 //		        if (behavior == Behavior.IGNORE) 
122 //		        	continue; // skip					
123 //				execute(exec, rule, validation, (Object) el);
124 //			}
125 			//sequentially apply all (capable,enabled) rules to the object in a separate thread
126 			execute(exec, rules, validation, (Object) el);
127 		}
128 		exec.shutdown(); //end accepting new jobs
129 		try {
130 			exec.awaitTermination(Long.MAX_VALUE, TimeUnit.SECONDS);
131 		} catch (InterruptedException e) {
132 			throw new ValidatorException("Interrupted unexpectedly!");
133 		}
134 
135 		//Second, apply all Rule<Model> rules -
136 		//run Rule<Model> rules concurrently
137 		exec = Executors.newFixedThreadPool(50);
138 		for (Rule rule : rules)
139 		{
140 			Behavior behavior = utils.getRuleBehavior(rule.getClass().getName(),
141         validation.getProfile());
142 			if (behavior != Behavior.IGNORE)
143 				execute(exec, rule, validation, model);
144 		}
145 		exec.shutdown(); //end accepting jobs
146 		try {
147 			exec.awaitTermination(Long.MAX_VALUE, TimeUnit.SECONDS);
148 		} catch (InterruptedException e) {
149 			throw new ValidatorException("Interrupted unexpectedly!");
150 		}
151 
152 		log.debug("All rules checked!");
153 
154 		if (validation.isFix()) {
155 			// discover, explicitly add child elements to the model
156 			model.repair();
157 			// remove all dangling utility class objects
158 			ModelUtils.removeObjectsIfDangling(model, UtilityClass.class);
159 		}
160 
161 		// add comments and some statistics
162 		validation.addComment("number of interactions : "
163 			+ model.getObjects(Interaction.class).size());
164 		validation.addComment("number of physical entities : "
165 			+ model.getObjects(PhysicalEntity.class).size());
166 		validation.addComment("number of genes : "
167 			+ model.getObjects(Gene.class).size());
168 		validation.addComment("number of pathways : "
169 			+ model.getObjects(Pathway.class).size());
170 
171 		//update all error counts (total, fixed, notfixed)
172 		for(ErrorType errorType : validation.getError()) {
173 			errorType.setTotalCases(errorType.countErrors(null, null, false));
174 			errorType.setNotFixedCases(errorType.countErrors(null, null, true));
175 		}
176 		validation.setNotFixedProblems(validation.countErrors(null, null, null, null, false, true));
177 		validation.setNotFixedErrors(validation.countErrors(null, null, null, null, true, true));
178 		validation.setTotalProblemsFound(validation.countErrors(null, null, null, null, false, false));
179 		validation.setSummary("different types of problem: " + validation.getError().size());
180 	}

182 	private void execute(ExecutorService exec, final Rule rule,
183 											 final Validation validation, final Object obj)
184 	{
185 		exec.execute(new Runnable() {
186 			@SuppressWarnings("unchecked") //obj can be either Model or a BPE
187 			public void run() {
188 				try {
189 					if (rule.canCheck(obj))
190 						rule.check(validation, obj);
191 				} catch (Throwable t) {
192 					//if we're here, there is probably a bug in the rule or validator!
193 					String id = validation.identify(obj);
194 					log.fatal(rule + ".check(" + id
195 						+ ") threw the exception: " + t.toString(), t);
196 					// anyway, report it almost normally (for a user to see this in the results too)
197 					validation.addError(utils.createError(id, "exception",
198 						rule.getClass().getName(), null, false, t));
199 				}
200 			}
201 		});
202 	}

204 	private void execute(ExecutorService exec, final Set<Rule<?>> rules,
205 											 final Validation validation, final Object obj)
206 	{
207 		exec.execute(new Runnable() {
208 			@SuppressWarnings("unchecked") //obj can be either Model or a BPE
209 			public void run() {
210 				for(Rule rule : rules) {
211 					Behavior behavior = utils.getRuleBehavior(rule.getClass().getName(),
212             validation.getProfile());
213 					if (behavior == Behavior.IGNORE)
214 						continue; // skip disabled rule
215 
216 					try {
217 						if (rule.canCheck(obj))
218 							rule.check(validation, obj);
219 					} catch (Throwable t) {
220 						//if we're here, there is probably a bug in the rule or validator!
221 						String id = validation.identify(obj);
222 						log.fatal(rule + ".check(" + id
223 							+ ") threw the exception: " + t.toString(), t);
224 						// anyway, report it almost normally (for a user to see this in the results too)
225 						validation.addError(utils.createError(id, "exception",
226 							rule.getClass().getName(), null, false, t));
227 					}
228 				}
229 			}
230 		});
231 	}

234 	public void importModel(Validation validation, InputStream inputStream) {
235 		// add the parser
236 		SimpleIOHandler simpleReader = new SimpleIOHandler();
237 		simpleReader.mergeDuplicates(true);
238 		associate(inputStream, validation);
239 		associate(simpleReader, validation);
240 		/*
241 		 * build a model and associate it with the validation (for post-validation later on);
242 		 * during this, many errors/warnings may be caught and reported via AOP ;))
243 		 */
244 		Model model = simpleReader.convertFromOWL(inputStream);
245 
246 		if(model == null)
247 			throw new ValidatorException("Failed importing a BioPAX model!");
248 
249 		associate(model, validation);
250 	}

252 	public void associate(Object obj, Validation validation) {
253 		assert(validation != null);
254 
255 		if (!getResults().contains(validation)) {
256 			if(validation != null)
257 				getResults().add(validation); // registered a new one
258 			else
259 				log.warn("Object " + obj +
260 					" is being associated with NULL (Validation)!");
261 		}
262 
263 		if(obj instanceof Model) {
264 			validation.setModel(obj);
265 		} else if(obj != null) {
266 			validation.getObjects().add(obj);
267 		}
268 	}

271 	public Collection<Validation> findValidation(Object o)
272 	{
273 		// add forcedly associated keys
274 		Collection<Validation> keys = new HashSet<Validation>();
275 
276 		if(o == null
277 			|| o.getClass().isPrimitive()
278 			|| o instanceof String) {
279 			return keys;
280 		}
281 
282 		for(Validation r: results) {
283 			if (r.getObjects().contains(o)
284 				|| o.equals(r.getModel()))
285 			{
286 				keys.add(r);
287 			}
288 			else if (o instanceof BioPAXElement)
289 			{
290 				// associate using member models
291 				BioPAXElement bpe = (BioPAXElement) o;
292 				Model m = (Model) r.getModel();
293 				if (m != null && m.contains(bpe)) {
294 					keys.add(r);
295 				}
296 			}
297 		}
298 
299 		if (keys.isEmpty())
300 			log.debug("findKey: no result keys found "
301 				+ "for the object : " + o);
302 
303 		return keys;
304 	}

306 	public void indirectlyAssociate(Object parent, Object child) {
307 		if (parent == null || child==null
308 			|| child.getClass().isPrimitive()
309 			|| child instanceof String) {
310 			return;
311 		}
312 
313 		for (Validation key : findValidation(parent)) {
314 			associate(child, key);
315 		}
316 	}

318 	/**
319 	 * {@inheritDoc}
320 	 *
321 	 * Saves or updates an error/warning case, found by an external tool, in all
322 	 * {@link Validation} tasks known to this validator instance. 
323 	 * This method is used by, e.g., AOP interceptors, which sense, catch, and
324 	 * register problems that occur in external modules, such as Paxtools,
325 	 * during the data read and/or modify.
326 	 *
327 	 */
328 	public void report(Object obj, String errorCode, String reportedBy, boolean isFixed, Object... args)
329 	{
330 		if(obj == null) {
331 			log.error("Attempted to registed an error " +
332 				errorCode + " by " + reportedBy + " with NULL object");
333 			return;
334 		}
335 
336 
337 		Collection<Validation> validations = findValidation(obj);
338 		if(validations.isEmpty()) {
339 			// the object is not associated neither with parser nor model
340 			log.warn("No validations are associated with the object: "
341 				+ obj + "; user won't receive this message: " + errorCode
342 				+ "(fixed=" + isFixed + ") by " + reportedBy + "; " + args );
343 		}
344 
345 		// add to the corresponding validation result
346 		for(Validation v: validations) {
347 			ErrorType err = utils.createError(v.identify(obj),
348 				errorCode, reportedBy, v.getProfile(), isFixed, args);
349 
350 			// add or update: if there was the same type error,
351 			// this will update/add to existing error cases (unique type-object-reporter combinations) 
352 			v.addError(err);
353 
354 			log.debug(v.getDescription() + " - added/updated " + err + " for " + obj + " as: " + isFixed);
355 		}
