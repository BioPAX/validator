68    /**
69     * JSP pages and RESTful web services controller, the main one that checks BioPAX data.
70     * All parameter names are important, i.e., these are part of public API (for clients)
71     *
72     * System parameters:
73     * @param request the web request object (may contain multi-part data, i.e., multiple files uploaded)
74     * @param response
75     * @param mvcModel Spring MVC Model
76     * @param writer HTTP response writer
77     *
78     * Validator parameters:
79     * @param url
80     * @param retDesired
81     * @param autofix
82     * @param filter
83     * @param maxErrors
84     * @param profile
85     *
86     * Normalizer parameters:
87     * @param normalizer binds to view options: normalizer.fixDisplayName,
88     *                   normalizer.inferPropertyOrganism, normalizer.inferPropertyDataSource
89     * @return results view name (or null if XML or normalized RDF/XML were requested)
90     * @throws IOException when data cannot be read from the files or URL, etc.
91     */
92    @RequestMapping(value={"check","check.html"}, method=RequestMethod.POST)
93    public String check(HttpServletRequest request, HttpServletResponse response,
94                        Model mvcModel, Writer writer,
95      @RequestParam(required=false) String url,
96      @RequestParam(required=false) String retDesired,
97      @RequestParam(required=false) Boolean autofix,
98      @RequestParam(required=false) Behavior filter,
99      @RequestParam(required=false) Integer maxErrors,
100     @RequestParam(required=false) String profile,
101     //normalizer!=null when called from the JSP;
102     //but it's usually null when from the validator-client or a web script
103     @ModelAttribute("normalizer") Normalizer normalizer) throws IOException
104   {
105     Resource resource; //to validate
106     final int lim = (maxErrors != null)? maxErrors.intValue() : 0; //0->no error limit
107     final boolean fix = Boolean.TRUE.equals(autofix);
108 
109     // create the response container
110     ValidatorResponse validatorResponse = new ValidatorResponse();
111 
112     if(url != null && url.length()>0) {
113       log.info("url : " + url);
114 
115       try {
116         resource = new UrlResource(url);
117       } catch (MalformedURLException e) {
118         return errorView(mvcModel, e.toString());
119       }
120 
121       try {
122         Validation v = service.validate(resource, lim, fix, filter, profile, normalizer);
123         validatorResponse.addValidationResult(v);
124       } catch (Exception e) {
125         return errorView(mvcModel, e.toString());
126       }
127 
128     } else if (request instanceof MultipartHttpServletRequest) {
129       MultipartHttpServletRequest multiRequest = (MultipartHttpServletRequest) request;
130       Map files = multiRequest.getFileMap();
131       Assert.state(!files.isEmpty(), "No files to validate");
132       for (Object o : files.values()) {
133         MultipartFile file = (MultipartFile) o;
134         String filename = file.getOriginalFilename();
135         // a workaround (for some reason there is always a no-name-file;
136         // this might be a javascript isue)
137         if(file.getBytes().length==0 || filename==null || "".equals(filename))
138           continue;
139 
140         log.info("check : " + filename);
141         resource = new ByteArrayResource(file.getBytes(), filename);
142         try {
143           Validation v = service.validate(resource, lim, fix, filter, profile, normalizer);
144           validatorResponse.addValidationResult(v);
145         } catch (Exception e) {
146           return errorView(mvcModel, e.toString());
147         }
148       }
149     } else {
150       return errorView(mvcModel, "No BioPAX input source provided!");
151     }
152 
153     if("xml".equalsIgnoreCase(retDesired)) {
154       response.setContentType("application/xml");
155       ValidatorUtils.write(validatorResponse, writer, null);
156     } else if("html".equalsIgnoreCase(retDesired)) {
157     		/* could also use ValidatorUtils.write with a xml-to-html xslt source
158     		 but using JSP here makes it easier to keep the same style, header, footer*/
159       mvcModel.addAttribute("response", validatorResponse);
160       return "groupByCodeResponse";
161     } else { //the fixed/normalized OWL (BioPAX RDF/XML) was requested
162       response.setContentType("text/plain");
163       // write all the models one after another (RDF spec. allows that,
164       // despite Paxtools might not be able to parse it)
165       for(Validation result : validatorResponse.getValidationResult()) {
166         if(result.getModelData() != null)
167           writer.write(result.getModelData() + NEWLINE);
168         else // write "empty" rdf
169           writer.write("<rdf:RDF></rdf:RDF>" + NEWLINE);
170       }
171     }
172     //no view as result/error was written to the response stream
173     return null;
174   }
175 
176   private String errorView(Model model, String msg) {
177     model.addAttribute("error", msg);
178     return null;
179   }
