1   package org.biopax.validator.web.controller;
2   
3   import org.biopax.validator.web.dto.Clue;
4   import org.biopax.validator.web.service.Suggester;
5   import org.biopax.validator.web.dto.Xref;
6   import org.springframework.beans.factory.annotation.Autowired;
7   import org.springframework.web.bind.annotation.*;
8   
9   import javax.servlet.http.HttpServletResponse;
10  import java.io.IOException;
11  
12  import static org.springframework.http.MediaType.APPLICATION_JSON_VALUE;
13  
14  @RestController
15  public class SuggesterController {
16  
17    @Autowired
18    private Suggester service;
19  
20    @GetMapping(value = "/xref/{db}/{id}/", produces = APPLICATION_JSON_VALUE)
21    public String xrefDbId(@PathVariable String db, @PathVariable String id,
22                            HttpServletResponse response) throws IOException {
23      String uri = null;
24      try {
25        uri = service.xrefDbIdToUri(db, id);
26      } catch (IllegalArgumentException e) {
27        response.sendError(400, e.toString());
28      }
29  
30      return uri;
31    }
32  
33    @PostMapping(value = "/xref", produces = APPLICATION_JSON_VALUE)
34    public Clue xref(@RequestBody Xref[] xrefs, HttpServletResponse response)
35      throws IOException {
36  
37      try {
38        return service.xref(xrefs);
39      } catch (IllegalArgumentException e) {
40        response.sendError(400, e.toString());
41      }
42  
43      return null;
44    }
45  
46  }
