18  import java.io.IOException;
19  
20  @Service
21  public class ValidatorService implements ValidatorAdapter {
22    private final static DefaultResourceLoader LOADER = new DefaultResourceLoader();
23  
24    private Validator biopaxValidator; //to inject the biopax-validator
25    private final String schema;
26  
27    @Autowired
28    public ValidatorService(Validator biopaxValidator) throws IOException {
29      this.biopaxValidator = biopaxValidator;
30      this.schema = new String(FileCopyUtils.copyToByteArray(LOADER
31        .getResource("classpath:validator-response-2.0.xsd").getInputStream()),"UTF-8");
32    }
33  
34    public Validation validate(Resource data,
35                               int errMax, boolean isFix,
36                               Behavior errorLevel, String profile,
37                               Normalizer normalizer) throws IOException
38    {
39      Validation validationResult = new Validation(new BiopaxIdentifier(),
40        data.getDescription(), isFix, errorLevel, errMax, profile);
41      //run the biopax-validator (this updates the validationResult object)
42      biopaxValidator.importModel(validationResult, data.getInputStream());
43      biopaxValidator.validate(validationResult);
44      biopaxValidator.getResults().remove(validationResult);
45  
46      if(isFix) { // do normalize too
47        if(normalizer == null) //e.g., when '/check' called from a client/script, not JSP
48          normalizer = new Normalizer();
49  
50        Model m = (Model) validationResult.getModel();
51        normalizer.normalize(m);//this further modifies the validated and auto-fixed model
52        //update the serialized model (BioPAX RDF/XML)
53        //for the client to get it (to possibly, unmarshall)
54        validationResult.setModel(m);
55        validationResult.setModelData(SimpleIOHandler.convertToOwl(m));
56      } else {
57        validationResult.setModelData(null);
58        validationResult.setModel(null);
59      }
60  
61      return validationResult;
62    }
63  
64    public String getSchema() {
65      return schema;
66    }
67  
68  }
