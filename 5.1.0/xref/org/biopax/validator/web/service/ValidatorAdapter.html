10  public interface ValidatorAdapter {
11    /**
12     * Parses and checks BioPAX data from the given resource
13     * and returns the validation report.
14     *
15     * @param data input biopax model source
16     * @param maxErrors optional, if greater than 0, abort once the number of critical errors exceeds the threshold
17     * @param fixErrors optional, when true, some validator rules can auto-fix the model (issues still get reported)
18     * @param level optional, if 'ERROR' then warnings are ignored; 'WARNING' - both errors and warnings are reported.
19     * @param profile optional, validation rules' behavior settings profile, e.g., 'notstrict', 'default'
20     * @param normalizer optional, pre-configured biopax normalizer
21     * @return validation report
22     * @throws IOException when fails to read the data
23     */
24    Validation validate(Resource data, int maxErrors, boolean fixErrors,
25                        Behavior level, String profile,
26                        Normalizer normalizer) throws IOException;
27  
28    /**
29     * @return Validator results XML schema.
30     */
31    String getSchema();
32  }
