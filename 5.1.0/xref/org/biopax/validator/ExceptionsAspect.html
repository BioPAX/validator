22  /**
23   * This is the central aspect to report

42  	/**
43  	 * This captures the exceptions that occur
44  	 * during the model build and, more important,
45  	 * associates the just created model
46  	 * with the corresponding validation result
47  	 * (this is the earliest possibility to do so)!
48  	 *
49  	 * @param jp (AspectJ) joint point
50  	 * @param model biopax model
51  	 */
52  	@Around("execution(void org.biopax.paxtools.io.SimpleIOHandler+.createAndBind(*)) " +
53  		"&& args(model)")
54  	public Object adviseCreateAndBind(ProceedingJoinPoint jp, Model model) {
55  		SimpleIOHandler reader = (SimpleIOHandler) jp.getTarget();
56  		// associate the model with the reader (and validation results)
57  		getValidator().indirectlyAssociate(reader, model);
58  
59  		Object ret = null;
60  		try {
61  			ret = jp.proceed();
62  		} catch (Throwable ex) {
63  			reportException(ex, reader, "syntax.error", "SimpleIOHandler.createAndBind interceptor", null);
64  		}
65  
66  		return ret;
67  	}

83  	@Around("execution(private void org.biopax.paxtools.io.SimpleIOHandler+.bindValue(..))" +
84  		" && args(triple, model)")
85  	public Object adviseBindValue(ProceedingJoinPoint jp, Triple triple, Model model) {
86  		if(log.isDebugEnabled())
87  			log.debug("adviseBindValue, triple: " + triple);
88  
89  		SimpleIOHandler reader = (SimpleIOHandler) jp.getTarget();
90  
91  		// try to find the best object to report about...
92  		if(triple.domain==null || triple.property==null) {
93  			report(reader, "syntax.error",
94  				"SimpleIOHandler.bindValue interceptor", false, triple +	" - skipped");
95  		} else {
96  			BioPAXElement el = model.getByID(triple.domain);
97  			if (el != null) {
98  				PropertyEditor<?, ?> editor = reader.getEditorMap()
99  					.getEditorForProperty(triple.property, el.getModelInterface());
100 				if (editor == null) {
101 					// auto-fix (for some)
102 					if (triple.property != null && triple.property.equals("taxonXref")) {
103 						report(el, "unknown.property",
104 							"SimpleIOHandler.bindValue interceptor",
105 							true, triple.property + " - replaced with 'xref'");
106 						triple.property = "xref";
107 					} else {
108 						report(el, "unknown.property",
109 							"SimpleIOHandler.bindValue interceptor",
110 							false, triple.property + " - skipped");
111 					}
112 				}
113 			}
114 		}
115 
116     Object ret = null;
117 		try {
118 			ret = jp.proceed();
119 		} catch (Throwable t) {
120 			reportException(t, reader, "syntax.error", "SimpleIOHandler.bindValue interceptor",
121         String.valueOf(triple));
122 		}
123 
124 		return ret;
125 	}
