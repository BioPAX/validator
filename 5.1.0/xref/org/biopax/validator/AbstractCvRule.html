1   package org.biopax.validator;
2   
3   import org.biopax.paxtools.controller.PropertyEditor;
4   import org.biopax.paxtools.model.BioPAXElement;
5   import org.biopax.validator.api.CvRestriction;
6   import org.biopax.validator.api.CvRule;
7   import org.biopax.validator.api.CvUtils;
8   import org.springframework.beans.factory.annotation.Autowired;
9   
10  import javax.annotation.PostConstruct;
11  import java.util.HashSet;
12  import java.util.Set;
13  
14  /**
15   * An abstract class for CV terms checks.
16   *
17   * @param <D> property domain
18   * @author rodche
19   */
20  public abstract class AbstractCvRule<D extends BioPAXElement> extends AbstractRule<D> implements CvRule<D> {
21  
22    @Autowired
23    protected CvUtils ontologyUtils;
24  
25    protected final Class<D> domain;
26    protected final String property; // helps validate generic ControlledVocabulary instances
27    protected final Set<CvRestriction> restrictions;
28    private Set<String> validTerms;
29    protected PropertyEditor<? super D, ?> editor;
30  
31    /**
32     * Constructor.
33     *
34     * @param domain       a BioPAX class for which the CV terms restrictions apply
35     * @param property     the name of the BioPAX property to get controlled vocabularies or null
36     * @param restrictions a list of beans, each defining names (a subtree of an ontology) that
37     *                     is either to include or exclude (when 'not' flag is set) from the valid names set.
38     */
39    public AbstractCvRule(Class<D> domain, String property, CvRestriction... restrictions) {
40      this.domain = domain;
41      this.property = property;
42      this.restrictions = new HashSet<>(restrictions.length);
43      for (CvRestriction c : restrictions) {
44        this.restrictions.add(c);
45      }
46    }
47  
48    @PostConstruct
49    public void init() {
50      if (ontologyUtils != null) {
51        setValidTerms(ontologyUtils.getValidTermNames(this));
52      } else {
53        throw new IllegalStateException("ontologyUtils is NULL!");
54      }
55    }
56  
57    public boolean canCheck(Object thing) {
58      return domain.isInstance(thing);
59    }
60  
61    /* (non-Javadoc)
62     * @see org.biopax.validator.impl.CvRule#getValidTerms()
63     */
64    public Set<String> getValidTerms() {
65      return validTerms;
66    }
67  
68    /* (non-Javadoc)
69     * @see org.biopax.validator.impl.CvRule#setValidTerms(java.util.Set)
70     */
71    public void setValidTerms(Set<String> validTerms) {
72      this.validTerms = validTerms;
73    }
74  
75    // for unit testing
76  
77    /* (non-Javadoc)
78     * @see org.biopax.validator.impl.CvRule#getRestrictions()
79     */
80    public Set<CvRestriction> getRestrictions() {
81      return restrictions;
82    }
83  
84    /* (non-Javadoc)
85     * @see org.biopax.validator.impl.CvRule#getDomain()
86     */
87    public Class<D> getDomain() {
88      return domain;
89    }
90  
91    /* (non-Javadoc)
92     * @see org.biopax.validator.impl.CvRule#getProperty()
93     */
94    public String getProperty() {
95      return property;
96    }
97  
98    /**
99     * Gets the internal BiopaxOntologyManager instance
100    *
101    * @return ontology manager
102    */
103   public CvUtils getBiopaxOntologyManager() {
104     return ontologyUtils;
105   }
106 
107   /**
108    * Gets the corresponding CV property editor.
109    * Returns null if either the 'domain' itself is of CV type
110    * or the 'property' is null.
111    *
112    * @return biopax property editor
113    */
114   public PropertyEditor<? super D, ?> getEditor() {
115     return editor;
116   }
117 
118   /**
119    * OntologyAccess IDs used to check this CV rule.
120    * These can be extracted from the CV restrictions
121    * used to define the rule.
122    * (other ontologies are not used).
123    *
124    * @return ontology IDs, such as e.g. 'MI', 'GO'
125    */
126   protected Set<String> getOntologyIDs() {
127     Set<String> ids = new HashSet<>();
128     for (CvRestriction restriction : restrictions)
129       ids.add(restriction.getOntologyId());
130     return ids;
131   }
132 
133 }
