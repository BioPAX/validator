32  	@Test
33  	public void testClientXml() throws IOException, JAXBException {
34  		BiopaxValidatorClient client = new BiopaxValidatorClient();
35  		
36  		File[] files = new File[] {
37  				new File(getClass().getResource(
38  					File.separator + "testBiopaxElementIdRule.owl").getFile())
39  		};
40  		
41  		ByteArrayOutputStream baos = new ByteArrayOutputStream();
42  		client.validate(true, null, RetFormat.XML, null, null, null, files, baos);
43  		
44  		//System.out.println(baos.toString());
45  		
46  		Assert.assertTrue(baos.size()>0);
47  		
48  		ValidatorResponse resp = client.unmarshal(baos.toString());
49  		
50  		Assert.assertNotNull(resp);
51  		Assert.assertFalse(resp.getValidation().isEmpty());
52  		
53  		System.out.println(resp.getValidation().get(0).getSummary()
54  				+ "; cases: " + resp.getValidation().get(0).getTotalProblemsFound());
55      }
