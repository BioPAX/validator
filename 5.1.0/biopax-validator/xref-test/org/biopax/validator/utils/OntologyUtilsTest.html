28    @BeforeClass
29    public static void init() throws IOException {
30      OntologyUtils utils = new OntologyUtils();
31      utils.setOntologyConfig(PropertiesLoaderUtils.loadProperties(new ClassPathResource("test-obo.properties")));
32      utils.init();
33      cvUtils = utils;
34      cvFactory = utils;
35      xrefUtils = utils;
36    }
37  
38    @Test
39    public void xcheck() {
40      assertTrue(xrefUtils.xcheck());
41    }
42  
43    @Test
44    public void ontologyLoading() {
45      Collection<String> ontologyIDs = cvUtils.getOntologyManager().getOntologyIDs();
46      assertTrue(ontologyIDs.contains("GO"));
47      assertEquals("gene ontology", cvUtils
48        .getOntologyManager().getOntology("GO").getName().toLowerCase());
49      assertTrue(ontologyIDs.contains("MOD"));
50    }
51  
52    @Test
53    public void testGetDirectChildren() {
54      Set<String> dc = cvFactory.getDirectChildren("urn:miriam:obo.go:GO%3A0005654");
55      assertFalse(dc.isEmpty());
56      assertTrue(dc.contains("http://identifiers.org/go/GO:0044451"));
57    }
58  
59    @Test
60    public void testGetAllChildren() {
61      Set<String> dc = cvFactory.getAllChildren("http://identifiers.org/obo.go/GO:0005654");
62      assertFalse(dc.isEmpty());
63      assertTrue(dc.contains("http://identifiers.org/go/GO:0044451"));
64      assertTrue(dc.contains("http://identifiers.org/go/GO:0071821"));
65      assertTrue(dc.contains("http://identifiers.org/go/GO:0070847"));
66    }
67  
68    @Test
69    public void testGetDirectParents() {
70      Set<String> dc = cvFactory.getDirectParents("urn:miriam:obo.go:GO%3A0005654");
71      assertFalse(dc.isEmpty());
72      assertTrue(dc.contains("http://identifiers.org/go/GO:0031981"));
73    }
74  
75    @Test
76    public void testGetAllParents() {
77      Set<String> dc = cvFactory.getAllParents("http://identifiers.org/obo.go/GO:0005654");
78      assertFalse(dc.isEmpty());
79      assertTrue(dc.contains("http://identifiers.org/go/GO:0031981"));
80      assertTrue(dc.contains("http://identifiers.org/go/GO:0044428"));
81      assertTrue(dc.contains("http://identifiers.org/go/GO:0044422"));
82    }

84    @Test // using correct ID(s)
85    public void testGetObject() {
86      CellularLocationVocabulary cv = cvFactory.getControlledVocabulary("urn:miriam:obo.go:GO%3A0005737", CellularLocationVocabulary.class,"");
87      assertNotNull(cv);
88      cv = cvFactory.getControlledVocabulary( //using now deprecated URL
89        "http://identifiers.org/obo.go/GO:0005737",CellularLocationVocabulary.class,"");
90      assertNotNull(cv);
91      //same
92      cv = cvFactory.getControlledVocabulary(
93        "http://identifiers.org/go/GO:0005737",CellularLocationVocabulary.class,"");
94      assertNotNull(cv);
95    }

97    @Test // using bad ID (with 'X' in it)

100       "urn:miriam:obo.go:GO%3A0005737X",CellularLocationVocabulary.class,"");
