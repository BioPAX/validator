18  /**
19   * This Is an Integration Test.
20   *
21   * Moreover, this here uses the 'production' application context,
22   * i.e., enabling all the AOP aspects and Load-Time Weaving (LTW)! 
23   * So, LTW is the one that sometimes makes things complicated...
24   *
25   * NOTE: 
26   *  Surprisingly, but the (sad) fact here is that, 
27   *  when using @DirtiesContext annotation with a test method,
28   *  it reloads the context, but, in fact, not everything.
29   *  E.g., ControlAspect references not the same 'biopaxValidator' bean
30   *  instance as the autowired one here... 
31   *
32   *  So, we do not want to use '@DirtiesContext' in this test class...
33   *
34   * @author rodche
35   */
36  //@Ignore

39  	"/META-INF/spring/appContext-loadTimeWeaving.xml",

43  	@Autowired
44  	private Validator biopaxValidator;

46  	@Autowired
47  	private ApplicationContext context;

49  	@Test
50  	public void testValidator() throws IOException {
51  		Resource resource = context
52  			.getResource("classpath:biopax3-short-metabolic-pathway.owl");
53  		Validation result = new Validation(new BiopaxIdentifier());
54  		result.setDescription(resource.getDescription());
55  		biopaxValidator.importModel(result, resource.getInputStream());
56  		biopaxValidator.validate(result); // check all rules
57  		biopaxValidator.getResults().clear(); // clean after itself
58  
59  		assertFalse(result.getError().isEmpty());
60  	}

62  	@Test
63  	public void testUnknownProperty() {
64  		Validation validation = new Validation(new BiopaxIdentifier());
65  		biopaxValidator.importModel(validation, getClass()
66  			.getResourceAsStream("testSyntaxErrors.xml"));
67  		biopaxValidator.getResults().clear(); // clean after itself
68  		assertEquals(1, validation.countErrors(null, null, "unknown.property",
69  			null, false, false));
70  		assertEquals(0, validation.countErrors(null, null, "syntax.error",
71  			null, false, false));
72  	}
