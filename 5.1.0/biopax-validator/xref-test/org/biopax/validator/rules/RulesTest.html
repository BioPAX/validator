37    @Test
38    public void testBiochemReactParticipantsLocationRule() {
39      Rule<BiochemicalReaction> rule = new BiochemReactParticipantsLocationRule();
40      BiochemicalReaction reaction = level3.create(BiochemicalReaction.class, "BiochemicalReaction");
41      Dna left = level3.create(Dna.class, "dna");
42      Dna right = level3.create(Dna.class, "modifiedDna");
43      EntityFeature feature = level3.create(FragmentFeature.class, "feature1");
44      right.addFeature(feature);
45      right.addComment("modified dna");
46  
47      DnaReference dnaReference = level3.create(DnaReference.class, "dnaref");
48      // set the same type (entity reference)
49      left.setEntityReference(dnaReference);
50      right.setEntityReference(dnaReference);
51  
52      CellularLocationVocabulary cl = level3.create(CellularLocationVocabulary.class, "cl");
53      CellularLocationVocabulary cr = level3.create(CellularLocationVocabulary.class, "cr");
54      cl.addTerm("cytoplasm");
55      cr.addTerm("membrane");
56      left.addName("dnaLeft");
57      right.addName("dnaRight");
58      reaction.addLeft(left);
59      reaction.addRight(right);
60  
61      // ok
62      Validation v = new Validation(new BiopaxIdentifier());
63      left.setCellularLocation(cl);
64      right.setCellularLocation(cl);
65      rule.check(v, reaction);
66      assertTrue(v.getError().isEmpty());
67  
68      // test complex (rule cannot use entityReference to match "the same" entity)
69      PhysicalEntity leftc = level3.create(Complex.class, "complex");
70      PhysicalEntity rightc = level3.create(Complex.class, "modifiedCmplex");
71      leftc.addName("cplx1");
72      rightc.addName("cplx1");
73      rightc.setCellularLocation(cl);
74      reaction.addLeft(leftc);
75      reaction.addRight(rightc);
76      leftc.setCellularLocation(cr);
77      leftc.addComment("location changed without transport?");
78  
79      v = new Validation(new BiopaxIdentifier());
80      rule.check(v, reaction);
81      assertEquals(1, v.countErrors(reaction.getUri(), null, "participant.location.changed",
82        null, false, false));
83  
84      // different complex, another location is ok for this rule,
85      // but this is another problem (other rule will catch this)
86      // for complexes, not ER but names are used to match...
87      leftc.removeName("cplx1");
88      leftc.addName("cplx2");
89      v = new Validation(new BiopaxIdentifier());
90      rule.check(v, reaction);
91      assertTrue(v.getError().isEmpty());
92  
93      right.setCellularLocation(cr);
94      right.addComment("location changed without transport?");
95      // check for: same entity (names intersection), diff. location
96      v = new Validation(new BiopaxIdentifier());
97      rule.check(v, reaction);
98      assertEquals(1, v.countErrors(reaction.getUri(), null, "participant.location.changed",
99        null, false, false));
100 
101     // generate the example OWL
102     Model m = level3.createModel();
103     m.add(reaction);
104     m.add(left); m.add(right);
105     m.add(dnaReference);
106     m.add(cl); m.add(cr);
107     m.add(feature);
108     m.add(leftc);
109     m.add(rightc);
110     writeExample("testBiochemReactParticipantsLocationRule.owl",m);
111 
112     //same locations is ok
113     leftc.removeName("cplx2");
114     leftc.addName("cplx1");
115     leftc.setCellularLocation(cr);
116     rightc.setCellularLocation(cr);
117     right.setCellularLocation(cr);
118     left.setCellularLocation(cr);
119     v = new Validation(new BiopaxIdentifier());
120     rule.check(v, reaction);
121     assertTrue(v.getError().isEmpty());
122   }

596     Model m = level3.createModel();
597     SimplePhysicalEntity p1 = m.addNew(Protein.class, "protein1");
598     SimplePhysicalEntity p2 = m.addNew(Protein.class, "protein2");
599     EntityReference pr1 = m.addNew(ProteinReference.class, "proteinreference1");
600     EntityReference pr2 = m.addNew(ProteinReference.class, "proteinreference2");
601     EntityFeature ef1 = m.addNew(ModificationFeature.class, "modfeature1");
602 
603     //p1 (a state of pr1) and pr1 do correctly use ef1 so far -
604     p1.setEntityReference(pr1);
605     pr1.addEntityFeature(ef1);
606     p1.addNotFeature(ef1);
607     //ef1 can be used as feature or notFeature of another PEs, but should not belong to other ERs
608     //(unfortunately, current Paxtools allows to do this, though it logs a warning)!
609 
610     //however p2 refers to pr2
611     p2.setEntityReference(pr2);
612     //and pr2 call intentionally (for testing) violates the inverse functional constraint
613     //on property 'entityFeature' and effectively re-sets ef1.entityFeatureOf=pr2 (was pr1 above)
614     //but pr1 still owns the ef1 too (an this is the rule must fix among other things)
615     pr2.addEntityFeature(ef1);
616     //and uses ef1 too
617     p2.addFeature(ef1);
618 
619     //Let's check the model by applying the rule (only one)...
620     Validation v = new Validation(new BiopaxIdentifier());
621     rule.check(v, m);
622     assertEquals(1, v.countErrors(ef1.getUri(), null, "inverse.functional.violated", null, false, false));
623     writeExample("testEntityFeatureInverseFunctionalRule.owl", m);
624 
625     //Now, let's apply the same rule but tell the validator to fix the model...
626     //expected result: in the modified model, a new feature must be created
627     //that replaces ef1 in the corresponding properties of p1 and pr1;
628     //whereas p2 and pr2 should stay untouched
629     v = new Validation(new BiopaxIdentifier(),"", true, null, 0, null);
630     rule.check(v, m);
631     assertEquals(1, v.countErrors(ef1.getUri(), null, "inverse.functional.violated", null, false, false));
632     assertEquals(0, v.countErrors(ef1.getUri(), null, "inverse.functional.violated", null, false, true));
633     writeExample("testEntityFeatureInverseFunctionalRuleFixed.owl", m);
634   }
