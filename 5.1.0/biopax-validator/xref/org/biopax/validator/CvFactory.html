5   import java.util.Set;

7   public interface CvFactory {
8     /**
9      * Gets a normalized BioPAX Controlled Vocabulary by standard (MIRIAM EBI) URI.
10     *
11     * @param <T>  BioPAX ControlledVocabulary type or sub-type
12     * @param uri  recommended URI, e.g., urn:miriam:obo.go:GO%3A0005654 or http://identifiers.org/obo.go/GO:0005654
13     * @param cvClass ControlledVocabulary or sub-class
14     * @param xmlBase biopax model xml:base (URI namespace prefix for generated CV objects)
15     * @return controlled vocabulary object
16     */
17    <T extends ControlledVocabulary> T getControlledVocabulary(String uri, Class<T> cvClass, String xmlBase);
18  
19    /**
20     * Lookup for a CV of given class by ontology (name, synonym, or URI)
21     * and accession (ID) or term name/synonym.
22     *
23     * @param <T> BioPAX ControlledVocabulary type or sub-type
24     * @param db  OBO ontology name, synonym, or URI (e.g., "Gene Ontology", "go",
25     *            "urn:miriam:obo.go", or "http://identifiers.org/obo.go/")
26     * @param id  term's accession number (identifier) or name/synonym
27     * @param cvClass ControlledVocabulary or sub-class
28     * @param xmlBase biopax model xml:base (URI namespace prefix for generated CV objects)
29     * @return the controlled vocabulary or null (when no match found or ambiguous)
30     */
31    <T extends ControlledVocabulary> T getControlledVocabulary(String db, String id, Class<T> cvClass, String xmlBase);
32  
33    //controlled vocabulary terms hierarchy methods

37    Set<String> getDirectParents(String urn);
38  
39    Set<String> getAllChildren(String urn);
40  
41    Set<String> getAllParents(String urn);
42  
43    boolean isChild(String parentUrn, String urn);
