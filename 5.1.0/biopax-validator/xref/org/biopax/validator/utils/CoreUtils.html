17   * Validation rules and errors utilities.
18   *
19   * This is injected into other beans, keeps several global settings and objects,
20   * e.g., marshaller, and also provides static service methods to register,
21   * merge, do OXM, and resolve validation errors to human-readable verbose messages.

60    public ErrorType createError(String objectName, String errorCode,
61                                 String ruleName, String profile,
62                                 boolean isFixed, Object... msgArgs)
63    {
64      if (objectName == null) {
65        objectName = "null";
66        logger.warn("Creating an error " + errorCode + " for Null object!");
67      }
68  
69      //get/use current behavior
70      Behavior behavior = getRuleBehavior(ruleName, profile);
71      // new error object
72      ErrorType error = new ErrorType(errorCode, behavior);
73  
74      // build human-friendly messages using default locale and property files (msg sources)
75      String commonMsg = (messageSource != null)
76        ? messageSource.getMessage(errorCode + ".default", new Object[]{}, "No description.", locale)
77        : "No description.";
78      error.setMessage(commonMsg);
79  
80      String msg =
81        (messageSource != null)
82          ? messageSource.getMessage(errorCode, msgArgs, StringUtils.join(msgArgs, "; "), locale)
83              .replaceAll("\r|\n+", " ")
84          : StringUtils.join(msgArgs, "; ");
85  
86      // resolve/set BioPAX problem category
87      String category = (messageSource != null)
88        ? messageSource.getMessage(errorCode + ".category", null, Category.INFORMATION.name(), locale)
89        : null;
90      if (category != null)
91        error.setCategory(Category.valueOf(category.trim().toUpperCase()));
92  
93      // add one error case
94      ErrorCaseType errorCase = new ErrorCaseType(ruleName, objectName, msg);
95      errorCase.setFixed(isFixed); //
96      error.addErrorCase(errorCase);
97  
98      return error;
99    }

101   public Behavior getRuleBehavior(String ruleName, String profile) {
102     if (messageSource == null) return Behavior.ERROR;
103 
104     //using the default locale for values below (those ain't info messages)
105     final Locale defaultLocale = Locale.getDefault();
106 
107     // get the default behavior value first
108     String value = messageSource.getMessage(ruleName + ".behavior", null,
109       "ERROR", defaultLocale);
110 
111     // override from the profile
112     if (profile != null && !profile.isEmpty())
113       value = messageSource.getMessage(ruleName + ".behavior." + profile,
114         null, value,  defaultLocale);
115 
116     return Behavior.valueOf(value.toUpperCase());
117   }
118 
119   public String getRuleDescription(String ruleName) {
120     //the default locale is used to get the ruleName intentionally
121     String tip = messageSource.getMessage(ruleName, null, "", Locale.getDefault());
122     if (tip == null || "".equals(tip)) {
123       tip = "description is not found in the messages.properties file";
124     } else {
125       tip = StringEscapeUtils.escapeHtml4(tip);
126     }
127 
128     return tip;
129   }
