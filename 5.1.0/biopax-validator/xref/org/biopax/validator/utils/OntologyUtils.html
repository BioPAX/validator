51    public void setExtraGroups(Set<List<String>> extraDbSynonyms){
52      // normalize and organize provided synonyms
53      this.extraGroups = new CompositeCollection<>();
54      if (extraDbSynonyms != null) {
55        for (List<String> group : extraDbSynonyms) {
56          Collection<String> newG = new ArrayList<>();
57          for (String s : group)
58            newG.add(dbName(s)); //trim, uppercase, add
59          this.extraGroups.addComposited(newG);
60        }
61      }
62    }

96  	public Set<OntologyTermI> getValidTerms(Collection<CvRestriction> restrictions) {
97  		Set<OntologyTermI> terms = new HashSet<>();
98  		
99  		// first, collect all the valid terms
100 		for(CvRestriction restriction : restrictions) {
101 			if(!restriction.isNot()) {
102 				terms.addAll(getTerms(restriction));
103 			}
104 		}
105 		
106 		// now remove all those where restriction 'not' property set to true
107 		for(CvRestriction restriction : restrictions) {
108 			if(restriction.isNot()) {
109 				terms.removeAll(getTerms(restriction));
110 			}
111 		}
112 		
113 		return terms;
114 	}

116 	public Set<OntologyTermI> getTerms(CvRestriction restriction) {
117 		Set<OntologyTermI> terms = new HashSet<>();
118 		OntologyAccess ontologyAccess = ontologyManager.getOntology(restriction.getOntologyId());
119 		if(ontologyAccess == null) {
120 			throw new IllegalArgumentException(
121 					"Cannot get access to the ontology using ID: " 
122 					+ restriction.getOntologyId() +
123 					"; I know the following IDs: " 
124 					+ ontologyManager.getOntologyIDs().toString());
125 		}
126 		OntologyTermI term = ontologyAccess.getTermForAccession(restriction.getId());
127 		if(term == null) {
128 			log.error("Cannot Get " + restriction.getOntologyId()
129 					+ " OntologyAccess Term for the Accession: " + restriction.getId());
130 			return terms;
131 		}
132 		
133 		if(restriction.isTermAllowed()) {
134 			terms.add(term);
135 		}
136 		
137 		if (restriction.getChildrenAllowed() == UseChildTerms.ALL) {
138 			terms.addAll(ontologyAccess.getAllChildren(term));
139 		} else if (restriction.getChildrenAllowed() == UseChildTerms.DIRECT) {
140 			terms.addAll(ontologyAccess.getDirectChildren(term));
141 		}
142 		
143 		// FIX xml escape symbols that come from the OntologyManagerImpl (a bug?)
144 		for(OntologyTermI t : terms) {
145 			t.setPreferredName(StringEscapeUtils.unescapeXml(t.getPreferredName()));
146 			Set<String> synonyms = new HashSet<>();
147 			for(String s : t.getNameSynonyms()) {
148 				synonyms.add(StringEscapeUtils.unescapeXml(s));
149 			}
150 			t.setNameSynonyms(synonyms);
151 		}
152 		
153 		return terms;
154 	}

156   public <T extends ControlledVocabulary> T getControlledVocabulary(
157     String db, String id, Class<T> cvClass, String xmlBase)
158   {
159     OntologyAccess ontologyAccess = ontologyManager.getOntology(db);
160     if (ontologyAccess == null) // it may be urn - try again
161       ontologyAccess = getOntologyByUrn(db);
162     OntologyTermI term = ontologyManager.findTerm(ontologyAccess, id);
163 
164     return (term != null) ? getControlledVocabulary(term, cvClass, xmlBase) : null;
165   }

171     T cv = getControlledVocabulary(term, cvClass, xmlBase);

174     return cv;

197   private <T extends ControlledVocabulary> T getControlledVocabulary(
198     OntologyTermI term, Class<T> cvClass, String xmlBase)
199   {
200     if (term == null)
201       return null;
202 
203     BioPAXFactory factory = BioPAXLevel.L3.getDefaultFactory();
204 
205     String urn = ontologyTermToUri(term);
206     T cv = factory.create(cvClass, urn);
207     cv.addTerm(term.getPreferredName());
208 
209     String ontId = term.getOntologyId(); // like "GO"
210     String db = ontologyManager.getOntology(ontId).getName(); // names were fixed in the constructor!
211     String rdfid = Normalizer.uri(xmlBase, db, term.getTermAccession(), UnificationXref.class);
212     UnificationXref uref = factory.create(UnificationXref.class, rdfid);
213     uref.setDb(db);
214     uref.setId(term.getTermAccession());
215     cv.addXref(uref);
216 
217     return cv;
218   }

226       String urn = MiriamLink.getDataTypeURI(id);

235     Set<String> urns = new HashSet<>();

237       urns.add(ontologyTermToUri(term));

242   private String ontologyTermToUri(OntologyTermI term) {
243     String uri = null;
244     if (term != null) {
245       String ontologyID = term.getOntologyId();
246       String accession = term.getTermAccession();
247       uri = MiriamLink.getIdentifiersOrgURI(ontologyID, accession);
248     }
249     return uri;
250   }

252   /*
253    * Post-construct initializer.
254    *
255    * Loads and merges bio database/datasource names and synonyms
256    * from Miriam resource, PSI-MI ontology ("database citation" branch),
257    * and extra (configured by user) names.
258    */
259   @PostConstruct //vital
260   public synchronized void init() {
261     try {
262       //create new ontology manager and load OBO files as specified in the properties.
263       ontologyManager = new OntologyManagerImpl();
264       ontologyManager.loadOntologies(ontologyConfig);
265       //Normalize ontology names
266       for (String id : ontologyManager.getOntologyIDs()) {
267         String officialName = MiriamLink.getName(id);
268         ontologyManager.getOntology(id).setName(officialName);
269         log.debug(id + " (" + officialName + ")");
270       }
271     } catch (Throwable e) {
272       throw new RuntimeException("Failed to load ontologies", e);
273     }
274 
275     // Build collections of the recommended xref.db names and synonyms and corresponding id patterns
276     this.dataPatterns = new ConcurrentHashMap<>();
277     this.allSynonyms = new CompositeCollection<>();
278     this.unofficialDbNames = Collections.newSetFromMap(new ConcurrentHashMap<>());
279     // need a temporaty, local "all synonyms" collection
280     // (hide the member one)
281     CompositeCollection<String> allSynonyms = new CompositeCollection<>();
282     // first, we simple prepare lists of db synonyms,
283     // from  Miriam, MI, and "extra" set (already done - constructor arg.)
284     for (String dt : MiriamLink.getDataTypesName()) {
285       String regexp = MiriamLink.getDataTypePattern(dt);
286       Pattern pattern = null;
287       try {
288         pattern = Pattern.compile(regexp);
289       } catch (PatternSyntaxException e) {
290         log.error(
291           "Pattern compilation failed for MIRIAM " +
292             "db=" + dt + "; regexp=" + dt + "; " + e);
293       }
294       String db = dbName(dt); //uppercase
295       List<String> synonyms = new ArrayList<>();
296       synonyms.add(db);
297       String[] names = MiriamLink.getNames(dt);
298       if (names != null)
299         for (String name : names) {
300           String s = dbName(name);
301           if (!synonyms.contains(s))
302             synonyms.add(s);
303         }
304       //save
305       allSynonyms.addComposited(synonyms);
306       // also associate primary name with ID patterns
307       if (pattern != null) //null value is not allowed for ConcurrentHashMap
308         dataPatterns.put(db, pattern); // will be used with all synonyms
309     }
310 
311     // load all names from MI 'database citation'
312     OntologyAccess mi = ontologyManager.getOntology("MI");
313     Collection<OntologyTermI> terms = mi.getAllChildren(new OntologyTermImpl("MI:0444"));
314     for (OntologyTermI term : terms) {
315       List<String> synonyms = new ArrayList<>();
316       synonyms.add(dbName(term.getPreferredName()));
317       for (String name : term.getNameSynonyms()) {
318         String s = dbName(name);
319         if (!synonyms.contains(s))
320           synonyms.add(s);
321       }
322       allSynonyms.addComposited(synonyms);
323     }
324 
325     // second, we populate unofficialDbNames by comparing
326     // current allSynonyms (from Miriam and MI:0444) with extraSynonyms (from config. file)
327     if(extraGroups != null) {
328       for (String db : extraGroups)
329         if (!allSynonyms.contains(db))
330           unofficialDbNames.add(db);
331 
332       extraGroups.getCollections().forEach(allSynonyms::addComposited);
333     }
334 
335     // find synonyms groups that overlap,
336     Cluster<Collection<String>> clus = new Cluster<Collection<String>>() {
337       @Override
338       public boolean match(Collection<String> a, Collection<String> b) {
339         return !Collections.disjoint(a, b);
340       }
341     };
342 
343     // find all overlapping groups of names (can be single-group "clusters" as well))
344     // get the set of clusters (sets) of overlapping synonym lists! (i.e., we're in fact clustering List<String> objects)
345     Set<Set<Collection<String>>> groupsOfGroups = clus.cluster(allSynonyms.getCollections(), Integer.MAX_VALUE);
346 
347     // perform overlapping groups merging
348     // so that Miriam's preferred name goes first
349     for (Set<Collection<String>> groupsToMerge : groupsOfGroups) {
350       if (groupsToMerge.size() > 1) {
351         List<String> merged = new ArrayList<>();
352         String primary = null;
353         for (Collection<String> group : groupsToMerge) {
354           for (String name : group) {
355             if (dataPatterns.containsKey(name))
356               primary = name;
357             if (!merged.contains(name))
358               merged.add(name);
359           }
360         }
361 
362         if (primary != null) {
363           merged.remove(primary);
364           merged.add(0, primary);
365         }
366         this.allSynonyms.addComposited(merged);
367       } else {
368         assert !groupsToMerge.isEmpty(); //one!
369         //not overlapping (with other) single group
370         this.allSynonyms.addComposited(groupsToMerge.iterator().next());
371       }
372     }
373   }

385   private List<String> getSynonymsForName(String name, CompositeCollection<String> groups) {
386     String dbName = dbName(name);
387 
388     for (Collection<String> group : groups.getCollections()) {
389       if (group.contains(dbName)) {
390         return (List<String>) group;
391       }
392     }
393     return Collections.emptyList();
394   }

428   public boolean xcheck() {
429     Collection[] lists = allSynonyms.getCollections().toArray(new Collection[]{});
430     for (int i = 0; i < lists.length; i++) {
431       Collection li = lists[i];
432       for (int j = i + 1; j < lists.length; j++) {
433         if (!Collections.disjoint(li, lists[j])) {
434           log.error("Different synonym groups overlap: "
435                       + li.toString() + " has names in common with "
436                       + lists[j].toString());
437           return false;
438         }
439       }
440     }
441 
442     return true; //no overlap
443   }
