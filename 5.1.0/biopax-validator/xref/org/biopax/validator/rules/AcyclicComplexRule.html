16  /**
17   * Checks:
18   * Component's cellularLocation is empty
19   * when complex'es is not.
20   * 
21   * @author rodche
22   */
23  @Component
24  public class AcyclicComplexRule extends AbstractRule<Complex> {
25  	
26  	public boolean canCheck(Object thing) {
27  		return thing instanceof Complex;
28  	}
29  
30  	public void check(final Validation validation, final Complex thing) {
31  		final Traverser traverser
32        = new AbstractTraverser(SimpleEditorMap.L3, e -> e.getProperty().equals("component"))
33      {
34  			@Override
35  			protected void visit(Object range, BioPAXElement domain, Model model,
36  					PropertyEditor editor) {
37  				if (thing.equals(range)) {
38  					error(validation, thing, "cyclic.inclusion", false, "is a component of itself or its componets... : " + domain.getUri());
39  				} else if(range instanceof Complex) {
40  					traverse((Complex) range, model);
41  				}
42  			}
43  		};
44  		
45  		traverser.traverse(thing, null);
46  	}
47  
48  }
