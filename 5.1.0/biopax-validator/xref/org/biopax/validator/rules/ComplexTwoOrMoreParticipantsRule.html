6   import org.biopax.paxtools.model.level3.Complex;
7   import org.biopax.paxtools.model.level3.PhysicalEntity;
8   import org.biopax.paxtools.model.level3.Stoichiometry;
9   import org.biopax.validator.AbstractRule;
10  import org.biopax.validator.api.beans.Validation;
11  import org.springframework.stereotype.Component;

13  
14  /**
15   * Usually complexes have more than 1 component, unless stoichiometry of &gt;1
16   * has been set on the single component, as would be done for a homodimer.
17   * 
18   * @author rodch
19   *
20   */
21  @Component
22  public class ComplexTwoOrMoreParticipantsRule extends AbstractRule<Complex> {
23  
24  	public boolean canCheck(Object thing) {
25  		return thing instanceof Complex;
26  	}
27  
28  	public void check(final Validation validation, Complex thing) 
29  	{	
30  		Set<PhysicalEntity> components = thing.getComponent();	
31  		
32  		if(components.isEmpty()) {
33  			error(validation, thing, "complex.incomplete", false, "no components");
34  		} else if(components.size()==1) { 
35  			// one component? - then stoi.coeff. must be > 1 (dimer, trimer,..)
36  			PhysicalEntity pe = components.iterator().next();
37  			Set<Stoichiometry> stoi = thing.getComponentStoichiometry();
38  			String msg = "has one component";
39  			if(stoi.isEmpty()) {
40  				error(validation, thing, "complex.incomplete", false, msg + ", but no stoichiometry defined.");
41  			} else { 
42  				if(stoi.size() > 1)
43  					msg += ", but multiple stoichiometries...";
44  				boolean ok = false;
45  				for(Stoichiometry s : stoi) {
46  					if(pe.equals(s.getPhysicalEntity())
47  							&& s.getStoichiometricCoefficient() > 1) {
48  						ok = true;
49  						break;
50  					}
51  					
52  					if(!pe.equals(s.getPhysicalEntity()) && s.getPhysicalEntity() != null) {
53  						error(validation, thing, "complex.stoichiometry.notcomponent", false, s, s.getPhysicalEntity(), pe);
54  					}
55  				}
56  				if(!ok) {
57  					error(validation, thing, "complex.incomplete", false, msg + "; which stoichiometry < 2.");
58  				}
59  			}
60  		}
61  
62  	}
63  
64  }
