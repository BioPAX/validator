6   import org.biopax.paxtools.model.level3.Pathway;
7   import org.biopax.paxtools.model.level3.PathwayStep;
8   import org.biopax.paxtools.model.level3.Process;
9   import org.biopax.validator.AbstractRule;
10  import org.biopax.validator.api.beans.Validation;
11  import org.springframework.stereotype.Component;
12  
13  /**
14   * Checks: PathwayStep.stepProcess (or stepConversion) - 
15   * the same process must be listed in the 
16   * Pathway.pathwayComponent property.
17   * 
18   * @author rodche
19   *
20   */
21  @Component
22  public class PathwayStepProcessesArePathwayComponents extends AbstractRule<PathwayStep> {
23  	
24  	public boolean canCheck(Object thing) {
25  		return thing instanceof PathwayStep 
26  		/* can be null, e.g., when the check is called 
27  		 * during a new PathwayStep is added to the model 
28  		 * but not yet assigned to any Pathway! 
29  		 * This is ignored.
30  		 */
31  		 && ((PathwayStep)thing).getPathwayOrderOf() != null;
32  	}
33  
34  	public void check(final Validation validation, PathwayStep step) {
35  		Pathway pathway = step.getPathwayOrderOf();
36  		/* can be null, e.g., when the check is called 
37  		 * during a new PathwayStep is added to the model 
38  		 * but not yet assigned to any Pathway! This case is ignored.
39  		 */
40  		if (pathway != null) {
41  			Set<Process> pathwayComponents = pathway.getPathwayComponent();
42  			for (Process stepProcess : step.getStepProcess()) {
43  				if (!pathwayComponents.contains(stepProcess)) {
44  					error(validation, step, "component.not.found", false, stepProcess, pathway);
45  				}
46  			}
47  		}
48  	}
49  		
50  }
