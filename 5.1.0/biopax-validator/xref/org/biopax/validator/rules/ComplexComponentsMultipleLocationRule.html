4   import java.util.Collection;
5   import java.util.HashSet;
6   import java.util.Set;
7   
8   import org.biopax.paxtools.model.level3.Complex;
9   import org.biopax.paxtools.model.level3.PhysicalEntity;
10  import org.biopax.validator.AbstractRule;
11  import org.biopax.validator.api.beans.Validation;
12  import org.springframework.stereotype.Component;
13  
14  /**
15   * Warning on complex components
16   * span multiple cellular compartments.
17   * 
18   * @author rodche
19   *
20   */
21  @Component
22  public class ComplexComponentsMultipleLocationRule extends
23  		AbstractRule<Complex> {
24  
25  	public void check(final Validation validation, Complex thing) {
26  		Set<PhysicalEntity> ents = thing.getComponent();
27  		if (ents != null) {
28  			Collection<String> diffLocs = new HashSet<String>();
29  			PhysicalEntity ref = thing;
30  			for (PhysicalEntity e : ents) {
31  				if(ref.getCellularLocation()==null 
32  						&& e.getCellularLocation() != null) {
33  					ref = e;
34  					continue;
35  				}
36  					
37  				if (e.getCellularLocation() != null
38  						&& !ref.getCellularLocation().isEquivalent(e.getCellularLocation())) {
39  					diffLocs.add(e + " " + e.getCellularLocation());		
40  				}
41  			}
42  			
43  			if(!diffLocs.isEmpty()) {
44  				error(validation, thing, 
45  						"component.different.location", false, thing.getCellularLocation() + "", diffLocs);
46  			}
47  		}
48  	}
49  
50  	public boolean canCheck(Object thing) {
51  		return thing instanceof Complex;
52  	}
53  }
