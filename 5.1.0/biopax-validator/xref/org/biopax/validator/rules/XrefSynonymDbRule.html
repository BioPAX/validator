11  /**
12   * Checks xref.db is one of official db names
13   * (warn about known unofficial synonyms and misspelling)
14   *
15   * @author rodche
16   */
17  @Component
18  public class XrefSynonymDbRule extends AbstractRule<Xref> {
19  
20    @Autowired
21    XrefUtils xrefUtils;
22  
23    public boolean canCheck(Object thing) {
24      return (thing instanceof Xref);
25    }
26  
27    public void check(final Validation validation, Xref x) {
28      String db = x.getDb();
29      if (db == null) {
30        return; // another (cardinality) rule reports
31      }
32  
33      String primary = xrefUtils.getPrimaryDbName(db);
34      // if primary is null, do nothing, - another rule (XrefRule) reports this
35      if (primary != null && !primary.equalsIgnoreCase(db)) {
36        // report only if it is definitely not official db synonym
37        if (xrefUtils.isUnofficialOrMisspelledDbName(db))
38          error(validation, x, "db.name.spelling", validation.isFix(), db, primary);
39  
40        // fix, sometimes w/o error message, anyway ;)
41        if (validation.isFix()) {
42          x.setDb(primary);
43        }
44      }
45  
46    }
47  }
