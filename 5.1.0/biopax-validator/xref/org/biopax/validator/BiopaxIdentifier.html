4   import org.biopax.paxtools.io.SimpleIOHandler;
5   import org.biopax.paxtools.model.BioPAXElement;
6   import org.biopax.paxtools.model.Model;
7   import org.biopax.validator.api.Identifier;
8   
9   /**
10   * BioPAX object domain specific implementation 
11   * of the {@link Identifier} strategy.
12   * 
13   * @author rodche
14   *
15   */
16  public final class BiopaxIdentifier implements Identifier {
17  
18  	/* (non-Javadoc)
19  	 * @see org.biopax.validator.api.Identifier#getId(java.lang.Object)
20  	 */
21  	public String identify(Object obj) {
22      	String id = "";
23      	
24  		if(obj instanceof SimpleIOHandler) {
25  			SimpleIOHandler r = (SimpleIOHandler) obj;
26  			id = r.getClass().getSimpleName(); 
27  	    	try {
28  	    		id = r.getId(); //current element URI
29  	    	} catch (Throwable e) {
30  	    		id = r.getXmlStreamInfo(); //location
31  			}
32  		} else if (obj instanceof BioPAXElement 
33  				&& ((BioPAXElement)obj).getUri() != null) {
34  			id = ((BioPAXElement) obj).getUri().replaceFirst("^.+#", "");
35  			// - strictly spk., does not always get the local part (depends on xml:base) but is OK.
36  		} else if(obj instanceof Model) {
37  			id = obj + "; xml:base=" + ((Model) obj).getXmlBase();
38  		} else {
39  			id = "" + obj;
40  		}
41  		
42  		return id;
43  	}
44  }
