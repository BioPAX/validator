193    /**
194     * Facade method to simplify most of the Loader package code for user who simply want to obtain
195     * a fully-constructed ontology object from a single OBO source file. This method will produce an
196     * object that is still valid for persistence however, so mandatory fields need to be provided.
197     *
198     * @param oboFilePath - The fully qualified path to the file on disk to be parsed [Mandatory]
199     * @param shortName   - The short label of the ontology (eg "GO", "MI", etc...) [Mandatory]
200     * @param fullName    - The full name of the ontology (eg Gene OntologyAccess)  [Optional]
201     * @param definition  - The definition of the ontology [Optional]
202     * @return a fully build OntologyAccess object.
203     * @throws IllegalArgumentException if the shortName is null or an empty string
204     * @throws IOException              on parse errors
205     */
206    public static Ontology loadFromOBOFile(
207            String oboFilePath, String shortName, String fullName, String definition
208    ) throws IOException {
209
210        //test parameters - file path
211        File oboFile = new File(oboFilePath);
212        if (!oboFile.exists()) {
213            throw new IOException("The file path to be parsed does not exist");
214        }
215
216        return loadFromOBOFile(oboFile, shortName, fullName, definition);
217
218    }

221    /**
222     * Facade method to simplify most of the Loader package code for user who simply want to obtain
223     * a fully-constructed ontology object from a single OBO source file that is accessed via an URL.
224     * The URL can be local or remote. If the URL is remote, a temporary copy will be obtained and
225     * parsed. This method will produce an object that is still valid for persistence however, so
226     * mandatory fields need to be provided.
227     *
228     * @param oboFileURL - The URL of the OBO file to be read (remote or local) [Mandatory]
229     * @param shortName  - The short label of the ontology (eg "GO", "MI", etc...) [Mandatory]
230     * @param fullName   - The full name of the ontology (eg Gene OntologyAccess)  [Optional]
231     * @param definition - The definition of the ontology [Optional]
232     * @return a fully build OntologyAccess object.
233     * @throws IllegalArgumentException if the shortName is null or an empty string
234     * @throws IOException              if the URL cannot be read or on parse errors
235     */
236    public static Ontology loadFromOBOFile(
237            final URL oboFileURL, final String shortName, final String fullName, final String definition
238    ) throws IOException {
239
240        //test parameters - URL
241        if (oboFileURL == null) {
242            throw new IOException("The URL to be parsed must not be null.");
243        }
244
245        //logging to System.out
246        System.out.println("OntologyBean.loadFromOBOFile will read from: " + oboFileURL.toString());
247
248        URLConnection conn = null;
249        BufferedReader in = null;
250        PrintWriter out = null;
251        File tempFile = null;
252
253        try {
254
255            boolean useNumericalProgressDisplay = false;
256
257            // Read URL content
258            conn = oboFileURL.openConnection();
259            int contentLength = conn.getContentLength();
260            if (contentLength > 0) {
261                useNumericalProgressDisplay = true;
262            }
263
264            in = new BufferedReader(new InputStreamReader(conn.getInputStream()), 4096);
265
266            // Create a temp file and write URL content in it.
267            File tempDirectory = new File(System.getProperty("java.io.tmpdir", "tmp"));
268            if (!tempDirectory.exists()) {
269                if (!tempDirectory.mkdirs()) {
270                    throw new IOException("Cannot create temp directory: " + tempDirectory.getAbsolutePath());
271                }
272            }
273            tempFile = File.createTempFile("ontologyBean.", ".obo", tempDirectory);
274            tempFile.deleteOnExit();
275
276            out = new PrintWriter(new FileWriter(tempFile));
277
278            //for progress reporting
279            int currentSize = 0;
280            int currentPct = 0;
281            int lineCount = 0;
282            int reportInc = 10;
283            int lastPct = -1;
284
285            String line;
286            while ((line = in.readLine()) != null) {
287
288                //write to file
289                out.println(line);
290
291                //if we have content length, display progress as percent of download
292                if (useNumericalProgressDisplay) {
293                    currentSize += line.length();
294                    currentPct = currentSize * 100 / contentLength;
295
296                    if (currentPct % reportInc == 0 && currentPct != lastPct) {
297                        System.out.print(currentPct + "% ");
298                        System.out.flush();
299                        lastPct = currentPct;
300                    }
301                    //otherwise, display progress as dots on the console
302                } else {
303                    lineCount++;
304                    // display progress bar on STDOUT
305                    if ((lineCount % 20) == 0) {
306                        System.out.print(".");
307                        System.out.flush();
308                        if ((lineCount % 500) == 0) {
309                            System.out.println("   " + lineCount);
310                        }
311                    }
312                }
313            }
314            System.out.println();
315
316            //logging to System.out
317            System.out.println("OntologyBean.loadFromOBOFile is processing file.");
318
319
320        } finally {
321            if (in != null) {
322                in.close();
323            }
324            if (out != null) {
325                out.close();
326            }
327        }
328
329        return loadFromOBOFile(tempFile, shortName, fullName, definition);
330
331    }

333    /**
334     * Facade method to simplify most of the Loader package code for user who simply want to obtain
335     * a fully-constructed ontology object from a single OBO source file. This method will produce an
336     * object that is still valid for persistence however, so mandatory fields need to be provided.
337     *
338     * @param oboFile    - The File object to be parsed [Mandatory]
339     * @param shortName  - The short label of the ontology (eg "GO", "MI", etc...) [Mandatory]
340     * @param fullName   - The full name of the ontology (eg Gene OntologyAccess)  [Optional]
341     * @param definition - The definition of the ontology [Optional]
342     * @return a fully build OntologyAccess object.
343     * @throws IllegalArgumentException if the shortName is null or an empty string
344     * @throws IOException              on parse errors
345     */
346    public static Ontology loadFromOBOFile(final File oboFile, final String shortName, final String fullName, final String definition) throws IOException {
347
348        //test parameters - shortName
349        if (shortName == null || (shortName != null && "".equals(shortName.trim()))) {
350            throw new IllegalArgumentException("The ShortName of an ontology cannot be null");
351        }
352
353        //test parameters - file path
354        if (!oboFile.exists()) {
355            throw new IOException("The file path to be parsed does not exist");
356        }
357
358        Loader loader = new BaseOBO2AbstractLoader() {
359
360
361            protected void configure() throws DataAdapterException {
362                parser = new OBO2FormatParser(oboFile.getAbsolutePath());
363//                setUseGreedy(true);
364                setUseGreedy(false);
365                ONTOLOGY_DEFINITION = definition;
366                FULL_NAME = fullName;
367                SHORT_NAME = shortName;
368            }
369
370            public Ontology getOntology() throws IOException {
371                try {
372                    configure();
373                    //create the object model
374                    process();
375                    //get the ontology
376                    return super.getOntology();
377                } catch (IllegalStateException ise) {
378                    throw new IOException("There were errors while loading the ontology: " + ise.getMessage());
379                } catch (DataAdapterException ioe) {
380                    throw new IOException("There were errors while loading the ontology: " + ioe.getMessage());
381                }
382            }
383        };
384
385        return loader.getOntology();
386
387    }
