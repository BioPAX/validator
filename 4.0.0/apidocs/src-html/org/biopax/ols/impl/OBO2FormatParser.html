034    private OBOSession session = null;
035
036    public OBO2FormatParser(String filePath) throws DataAdapterException {
037        OBOFileAdapter adapter = new OBOFileAdapter();
038        OBOFileAdapter.OBOAdapterConfiguration config = new OBOFileAdapter.OBOAdapterConfiguration();
039        config.getReadPaths().add(filePath);
040        session = adapter.doOperation(OBOAdapter.READ_ONTOLOGY, config, null);
041    }
042
043    public OBO2FormatParser(Collection<String> filePaths) throws DataAdapterException {
044        OBOFileAdapter adapter = new OBOFileAdapter();
045        OBOFileAdapter.OBOAdapterConfiguration config = new OBOFileAdapter.OBOAdapterConfiguration();
046        config.getReadPaths().addAll(filePaths);
047        session = adapter.doOperation(OBOAdapter.READ_ONTOLOGY, config, null);
048    }

050    /**
051     * returns all the terms parsed from an ontology
052     *
053     * @return a set of OBOClass/Instance objects, or null if session is not initialized
054     */
055    public Set<OBOObject> getTerms() {
056
057        LinkDatabase lnkDb = session.getLinkDatabase();
058        HashSet<OBOObject> terms = new HashSet<OBOObject>();
059        for (IdentifiedObject io : lnkDb.getObjects()) {
060            //do not return built-in obo: constructs
061            if ((io instanceof OBOClass || io instanceof Instance) 
062                        && !io.getID().toLowerCase().startsWith("obo:")) {
063                terms.add((OBOObject) io);
064            }
065        }
066        return terms;
067
068    }
069

079    /**
080     * returns the root terms of an ontology
081     *
082     * @return a collection of OBOClass terms
083     * @throws IllegalStateException if the session is not initialized
084     */
085    public Set<OBOObject> getRootTerms(boolean useGreedy) {
086
087        HashSet<OBOObject> roots = new HashSet<OBOObject>();
088        /*
089             * {@link RootAlgorithm#GREEDY GREEDY} root algorithm.
090         *
091         * @param linkDatabase
092         *            the linkDatabase to check
093         * @param includeTerms
094         *            whether to include root terms
095         * @param includeProperties
096         *            whether to include root properties
097         * @param includeObsoletes
098         *            whether to include obsolete terms & properties
099         * @param includeInstances
100         *            whether to include instances
101         */
102
103        if (useGreedy) {
104
105            //use greedy root detection
106            /**
107             * Returns all non-obsolete root terms in a given {@link LinkDatabase}.
108             * Equivalent to {@link #getRoots(session.getLinkDatabase(), true, false, false, false)}.
109             */
110            Collection<OBOClass> tmpRoots = TermUtil.getRoots(session);
111            roots.addAll(tmpRoots);
112
113        } else {
114
115            //use strict root detection
116            Collection<LinkedObject> tmpRoots = TermUtil.getRoots(RootAlgorithm.STRICT, session.getLinkDatabase());
117            for (LinkedObject lnk : tmpRoots) {
118                if (lnk instanceof OBOClass) {
119                    roots.add((OBOClass) lnk);
120                }
121            }
122
123        }
124
125        return roots;
126
127    }

130    /**
131     * Will compute all the term paths from a given term to all its children, irrespective of distance
132     *
133     * @param distance         Distance in arcs between nodes. Set to 1 to begin, will be incremented in recursive method calls
134     * @param term             The term to scan for a given distance
135     * @param relationshipType - a set of possible synonyms for the type of relationship to link the terms (is_a, part_of, develops_from)
136     * @return A hashmap where the key is a term ID and the value is an Integet repersenting the distance
137     */
138    private HashMap<String, Integer> computeChildPaths(int distance, Set relationshipType, LinkedObject term, HashMap<String, Integer> paths) {
139
140        //get all children
141        Collection<Link> children = term.getChildren();
142
143        //iterate over children
144        for (Link trm : children) {
145            //add relationship
146            if (relationshipType.contains(trm.getType().getID())) {
147                paths.put(trm.getChild().getID(), distance);
148                //we're building up the map here, so we don't care about the return
149                try {
150                    computeChildPaths(distance + 1, relationshipType, trm.getChild(), paths);
151                } catch (StackOverflowError e) {
152                    logger.error("Stack overflow when computing child paths for: " + term.getID() + " for relationships: " + relationshipType);
153                    throw new IllegalStateException(e);
154                }
155//            } else {
156//                logger.debug("reltype is:" + trm.getType().getID() + " but expected :" + relationshipType);
157//                }
158            }
159        }
160
161        return paths;
162    }
163
164    /**
165     * Will compute all the term paths from a given term to all its children, irrespective of distance
166     *
167     * @param distance         Distance in arcs between nodes. Set to 1 to begin, will be incremented in recursive method calls
168     * @param relationshipType - a set of possible synonyms for the type of relationship to link the terms (is_a, part_of, develops_from)
169     * @param term             The term to scan for a given distance
170     * @return A hashmap where the key is a term ID and the value is an Integet repersenting the distance
171     */
172    public HashMap<String, Integer> computeChildPaths(int distance, Set relationshipType, LinkedObject term) {
173        //the true call is required only once, at each call
174        return computeChildPaths(distance, relationshipType, term, new HashMap<String, Integer>());
175    }
