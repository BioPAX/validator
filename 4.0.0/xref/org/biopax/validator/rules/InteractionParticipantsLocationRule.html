7   import java.util.Set;

9   import org.biopax.paxtools.model.level3.Entity;

13  import org.biopax.paxtools.model.level3.Transport;
14  import org.biopax.validator.api.AbstractRule;
15  import org.biopax.validator.api.beans.Validation;
16  import org.springframework.stereotype.Component;

18  /**
19   * Warning on participants of one interaction span multiple cellular compartments.
20   * For Transport however, it checks separately on the left and right sides, 
21   * and also it does not check for TemplateReaction.
22   * 
23   * @author rodche
24   */
25  @Component
26  public class InteractionParticipantsLocationRule extends
27  		AbstractRule<Interaction> {
28  
29  	public void check(final Validation validation, Interaction thing) {
30  		// exclude template reactions (second time, - after canCheck ;))
31  		if(thing instanceof TemplateReaction)
32  			return;
33  		
34  		Set<Entity> ents = thing.getParticipant();
35  		if(ents != null) {
36  		for (Entity e1 : ents) {
37  			if (e1 instanceof PhysicalEntity) 
38  			{
39  				for (Entity e2 : ents) 
40  				{
41  					if (e2 instanceof PhysicalEntity && !e1.equals(e2)
42  					&& !((PhysicalEntity) e1).hasEquivalentCellularLocation((PhysicalEntity) e2)) 
43  					{ 
44  						if(thing instanceof Transport) 
45  						{ // it's a bit different rule for the transport
46  							Transport tr = (Transport) thing;
47  							boolean onDifferentSides = 
48  								(tr.getLeft().contains(e1) && !tr.getLeft().contains(e2))
49  								||
50  								(!tr.getLeft().contains(e1) && tr.getLeft().contains(e2));
51  							
52  							if(onDifferentSides) {
53  								continue; // no error
54  							}
55  						}
56  						// report error
57  						error(validation, thing, "multiple.location", false, e1, e2);
58  					  }
59  					}
60  				}
61  			}
62  		}
63  	}
64  
65  	public boolean canCheck(Object thing) {
66  		return thing instanceof Interaction 
67  			&& !(thing instanceof TemplateReaction);
68  	}
69  
70  }
