131   /**
132    * Writes the multiple results report.
133    * (as transformed XML).
134    *
135    * @param validatorResponse results
136    * @param writer output
137    * @param xslt source
138    */
139   public static void write(ValidatorResponse validatorResponse, Writer writer, Source xslt) {
140     try {
141       if (xslt != null) {
142         Document doc = asDocument(validatorResponse);
143         Source xmlSource = new DOMSource(doc);
144         Result result = new StreamResult(writer);
145         TransformerFactory transFact = TransformerFactory.newInstance();
146         Transformer trans = transFact.newTransformer(xslt);
147         trans.setOutputProperty(OutputKeys.INDENT, "yes");
148         trans.setOutputProperty("{http://xml.apache.org/xslt}indent-amount", "2");
149         trans.transform(xmlSource, result);
150       } else {
151         // write without any xslt
152         getMarshaller().marshal(validatorResponse, writer);
153       }
154 
155     } catch (Exception e) {
156       throw new RuntimeException("Cannot transform/serialize/write: "
157         + validatorResponse, e);
158     }
159   }

161   /**
162    * Writes the single validation report.
163    * (as transformed XML).
164    *
165    * @param validationResult results
166    * @param writer XML validation result output
167    * @param xslt source
168    */
169   public static void write(Validation validationResult, Writer writer, Source xslt) {
170     ValidatorResponse resp = new ValidatorResponse();
171     resp.addValidationResult(validationResult);
172     write(resp, writer, xslt);
173   }

175   /**
176    * Converts a validator response (contains one or
177    * several validation results) into the document.
178    *
179    * @param validatorResponse results
180    * @return results as XML DOM
181    */
182   public static Document asDocument(ValidatorResponse validatorResponse) {
183     DOMResult domResult = marshal(validatorResponse);
184     Document validation = (Document) domResult.getNode();
185     return validation;
186   }

195   public static Element asElement(Validation validationResult) {
196     DOMResult domResult = marshal(validationResult);
197     Element validation = (Element) domResult.getNode().getFirstChild();

202   protected static DOMResult marshal(Object obj) {
203     DOMResult domResult = new DOMResult();
204     try {
205       getMarshaller().marshal(obj, domResult);
206     } catch (Exception e) {
207       throw new RuntimeException("Cannot serialize object: " + obj, e);
208     }
209     return domResult;
210   }

230   /**
231    * Creates an error type with single error case.
232    * If the message source is null (e.g., in tests, when a rule
233    * is checked outside the validator framework),
234    *
235    * @param messageSource error message source
236    * @param locale locale
237    * @param objectName name of the problematic model object
238    * @param errorCode validator error code
239    * @param ruleName validation rule name
240    * @param profile validation profile
241    * @param isFixed true/false - whether it's auto-fixed or not
242    * @param msgArgs error message details
243    * @return error type
244    */
245   public static ErrorType createError(
246     MessageSource messageSource, Locale locale,
247     String objectName, String errorCode,
248     String ruleName, String profile, boolean isFixed,
249     Object... msgArgs) {
250 
251     if (objectName == null) {
252       objectName = "null";
253       logger.warn("Creating an error " + errorCode + " for Null object!");
254     }
255 
256     //get/use current behavior
257     Behavior behavior = getRuleBehavior(ruleName, profile, messageSource);
258     // new error object
259     ErrorType error = new ErrorType(errorCode, behavior);
260 
261     // build human-friendly messages using default locale and property files (msg sources)
262     String commonMsg = (messageSource != null)
263       ? messageSource.getMessage(errorCode + ".default", new Object[]{}, "No description.", locale)
264       : "No description.";
265     error.setMessage(commonMsg);
266 
267     String msg = (messageSource != null)
268       ? messageSource.getMessage(errorCode, msgArgs, StringUtils.join(msgArgs, "; "), locale).replaceAll("\r|\n+", " ")
269       : StringUtils.join(msgArgs, "; ");
270 
271     // resolve/set BioPAX problem category
272     String category = (messageSource != null)
273       ? messageSource.getMessage(errorCode + ".category", null, Category.INFORMATION.name(), locale)
274       : null;
275     if (category != null)
276       error.setCategory(Category.valueOf(category.trim().toUpperCase()));
277 
278     // add one error case
279     ErrorCaseType errorCase = new ErrorCaseType(ruleName, objectName, msg);
280     errorCase.setFixed(isFixed); //
281     error.addErrorCase(errorCase);
282 
283     //done.
284     return error;
285   }

314   /**
315    * Gets rule's behavior (mode) during unit testing
316    * when messageSource can be null.
317    *
318    * @param ruleName      validation rule class name, e.g., org.biopax.validator.rules.MyRule
319    * @param profile       validation profile name or null (default profile)
320    * @param messageSource error message source (properties)
321    * @return error level/mode
322    * @see ValidatorUtils#getRuleBehavior(String, String)
323    * @see ValidatorUtils#createError(MessageSource, Locale, String, String, String, String, boolean, Object...)
324    */
325   private static Behavior getRuleBehavior(String ruleName, String profile, MessageSource messageSource) {
326     if (messageSource == null) return Behavior.ERROR;
327 
328     // get the default behavior value first
329     String value = messageSource.getMessage(ruleName + ".behavior", null, "ERROR", Locale.getDefault());
330 
331     // - override from the profile, if set/available
332     if (profile != null && !profile.isEmpty())
333       value = messageSource.getMessage(ruleName + ".behavior." + profile, null, value, Locale.getDefault());
334 
335     return Behavior.valueOf(value.toUpperCase());
336   }

339   /**
340    * @param ruleName validation rule class name, e.g., org.biopax.validator.rules.MyRule
341    * @return rule description
342    */
343   public String getRuleDescription(String ruleName) {
344 
345     String tip = messageSource.getMessage(ruleName, null, "",
346       Locale.getDefault());
347     if (tip == null || "".equals(tip)) {
348       tip = "description is not found in the messages.properties file";
349     } else {
350       tip = StringEscapeUtils.escapeHtml4(tip);
351     }
352 
353     return tip;
354   }
