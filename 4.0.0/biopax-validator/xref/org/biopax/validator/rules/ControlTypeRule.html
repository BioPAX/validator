7   import org.biopax.paxtools.model.level3.Catalysis;
8   import org.biopax.paxtools.model.level3.Control;
9   import org.biopax.paxtools.model.level3.ControlType;
10  import org.biopax.paxtools.model.level3.TemplateReactionRegulation;
11  import org.biopax.validator.api.AbstractRule;
12  import org.biopax.validator.api.beans.Validation;
13  import org.springframework.stereotype.Component;
14  
15  /**
16   * Checks:
17   * 
18   * Catalysis.controlType - only "ACTIVATION"
19   * TemplateReactionRegulation.controlType - either "ACTIVATION" or "INHIBITION"
20   *
21   * @author rodche
22   */
23  @Component
24  public class ControlTypeRule extends AbstractRule<Control> {
25  
26  	private void fix(Control t, Object... values) {
27  		if(t instanceof Catalysis) {
28  			((Catalysis)t).setControlType(ControlType.ACTIVATION);
29  		} else if (t instanceof TemplateReactionRegulation){
30  			if(values.length >0 && values[0] instanceof ControlType) {
31  				ControlType ct = (ControlType) values[0];
32  				((TemplateReactionRegulation)t).setControlType(ct);
33  			} else {
34  				((TemplateReactionRegulation)t).setControlType(null);
35  				if(logger.isInfoEnabled()) 
36  					logger.info(t.getUri() +
37  							" - invalid controlType value deleted");
38  			}
39  		} else {
40  			logger.error("This does not auto-fix " +
41  					" controlType property of "
42  					+ t.getModelInterface().getSimpleName());
43  		}
44  	}
45  
46  	public boolean canCheck(Object thing) {
47  		return thing instanceof Control;
48  	}
49  
50  	public void check(final Validation validation, Control thing) {
51  		if(thing.getControlType() != null) 
52  		{			
53  			if(thing instanceof Catalysis) {
54  				Catalysis cat = (Catalysis) thing;
55  				if(cat.getControlType() != ControlType.ACTIVATION) {
56  					error(validation, thing, "range.violated", 
57  							validation.isFix(), "controlType",
58  							cat.getControlType().name(), "", ControlType.ACTIVATION.name()
59  							+ " (or empty)");
60  					if(validation.isFix()) {
61  						fix(thing);
62  					}
63  				}
64  			} else if(thing instanceof TemplateReactionRegulation) {
65  				TemplateReactionRegulation trr = (TemplateReactionRegulation) thing;
66  				if(! (trr.getControlType() == ControlType.ACTIVATION
67  						|| trr.getControlType() == ControlType.INHIBITION) ) 
68  				{
69  					error(validation, thing, "range.violated", 
70  							validation.isFix(), "controlType",
71  							trr.getControlType().name(), "", ControlType.ACTIVATION.name() + " or " 
72  							+ ControlType.INHIBITION.name()
73  							+ " (or empty)");
74  					if(validation.isFix()) {
75  						fix(thing);
76  					}
77  				}
78  			}
79  		}
80  	}
81  	
82  }
