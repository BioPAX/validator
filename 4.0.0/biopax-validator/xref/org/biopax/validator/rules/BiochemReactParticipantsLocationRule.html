19  /**
20   * BiochemicalReaction (base) - the participants on one side don't 
21   * change compartments on the other side (otherwise it should be a 
22   * TransportWithBiochemicalReaction)
23   * 
24   * TODO Complexes are matched by name, other physical entities can be "identified" using entity reference.
25   * 
26   * @author rodche
27   */
28  @Component
29  public class BiochemReactParticipantsLocationRule extends AbstractRule<BiochemicalReaction>
30  {
31  
32      public void check(final Validation validation, BiochemicalReaction react)
33      {
34         Set<PhysicalEntity> left = react.getLeft();
35         Set<PhysicalEntity> right = react.getRight();
36         for (PhysicalEntity lefty : left)
37         {
38              for (PhysicalEntity righty : right)
39              {
40              	// PhysicalEntity is either Complex or SimplePhysicalEntity (NucleicAcid, Protein or SmallMolecule)
41              	
42              	if(!lefty.getModelInterface().equals(righty.getModelInterface())) {
43                  	continue;
44                  }
45              	
46              	boolean sameComplex = (lefty instanceof Complex) // means - righty is also Complex
47              		&& CollectionUtils.containsAny(lefty.getName(), righty.getName());
48              	// Complex does not have entityReference to match the same kind on both sides...
49              	
50              	boolean sameSimplePhysicalEntity = (lefty instanceof SimplePhysicalEntity) 
51              		&& ( ((SimplePhysicalEntity)lefty).getEntityReference() == null 
52              			? ((SimplePhysicalEntity)righty).getEntityReference() == null 
53              			: ((SimplePhysicalEntity)lefty).getEntityReference()
54              				.isEquivalent(((SimplePhysicalEntity)righty).getEntityReference()) );
55              	
56              	if(sameComplex || sameSimplePhysicalEntity) {
57                  	boolean sameLoc = lefty.hasEquivalentCellularLocation(righty);
58                  	if(!sameLoc	&& !(react instanceof Transport)) {
59                  		error(validation, react, "participant.location.changed", false, lefty, righty); 
60                  	} else if(sameLoc && react instanceof Transport) {
61                  		error(validation, react, "transport.location.same", false, lefty, righty); 
62                  	}
63                  } 
64              }
65          }
66      }
67  
68      public boolean canCheck(Object thing) {
69          return thing instanceof BiochemicalReaction;
70      }
71  }
