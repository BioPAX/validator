19  /**
20   * Checks that defferent kind physical entities 
21   * (using different EntityReference) do not share
22   * names.
23   * 
24   * @author rodche
25   */
26  @Component
27  public class SameNameDiffKindPhysEntitiesRule extends
28  		AbstractRule<Model> {
29  
30  	public void check(final Validation validation, Model model) {
31  			Set<SimplePhysicalEntity> peers = new HashSet<SimplePhysicalEntity>(
32  				model.getObjects(SimplePhysicalEntity.class));
33  			
34  			Cluster<SimplePhysicalEntity> groupping =
35  				new Cluster<SimplePhysicalEntity>() {
36  					@Override
37  					public boolean match(SimplePhysicalEntity a,
38  							SimplePhysicalEntity b) {
39  						return !a.equals(b) 
40  							&& a.getEntityReference() != null
41  							&& !a.getName().isEmpty() && !b.getName().isEmpty()
42  							&& !a.getEntityReference().isEquivalent(b.getEntityReference())
43  							&& CollectionUtils.containsAny(a.getName(), b.getName());
44  					}
45  			};
46  			
47  			Set<Set<SimplePhysicalEntity>> sharedNamesClusters = 
48  				groupping.cluster(peers, Integer.MAX_VALUE);
49  				
50  			// report the error once for each cluster
51  			for (Set<SimplePhysicalEntity> sharedNames : sharedNamesClusters) 
52  			{
53  				if(sharedNames.size() > 1) {
54  					SimplePhysicalEntity a = sharedNames.iterator().next();
55  					error(validation, a, "diff.kind.same.name",	false, sharedNames);
56  				}
57  			}
58  	}
59  
60  	
61  	public boolean canCheck(Object thing) {
62  		return thing instanceof Model
63  			&& ((Model)thing).getLevel()==BioPAXLevel.L3;
64  	}
65  
66  }
