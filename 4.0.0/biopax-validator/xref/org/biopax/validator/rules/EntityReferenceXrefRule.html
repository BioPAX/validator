7   import org.biopax.paxtools.model.BioPAXElement;
8   import org.biopax.paxtools.model.level3.EntityReference;
9   import org.biopax.paxtools.model.level3.UnificationXref;
10  import org.biopax.paxtools.model.level3.Xref;
11  import org.biopax.validator.api.AbstractRule;
12  import org.biopax.validator.api.beans.Validation;
13  import org.springframework.stereotype.Component;
14  
15  /**
16   * Checks a non-generic EntityReference has a unification xref.
17   * 
18   * @author rodche
19   */
20  @Component
21  public class EntityReferenceXrefRule extends AbstractRule<EntityReference> {
22  
23     public void check(final Validation validation, EntityReference er) {
24  	   if(er.getMemberEntityReference().isEmpty()) { // for non-generic ERs only
25          if (er.getXref().isEmpty()) {
26              error(validation, er, "no.xrefs", false);
27          } else {
28              boolean present = false;
29              for (Xref x : er.getXref()) {
30                  Class<? extends BioPAXElement> face = x.getModelInterface();
31                  if (UnificationXref.class.equals(face)) {
32                      present = true;
33                  }
34              }
35              if (!present) {
36                  error(validation, er, "no.unification.xref", false);
37              }
38          }
39  	   }
40      }
41  
42  	public boolean canCheck(Object thing) {
43  		return thing instanceof EntityReference;
44  	}
45  
46  }
