17  	@Test
18  	public void ontologyLoading() throws OntologyLoaderException {
19  		
20  		final Properties cfg = new Properties();
21  		cfg.put("MI", "classpath:mi.obo");
22  		cfg.put("MOD", "classpath:mod.obo");
23  		
24  		OntologyManager manager = new OntologyManagerImpl(cfg);
25  		
26  		Collection<String> ontologyIDs = manager.getOntologyIDs();
27  		assertTrue(ontologyIDs.contains("MOD"));
28  		assertTrue(ontologyIDs.contains("MI"));
29  
30  		OntologyAccess oa2 = manager.getOntology("MOD");
31  		assertNotNull(oa2);
32  		assertTrue(oa2 instanceof OntologyAccessImpl);
33  		
34  		OntologyTermI t = oa2.getTermForAccession("MOD:00048");
35  		assertNotNull(t);
36  		//test that apostrophe is not escaped (-due to a bug in the OBO parser, part of ols-1.18)!
37  		assertTrue(t.getPreferredName().equalsIgnoreCase("O4'-phospho-L-tyrosine"));
38  		
39  		oa2 = manager.getOntology("MI");
40  		assertNotNull(oa2);
41  		assertTrue(oa2 instanceof OntologyAccessImpl);
42  	}
43  }
