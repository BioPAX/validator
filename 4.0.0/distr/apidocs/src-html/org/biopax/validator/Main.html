036public class Main {
037        static final Log log = LogFactory.getLog(Main.class);
038        static ApplicationContext ctx;
039        static boolean autofix = false;
040        static int maxErrors = 0;
041        static final String EXT = ".modified.owl";
042        static String profile = null;
043        static String xmlBase = null;
044        static String outFormat = "html";
045        static String output = null;

048        private static void setUpLogger() {
049                //set defaults
050                Properties properties = new Properties();
051                properties.put("log4j.rootLogger", "ERROR, Console");
052                properties.put("log4j.appender.Console", "org.apache.log4j.ConsoleAppender");
053                properties.put("log4j.appender.Console.layout", "org.apache.log4j.PatternLayout");
054                properties.put("log4j.appender.Console.layout.ConversionPattern", "%-4r [%t] %-5p %c %x - %m%n");
055                PropertyConfigurator.configure(properties);
056                
057                properties = new Properties(properties);
058                
059                try {
060                        properties.load(new FileReader("log4j.properties"));
061                } catch (IOException e) {
062                        throw new RuntimeException("Failed to load cPath2 properties " +
063                                        "from log4j.properties", e);
064                }
065                
066                PropertyConfigurator.configure(properties);
067        }
068        
069        public static void main(String[] args) throws Exception {
070                
071                setUpLogger();
072                
073        if(args == null || args.length == 0) {
074                log.warn("At least input file/dir must be specified.");
075                printHelpAndQuit();
076        }
077
078                String input = args[0];
079                
080                if(input == null || input.isEmpty() || input.startsWith("--")) {
081                        log.warn("Input (file, url, or directory) is probably missing");
082                        printHelpAndQuit();
083                }
084        
085                // match optional parameters
086                for (int i = 1; i < args.length; i++) {
087                        if("--auto-fix".equalsIgnoreCase(args[i])) {
088                                autofix = true;
089                        } else if(args[i].startsWith("--max-errors=")) {
090                                String n = args[i].substring(13);
091                                maxErrors = Integer.parseInt(n);
092                        } else if(args[i].startsWith("--profile=")) {
093                                profile = args[i].substring(10);
094                        } else if(args[i].startsWith("--xmlBase=")) {
095                                xmlBase = args[i].substring(10);
096                        } else if(args[i].startsWith("--output=")) {
097                                output = args[i].substring(9);  
098                        } else if(args[i].startsWith("--out-format=")) {
099                                outFormat = args[i].substring(13);
100                                if(outFormat.isEmpty())
101                                        outFormat = "html";
102                        }
103                }
104
105                // this does 90% of the job ;)
106                ctx = new ClassPathXmlApplicationContext(
107                        new String[] {"META-INF/spring/appContext-loadTimeWeaving.xml",
108                                        "META-INF/spring/appContext-validator.xml"});
109                // Rules are now loaded, and AOP is listening for BioPAX model method calls.
110                
111        // get the beans to work with
112        Validator validator = (Validator) ctx.getBean("validator");
113                
114                // go validate all
115                runBatch(validator, getResourcesToValidate(input));
116        }

119        private static void printHelpAndQuit() {
120        final String usage = 
121                        "\nThe BioPAX Validator v3\n\n" +
122                    "Usage (arguments):\n <input> [--output=<filename>] [--out-format=xml|html] [--auto-fix] " +
123                    "[--xmlBase=<base>] [--max-errors=<n>] [--profile=notstrict]\n\n" + 
124                    "Given --output=<filename>, a one-file validation report will be \n" +
125                    "generated (HTML or XML) instead of default report file(s) in the \n" +
126                    "current directory. Optional arguments can go in any order.\n" +
127                    "For example:\n" +
128                    "  path/dir --out-format=xml\n" +
129                    "  list:batch_file.txt --output=reports.html\n" +
130                    "  file:biopax.owl --out-format=xml --auto-fix\n" +
131                    "  http://www.some.net/data.owl\n\n" +
132                    "A batch file should list one task (resource) per line, i.e.,\n" +
133                    "file:path/file or URL (to BioPAX data)\n" +
134                    "If '--auto-fix' option was used, it also creates a new BioPAX file \n" +
135                    "for each input file in the current working directory \n" +
136                    "(adding '.modified.owl' exention). If the outFormat file extension \n" +
137                    "is '.html', the XML result will be auto-transformed to a stand-alone \n" +
138                    "HTML/javascript page, which is very similar to what the online version returns.";
139        System.out.println(usage);
140        System.exit(-1);
141        }

144        protected static void runBatch(Validator validator, 
145                        Collection<Resource> resources) throws IOException {                                          
146
147                //collect all reports in this object (only if --output option was used)
148                final ValidatorResponse consolidatedReport = new ValidatorResponse();  
149                
150                // Read from the batch and validate from file, id or url, line-by-line (stops on first empty line)
151        for (Resource resource: resources) {
152                Validation result = new Validation(new IdentifierImpl(), resource.getDescription(), 
153                                autofix, null, maxErrors, profile);
154                result.setDescription(resource.getDescription()); 
155                log.info("BioPAX DATA IMPORT FROM: " + result.getDescription());
156                        try{                            
157                                validator.importModel(result, resource.getInputStream());
158                                validator.validate(result);
159                                
160                                //if autofix is enabled, then do normalize too
161                                if(autofix) {
162                                        Model model = (Model) result.getModel();
163                                        Normalizer normalizer = new Normalizer();
164                                        normalizer.setXmlBase(xmlBase); //if xmlBase is null, the model's one is used
165                                        normalizer.normalize(model);
166                                }
167                                
168                                if(output != null)
169                                        consolidatedReport.addValidationResult(result);
170                                
171                        } catch (Exception e) {
172                                log.error("failed", e);
173                        }
174                        
175                        final String filename = outFileName(result);
176                        PrintWriter writer;
177                        
178                        // save modified (normalized) biopax if the option was used
179                        if (autofix) {
180                                Model model = (Model) result.getModel();
181                                (new SimpleIOHandler()).convertToOWL(model, new FileOutputStream(filename + EXT));
182                        }
183                        
184                        // remove the BioPAX data before writing report
185                        result.setModel(null);
186                        result.setModelData(null);
187                        
188                        // save the individual validation results
189                        //unless the user specified the output file explicitly
190                        if(output == null || output.isEmpty()) { 
191                                writer = new PrintWriter(filename + ".validation." + outFormat);
192                                Source xsltSrc = (outFormat.equalsIgnoreCase("html"))
193                                                ? new StreamSource(ctx.getResource("classpath:html-result.xsl").getInputStream())
194                                                : null;
195                                ValidatorUtils.write(result, writer, xsltSrc);                  
196                                writer.close();
197                        }
198                        
199                        validator.getResults().remove(result);
200                        log.info("Done with " + filename);
201                }
202        
203                // save if the user specified the output file explicitly
204                if(output != null) { 
205                        Writer writer = new PrintWriter(output);
206                        Source xsltSrc = (outFormat.equalsIgnoreCase("html"))
207                                ? new StreamSource(ctx.getResource("classpath:html-result.xsl").getInputStream())
208                                        : null;
209                        ValidatorUtils.write(consolidatedReport, writer, xsltSrc);                      
210                        writer.close();
211                }        
212    }

215        private static String outFileName(Validation result) {
216                String filename = result.getDescription();
217                // if was URL, create a shorter name;
218                // remove ']', '[', and ending '/', if any
219                filename = filename.replaceAll("\\[|\\]","").replaceFirst("/&", ""); 
220                int idx = filename.lastIndexOf('/');
221                if(idx >= 0) {
222                        if(idx < filename.length() - 1)
223                                filename = filename.substring(idx+1);
224                }
225                
226                return filename;
227        }

230        public static Collection<Resource> getResourcesToValidate(String input) throws IOException {
231                        Set<Resource> setRes = new HashSet<Resource>();
232                        
233                File fileOrDir = new File(input);
234                if(fileOrDir.isDirectory()) {
235                        // validate all the OWL files in the folder
236                        FilenameFilter filter = new FilenameFilter() {
237                            public boolean accept(File dir, String name) {
238                                return (name.endsWith(".owl"));
239                            }
240                        };
241                        for (String s : fileOrDir.list(filter)) {
242                                String uri = "file:" + fileOrDir.getCanonicalPath() + File.separator + s;
243                                setRes.add(ctx.getResource(uri));
244                        }
245                } else if (input.startsWith("list:")) {
246                        // consider it's a file that contains a list of (pseudo-)URLs
247                        String batchFile = input.replaceFirst("list:", "file:");
248                    Reader isr = new InputStreamReader(ctx.getResource(batchFile).getInputStream());
249                    BufferedReader reader = new BufferedReader(isr);
250                        String line;
251                    while ((line = reader.readLine()) != null && !"".equals(line.trim())) {
252                        // check the source URL
253                        if( !ResourceUtils.isUrl(line)) {
254                                log.error("Invalid URL: " + line + 
255                                                ". A resource must be either a " +
256                                                "pseudo URL (classpath: or file:) or standard URL!");
257                                continue;
258                        }
259                                setRes.add(ctx.getResource(line));
260                        }
261                    reader.close();
262                } else {
263                        // a single local OWL file or remote data
264                        Resource resource = null;
265                                if (!ResourceUtils.isUrl(input)) 
266                                        input = "file:" + input;
267                                resource = ctx.getResource(input);
268                        setRes.add(resource);
269                }
270                      
271                return setRes;
272        }
