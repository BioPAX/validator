029/**
030 * Simple (example) BioPAX Validator client 
031 * to upload and check BioPAX OWL files.
032 * 
033 * @author rodche
034 *
035 */
036public class BiopaxValidatorClient {
037        private static final Log log = LogFactory.getLog(BiopaxValidatorClient.class);
038        
039        /**
040         * Default BioPAX Validator's URL
041         */
042        public static final String 
043                DEFAULT_VALIDATOR_URL = "http://www.biopax.org/validator/check.html";
044        
045        /**
046         * The Java Option to set a BioPAX Validator URL 
047         * (if set, it overrides the default as well as a value set via the Constructor)
048         */
049        public static final String JVM_PROPERTY_URL = "biopax.validator.url";
050        
051        public static enum RetFormat {
052                HTML,// errors as HTML/Javascript 
053                XML, // errors as XML
054                OWL; // modified BioPAX only (when 'autofix' or 'normalize' is true)
055        }
056        
057        private String url;
058
059        
060    /**
061     * Main Constructor
062     * 
063     * It configures for the validator's URL
064     * (defined by DEFAULT_VALIDATOR_URL constant)
065     * and result format ().
066     * 
067     * @param url - validator's file-upload form address
068     */
069    public BiopaxValidatorClient(String url) {
070                // 1) use the arg (if not empty/null) or the default URL
071        this.url = (url == null || url.isEmpty())
072                        ? DEFAULT_VALIDATOR_URL : url;
073        
074        // 2) override if the JVM option is set to another value
075        this.url = System.getProperty(JVM_PROPERTY_URL, this.url);
076                
077        // 3) get actual location (force through redirects, if any)
078                try {
079                        this.url = location(this.url);
080                } catch (IOException e) {
081                        log.warn("Failed to resolve to actual web service " +
082                                "URL using: " + url + " (if there is a 301/302/307 HTTP redirect, " +
083                                        "then validation requests (using HTTP POST method) will probably fail...)", e);
084                }
085        }
086

088    /**
089     * Default Constructor
090     * 
091     * It configures for the default validator URL.
092     */
093    public BiopaxValidatorClient() {
094        this(null);
095        }

098    /**
099     * Checks a BioPAX OWL file(s) or resource 
100     * using the online BioPAX Validator 
101     * and prints the results to the output stream.
102     * 
103     * @param autofix true/false (experimental)
104     * @param profile validation profile name
105     * @param retFormat xml, html, or owl (no errors, just modified owl, if autofix=true)
106     * @param biopaxUrl check the BioPAX at the URL
107     * @param biopaxFiles an array of BioPAX files to validate
108     * @param out
109     * @throws IOException
110     */
111    public void validate(boolean autofix, String profile, RetFormat retFormat, Behavior filterBy,
112                Integer maxErrs, String biopaxUrl, File[] biopaxFiles, OutputStream out) throws IOException 
113    {
114        MultipartEntityBuilder meb = MultipartEntityBuilder.create();
115        meb.setCharset(Charset.forName("UTF-8"));
116        
117        if(autofix)
118                meb.addTextBody("autofix", "true");
119
120                //TODO add extra options (normalizer.fixDisplayName, normalizer.xmlBase)?
121
122                if(profile != null && !profile.isEmpty())
123                meb.addTextBody("profile", profile);
124        if(retFormat != null)
125                meb.addTextBody("retDesired", retFormat.toString().toLowerCase());
126        if(filterBy != null)
127                meb.addTextBody("filter", filterBy.toString());
128        if(maxErrs != null && maxErrs > 0)
129                meb.addTextBody("maxErrors", maxErrs.toString());
130        if(biopaxFiles != null && biopaxFiles.length > 0)
131                for (File f : biopaxFiles) //important: use MULTIPART_FORM_DATA content-type
132                        meb.addBinaryBody("file", f, ContentType.MULTIPART_FORM_DATA, f.getName());
133        else if(biopaxUrl != null) {
134                meb.addTextBody("url", biopaxUrl);
135        } else {
136                log.error("Nothing to do (no BioPAX data specified)!");
137                return;
138        }
139        
140        HttpEntity httpEntity = meb.build();
141//      if(log.isDebugEnabled()) httpEntity.writeTo(System.err);
142        String content = Executor.newInstance()
143                        .execute(Request.Post(url).body(httpEntity))
144                                .returnContent().asString();    
145
146        //save: append to the output stream (file)
147                BufferedReader res = new BufferedReader(new StringReader(content));
148                String line;
149                PrintWriter writer = new PrintWriter(out);
150                while((line = res.readLine()) != null) {
151                        writer.println(line);
152                }
153                writer.flush();
154                res.close();
155    }

157    public void setUrl(String url) {
158                this.url = url;
159        }
160    
161    public String getUrl() {
162                return url;
163        }

165    /**
166     * Converts a biopax-validator XML response to the java object.
167     * 
168     * 
169     * @param xml
170     * @return
171     * @throws JAXBException
172     */
173    public static ValidatorResponse unmarshal(String xml) throws JAXBException {
174        JAXBContext jaxbContext = JAXBContext.newInstance("org.biopax.validator.jaxb");
175                Unmarshaller un = jaxbContext.createUnmarshaller();
176                Source src = new StreamSource(new StringReader(xml));
177                ValidatorResponse resp = un.unmarshal(src, ValidatorResponse.class).getValue();
178                return resp;
179    }

182    /**
183     * Checks BioPAX files using the online BioPAX Validator. 
184     * 
185     * @see <a href="http://www.biopax.org/validator">BioPAX Validator Webservice</a>
186     * 
187     * @param argv
188     * @throws IOException
189     */
190    public static void main(String[] argv) throws IOException 
191    {
192        if (argv.length == 0) {
193            System.err.println("Available parameters: \n" + 
194                "<path> <output> [xml|html|biopax] [auto-fix] [only-errors] [maxerrors=n] [notstrict]\n" +
195                "\t- validate a BioPAX file/directory (up to ~25MB in total size, -\n" +
196                "\totherwise, please use the biopax-validator.jar instead)\n" +
197                "\tin the directory using the online BioPAX Validator service\n" +
198                "\t(generates html or xml report, or gets the processed biopax\n" +
199                "\t(cannot fix all errros though) see http://www.biopax.org/validator)");
200            System.exit(-1);
201        }
202                
203        final String input = argv[0];
204        final String output = argv[1];
205        
206        File fileOrDir = new File(input);
207        if (!fileOrDir.canRead()) {
208            System.err.println("Cannot read from " + input);
209            System.exit(-1);
210        }        
211        if (output == null || output.isEmpty()) {
212            System.err.println("No output file specified (for the validation report).");
213            System.exit(-1);
214        }
215        
216        // default options
217        RetFormat outf = RetFormat.HTML;
218        boolean fix = false;
219        Integer maxErrs = null;
220        Behavior level = null; //will report both errors and warnings
221        String profile = null;
222        
223        // match optional arguments
224                for (int i = 2; i < argv.length; i++) {
225                        if ("html".equalsIgnoreCase(argv[i])) {
226                                outf = RetFormat.HTML;
227                        } else if ("xml".equalsIgnoreCase(argv[i])) {
228                                outf = RetFormat.XML;
229                        } else if ("biopax".equalsIgnoreCase(argv[i])) {
230                                outf = RetFormat.OWL;
231                        } else if ("auto-fix".equalsIgnoreCase(argv[i])) {
232                                fix = true;
233                        } else if ("only-errors".equalsIgnoreCase(argv[i])) {
234                                level = Behavior.ERROR;
235                        } else if ((argv[i]).toLowerCase().startsWith("maxerrors=")) {
236                                String num = argv[i].substring(10);
237                                maxErrs = Integer.valueOf(num);
238                        } else if ("notstrict".equalsIgnoreCase(argv[i])) {
239                                profile = "notstrict";
240                        }
241                }
242
243        // collect files
244        Collection<File> files = new HashSet<File>();
245        
246        if (fileOrDir.isDirectory()) {
247            // validate all the OWL files in the folder
248            FilenameFilter filter = new FilenameFilter() {
249                public boolean accept(File dir, String name) {
250                    return (name.endsWith(".owl"));
251                }
252            };
253            
254            for (String s : fileOrDir.list(filter)) {
255                files.add(new File(
256                        fileOrDir.getCanonicalPath() + File.separator + s));
257            }
258        } else {
259            files.add(fileOrDir);
260        }
261
262        // upload and validate using the default URL: http://www.biopax.org/biopax-validator/check.html        
263        if (!files.isEmpty()) {
264                BiopaxValidatorClient val = new BiopaxValidatorClient();
265                val.validate(fix, profile, outf, level, maxErrs, 
266                        null, files.toArray(new File[]{}), new FileOutputStream(output));
267        }
268    }
269    
270    private String location(final String url) throws IOException {
271        String location = url; //initially the same
272        // discover actual location, avoid going in circles:
273        int i=0;
274        for(String loc = url; loc != null && i<5; i++ ) 
275        {       
276                //do POST for location (Location header present if there's a 301/302/307 redirect on the way)
277                loc = Request.Post(loc).execute()
278                                .handleResponse(new ResponseHandler<String>() {
279                                                public String handleResponse(HttpResponse httpResponse)
280                                                                throws ClientProtocolException, IOException {
281                                                        Header header = httpResponse.getLastHeader("Location");
282//                                                      System.out.println("header=" + header);
283                                                        return (header != null) ? header.getValue().trim() : null;
284                                                }
285                        });
286                 
287                if(loc != null) {
288                        location = loc;         
289                        log.info("BioPAX Validator location: " + loc);
290                }
291        }
292        
293        return location;
294    }
