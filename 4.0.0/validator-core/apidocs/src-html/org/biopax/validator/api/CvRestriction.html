001package org.biopax.validator.api;
002
003/**
004 * @author rodch
005 */
006public final class CvRestriction {
007        
008        private final String id; //e.g., "GO:0005575"
009    private final String ontologyId; // "GO"
010        private final boolean allowed;
011    private final UseChildTerms childrenAllowed;
012    private final boolean not;
013        
014        public CvRestriction(String id, String ontologyId, boolean useThisTerm, 
015                        UseChildTerms useChildTerms, boolean isNot) {
016                this.id = id;
017                this.ontologyId = ontologyId;
018                this.allowed = useThisTerm;
019                this.childrenAllowed = useChildTerms;
020                this.not = isNot;
021        }
022        
023        public String getId() {
024                return id;
025        }
026
027        public String getOntologyId() {
028                return ontologyId;
029        }
030
031        public boolean isTermAllowed() {
032                return allowed;
033        }
034
035        
036        public UseChildTerms getChildrenAllowed() {
037                return childrenAllowed;
038        }
039        
040        public boolean isNot() {
041                return not;
042        }
043        
044        @Override
045        public String toString() {
046                return ((not)? "NOT allowed " : "Valid '") 
047                        + ontologyId + "' terms: " 
048                        + ((allowed)? id + " and " : "") 
049                        + childrenAllowed.toString() + " children of " + id;
050        }
051        
052        public enum UseChildTerms {
053                ALL, DIRECT, NONE;
054        }
055}
