25  import java.io.InputStream;
26  import java.util.Collection;
27  import java.util.Set;
28  
29  import org.biopax.validator.api.beans.Validation;
30  
31  /**
32   * BioPAX Validator interface
33   * 
34   * 
35   * @author rodch
36   *
37   */
38  public interface Validator {
39  	
40  	/**
41  	 * Get all the currently loaded BioPAX rules (beans).
42  	 * 
43  	 * @return
44  	 */
45  	Set<Rule<?>> getRules();
46  	
47  	
48  	/**
49  	 * Gets all the currently registered validation results.
50  	 * 
51  	 * @return
52  	 */
53  	Collection<Validation> getResults();
54  	
55  	
56  	/**
57  	 * Associates a validation report with the BioPAX data
58  	 * and creates the in-memory model (Paxtools Model).
59  	 * 
60  	 * The validation result is object where all the problems 
61  	 * related to this data instance are collected.
62  	 * 
63  	 * @param validation
64  	 * @param inputStream
65  	 */
66  	void importModel(Validation validation, InputStream inputStream);
67  
68  	/**
69  	 * Associates an object, such as BioPAX model, element, 
70  	 * InputStream; with the given validation result.
71  	 * 
72  	 * Some errors may occur during the data import
73  	 * but before the model is created, e.g., 
74  	 * those in the XML/RDF header or software bugs.
75  	 * And we want to report these problems as well.
76  	 * 
77  	 * @param element object (e.g., existing Model)
78  	 * @param validation 
79  	 */
80  	void associate(Object element, Validation validation);
81  	
82  	
83  	/**
84  	 * Associates, e.g., a property value with validation results
85  	 * by using the 'parent' element to find it. This allows for
86  	 * errors to be reported while model is being read,
87  	 * maybe before the child element is added to the model.
88  	 * 
89  	 * @param parent BioPAX element
90  	 * @param child value
91  	 */
92  	void indirectlyAssociate(Object parent, Object child);
93  
94  	
95  	/**
96  	 * Given the object, finds
97  	 * the corresponding validations
98  	 * (that it's been associated with).
99  	 * 
100 	 * @param obj a BioPAX element, Model, or even SimpleIOHandler
101 	 * @return
102 	 */
103 	Collection<Validation> findValidation(Object obj);
104 	
105 	/**
106 	 * Post-model validation. 
107 	 * 
108 	 * The result also includes other  
109 	 * problems that might occur during 
110 	 * the model initialization and creation.
111 	 * 
112 	 * @param validation
113 	 */
114 	void validate(Validation validation);
115 	
116 	/**
117      * Adds the validation error (with proper attributes)
118      * to registered validation objects associated with the object.
119 	 * 
120 	 * @param obj object associated with a validation result (can be even InputStream, during import, but usually is a BioPAX element)
121 	 * @param errorCode error code
122 	 * @param reportedBy class name of a validation rule or a name of another BioPAX validating class, method (e.g., AOP aspect's method/joinpoint name).
123 	 * @param isFixed if true, - find and set the attribute
124 	 * @param args
125 	 */
126 	void report(Object obj, String errorCode, String reportedBy, boolean isFixed, Object... args);
127 	
128 }
