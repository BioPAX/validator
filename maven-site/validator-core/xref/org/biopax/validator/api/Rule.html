25  import org.biopax.validator.api.beans.Validation;
26  
27  
28  /**
29   * Validation rule interface.
30   * 
31   * Although at run time T is a BioPAXElement
32   * (this is 100% true for the current version),
33   * at compile time, when implementing a rule, one may
34   * prefer to use other interfaces as well, e.g.,
35   * Process or Named. Therefore, "T extends BioPAXElement"
36   * is not used here.
37   *
38   * @author rodche
39   */
40  public interface Rule<T> {  
41  
42  	/**
43       * Validates the object.
44  	 * @param validation the object where the model, validation settings and errors are stored
45  	 * @param thing to validate
46  	 * @param fix try to fix the error case if found
47       */
48  	void check(Validation validation, T thing);
49  
50      /**
51       * Can check it?
52       *
53       * @param thing
54       * @return True when it can.
55       */
56      boolean canCheck(Object thing);
57      
58      
59      /**
60       * Saves the error or warning that occurred or was fixed.
61       * 
62       * Call this method from a validation rule implementation 
63       * every time after a problem is found and/or fixed. 
64  	 * 
65  	 * @param validation the object where the model, validation settings and errors are stored
66  	 * @param object that is invalid or caused the error
67  	 * @param code error code, e.g., 'illegal.value'
68  	 * @param setFixed 
69  	 * @param args extra parameters for the error message template
70  	 */
71      void error(Validation validation, Object object, String code, boolean setFixed, Object... args);
72      
73  }
