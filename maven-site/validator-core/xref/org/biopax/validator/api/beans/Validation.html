120 		if(profile != null && !profile.isEmpty())
121 			this.profile = profile;

217 	/**
218 	 * Adds or updates the error (with cases) to the errors collection.
219 	 * This is the primary method to save or update a just discovered or 
220 	 * fixed validation problem!
221 	 * 
222 	 * If the same 'error' already exists there 
223 	 * (i.e. type:code where type is either 'warning' or 'error'), 
224 	 * the new error cases will be copied to it;
225 	 * otherwise, the new one is simply added to the set.
226 	 * 
227 	 * It also updates the not-fixed errors counter,
228 	 * taking into account current validation threshold (level)
229 	 * and error's own type {@link Behavior} set by a validation rule
230 	 * that created it!
231 	 * 
232 	 * @see #setThreshold(Behavior)
233 	 * @see ErrorType#hashCode()
234 	 * @see ErrorType#equals(Object)
235 	 * 
236 	 * @param e Error type
237 	 */
238 	public synchronized void addError(ErrorType e) {	
239 		
240 		int numNotFixedErrors = countErrors(null, null, null, null, true, true);
241 
242 		if(isMaxErrorsSet() && numNotFixedErrors >= getMaxErrors())
243 		{
244 			log.info("Won't save the case: max. errors " +
245 				"limit exceeded for " + getDescription());
246 			return;
247 		}	
248 		
249 		switch (threshold) {
250 		case IGNORE:
251 			break; // do nothing
252 		case ERROR:
253 			if(e.getType() == Behavior.WARNING) {
254 				break; // do not add (only errors pass)
255 			}
256 		default: // add error with all cases
257 			synchronized (this) {
258 				if (error.contains(e)) {
259 					for (ErrorType et : error) {
260 						if (et.equals(e)) {
261 							et.addCases(e.getErrorCase());
262 							break;
263 						}
264 					}
265 				} else { // adding a new error type (code)
266 					error.add(e);
267 				}
268 			}
269 			break;
270 		}
271 	}

371 	 * Finds the same error type among
372 	 * existing (already happened to have error cases)
373 	 * 
374 	 * @see ErrorType#equals(Object)
375 	 * @see ErrorType#hashCode()
376 	 *
377 	 */
378 	private ErrorType findErrorType(ErrorType errorType) {
379 		if(getError().contains(errorType)) {
380 			for(ErrorType et: getError()) {
381 				if(et.equals(errorType)) {
382 					return et;
383 				}
384 			}
385 		} 
386 		return null;
387 	}

391 	 * Finds the same error case in the registry.
392 	 * 
393 	 * @see ErrorType#equals(Object)
394 	 * @see ErrorType#hashCode()
395 	 * @see ErrorCaseType#equals(Object)
396 	 * @see ErrorCaseType#hashCode()
397 	 *
398 	 */
399 	private ErrorCaseType findErrorCase(ErrorType errorType, ErrorCaseType errCase) {
400 		ErrorType etype = findErrorType(errorType);
401 		if (etype != null) {
402 			ErrorCaseType ecase = etype.findErrorCase(errCase);
403 			return ecase; //bug fixed ('return' was missing)
404 		}
405 		return null;
406 	}

409 	/**
410 	 * Counts the number of errors/warnings.
411 	 * Extra parameters are used to exclude 
412 	 * some of the cases.
413 	 * 
414 	 * @param forObject when 'null', counts all
415 	 * @param reportedBy when 'null', counts all
416 	 * @param code when 'null', counts all
417 	 * @param category when 'null', counts all
418 	 * @param ignoreWarnings do not count WARNINGs
419 	 * @param ignoreFixed do not count fixed
420 	 * @return number of error cases
421 	 */
422 	public int countErrors(String forObject, String reportedBy, 
423 			String code, Category category, boolean ignoreWarnings, boolean ignoreFixed) {
424 		int count = 0;
425 		
426 		for(ErrorType et : getError()) {
427 			// skip warnings?
428 			if(ignoreWarnings == true && et.getType() == Behavior.WARNING) {
429 				continue;
430 			}
431 			
432 			// skip other codes?
433 			if(code != null && !code.equalsIgnoreCase(et.getCode())) {
434 				continue;
435 			}
436 			
437 			// skip other categories?
438 			if(category != null && !(category == et.getCategory())) {
439 				continue;
440 			}
441 			
442 			count += et.countErrors(forObject, reportedBy, ignoreFixed);
443 		}
444 		
445 		return count;
446 	}

541 	 * Finds a previously reported
542 	 * by the rule error case and marks it as fixed.
543 	 * 
544 	 * @param objectId model element identifier (associated with the error)
545 	 * @param rule a validation rule name (that reports the error code)
546 	 * @param errCode specific error code
547 	 * @param newMsg a message, if not null/empty, to replace the original one
548 	 */	
549 	public void setFixed(String objectId, String rule, String errCode, String newMsg) 
550 	{
551 		Behavior type = Behavior.WARNING;
552 		ErrorCaseType ect = findErrorCase(new ErrorType(errCode, type), 
553 			new ErrorCaseType(rule, objectId, null)); // msg is ignored when comparing errors anyway
554 		
555 		if(ect == null) {
556 			type = Behavior.ERROR;
557 			ect = findErrorCase(
558 				new ErrorType(errCode, type), 
559 				new ErrorCaseType(rule, objectId, null));
560 		}
561 		
562 		if(ect != null && !ect.isFixed()) {
563 			ect.setFixed(true);
564 			if(newMsg != null && !"".equals(newMsg.trim()))
565 				ect.setMessage(newMsg);
566 		}
567 	}

652 	 * interface that contain the only method, which
653 	 * is called by the validator framework to get an
654 	 * identifier or name of a model element to report 
655 	 * an error about.

661 	 */

663 		
664 		/**
665 		 * Gets the id or name of a model element or 
666 		 * other object that matters during a validation
667 		 * and worth mentioning in a error message.
668 		 * 
669 		 * This is called by the Validator framework to get a
670 		 * domain specific identifier or name of a model element 
671 		 * to report an error about.
672 		 * 
673 		 * This interface is required, because a domain specific 'get id'
674 		 * method, if any, is not known to the core validation framework
675 		 * in advance, and toString method may not be satisfactory.
676 		 * 
677 		 * @param obj a model element or, e.g., stream
678 		 * @return id
679 		 */
