25  import java.io.Serializable;
26  import java.util.*;
27  
28  import javax.xml.bind.annotation.*;
29  
30  @XmlType(name="ErrorType")
31  @XmlAccessorType(XmlAccessType.FIELD)
32  public class ErrorType implements Serializable, Comparable<ErrorType> {
33  	
34  	private static final long serialVersionUID = 1L;
35  	
36  	/* 
37  	 * Collection of error cases 
38  	 * must have unique error types (identified by code).
39  	 * This is to be assured by implementing it as HashMap
40  	 * and overriding the 'equals' method. 
41  	 * 
42  	 */
43  	@XmlElement
44  	private final Set<ErrorCaseType> errorCase; 
45  	@XmlAttribute
46  	private String code = null;
47  	@XmlAttribute
48  	private String message = null;
49  	@XmlAttribute
50  	private Behavior type = Behavior.ERROR; // default
51  	@XmlAttribute
52  	private Category category = Category.INFORMATION; // default	
53  	@XmlAttribute
54  	private int notFixedCases = 0;
55  	@XmlAttribute
56  	private int totalCases = 0;
57  
58  	public ErrorType() {
59  		errorCase = new TreeSet<ErrorCaseType>();
60  	}
61  
62  	public ErrorType(String code, Behavior type) {
63  		this();
64  		this.code = code.toLowerCase();
65  		this.type = type;
66  	}
67  	
68  	public Category getCategory() {
69  		return category;
70  	}
71  
72  	public void setCategory(Category category) {
73  		this.category = category;
74  	}
75  	
76  	/**
77  	 * Unmodifiable set of validation current error cases.
78  	 * 
79  	 * @return
80  	 */
81  	public synchronized Collection<ErrorCaseType> getErrorCase() {
82  		return Collections.unmodifiableSet(errorCase);
83  	}
84  
85  	public synchronized void setErrorCase(Collection<ErrorCaseType> errorCases) {
86  		errorCase.clear();
87  		errorCase.addAll(errorCases);
88  	}
89  	
90  	public String getCode() {
91  		return code;
92  	}
93  
94  	public void setCode(String newCode) {
95  		code = newCode.toLowerCase();
96  	}
97  
98  	public String getMessage() {
99  		return message;
100 	}
101 	
102 	public void setMessage(String message) {
103 		this.message = message;
104 	}
105 	
106 	/**
107 	 * Adds (may be updates...) a error case;
108 	 * only the error message that happened last 
109 	 * (for the same object and rule) is kept
110 	 * (e.g., in a series of unsuccessful attempts 
111 	 * to set a biopax property, the last error message
112 	 * will override previous ones...)
113 	 * 
114 	 * @see ErrorCaseType#equals(Object)
115 	 * @see ErrorCaseType#hashCode()
116 	 * 
117 	 * @param newCase
118 	 */
119 	public synchronized void addErrorCase(ErrorCaseType newCase) {
120 		ErrorCaseType ect = findErrorCase(newCase);
121 		if(ect != null) { //found a previously reported case
122 			ect.setFixed(newCase.fixed);
123 			if(!newCase.fixed) {		
124 				// update the message
125 				ect.setMessage(newCase.getMessage());
126 			}
127 		} else { //fresh error case
128 			errorCase.add(newCase);
129 		}
130 	}
131 	
132 	/**
133 	 * Adds error cases
134 	 * 
135 	 * @param cases
136 	 */
137 	public void addCases(Collection<ErrorCaseType> cases) {
138 		for (ErrorCaseType errorCase : cases) {
139 			addErrorCase(errorCase);
140 		}
141 	}
142 		
143 	public synchronized void removeErrorCase(ErrorCaseType eCase) {
144 		errorCase.remove(eCase);
145 	}
146 
147 	
148 	@Override
149 	public String toString() {
150 		return type + " " + code; 
151 	}
152 	
153 	@Override
154 	public boolean equals(Object obj) {	
155 		if(obj instanceof ErrorType) {
156 			ErrorType that = (ErrorType) obj;
157 			return type.equals(that.type)
158 					&& code.equals(that.code);
159 		} 
160 		return false;
161 	}
162 	
163 	@Override
164 	public int hashCode() {
165 		int result = 31 + (type != null ? type.hashCode() : 0);
166 		result = 31 * result + (code != null ? code.hashCode() : 0);
167 		return result;
168 	}
169 	
170 	public int compareTo(ErrorType et) {
171 		return this.toString().compareToIgnoreCase(et.toString());
172 	}
173 	
174 	public Behavior getType() {
175 		return type;
176 	}
177 	
178 	public void setType(Behavior type) {
179 		this.type = type;
180 	}
181 	
182 	/**
183 	 * Total number of cases registered, including fixed.
184 	 * 
185 	 * @return
186 	 */
187 	public int getTotalCases() {
188 		return totalCases;
189 	}
190 	public void setTotalCases(int n) {
191 		this.totalCases = n;
192 	}
193 	
194 	/**
195 	 * Total number of cases not fixed yet.
196 	 * 
197 	 * @return
198 	 */
199 	public int getNotFixedCases() {
200 		return notFixedCases;
201 	}
202 	public void setNotFixedCases(int n) {
203 		this.notFixedCases = n;
204 	}
205 	
206 	/**
207 	 * Count current error cases.
208 	 * 
209 	 * @param forObject if 'null', count everything
210 	 * @param reportedBy if 'null', everything's counted
211 	 * @param ignoreFixed skip fixed if true
212 	 * @return
213 	 */
214 	public int countErrors(String forObject, String reportedBy, boolean ignoreFixed) {
215 		int count = 0;
216 		
217 		for(ErrorCaseType ec: errorCase) {
218 			if(forObject != null && !forObject.equals(ec.getObject())) {
219 				continue;
220 			}
221 			
222 			if(reportedBy != null && !reportedBy.equals(ec.getReportedBy())) {
223 				continue;
224 			}
225 			
226 			if(ignoreFixed == true && ec.isFixed()) {
227 				continue;
228 			}
229 			
230 			count++;
231 		}
232 		
233 		return count;
234 	}
235 	
236 	/**
237 	 * Returns the existing (happened) error case object by
238 	 * ErrorCaseType (new, not yet reported, or a copy).
239 	 * Note: 'equals' method is overridden in ErrorCaseType
240 	 * 
241 	 * @see ErrorCaseType#equals(Object) 
242 	 * 
243 	 * @param searchBy
244 	 * @return
245 	 */
246 	public synchronized ErrorCaseType findErrorCase(final ErrorCaseType searchBy) {
247 		if(errorCase.contains(searchBy)) {
248 			for (ErrorCaseType ec : getErrorCase()) {
249 				if (ec.equals(searchBy)) {
250 					return ec;
251 				}
252 			}
253 		}
254 		return null;
255 	}
