025import java.io.InputStream;
026import java.util.Collection;
027import java.util.Set;
028
029import org.biopax.validator.api.beans.Validation;
030
031/**
032 * BioPAX Validator interface
033 * 
034 * 
035 * @author rodch
036 *
037 */
038public interface Validator {
039        
040        /**
041         * Get all the currently loaded BioPAX rules (beans).
042         * 
043         * @return
044         */
045        Set<Rule<?>> getRules();
046        
047        
048        /**
049         * Gets all the currently registered validation results.
050         * 
051         * @return
052         */
053        Collection<Validation> getResults();
054        
055        
056        /**
057         * Associates a validation report with the BioPAX data
058         * and creates the in-memory model (Paxtools Model).
059         * 
060         * The validation result is object where all the problems 
061         * related to this data instance are collected.
062         * 
063         * @param validation
064         * @param inputStream
065         */
066        void importModel(Validation validation, InputStream inputStream);
067
068        /**
069         * Associates an object, such as BioPAX model, element, 
070         * InputStream; with the given validation result.
071         * 
072         * Some errors may occur during the data import
073         * but before the model is created, e.g., 
074         * those in the XML/RDF header or software bugs.
075         * And we want to report these problems as well.
076         * 
077         * @param element object (e.g., existing Model)
078         * @param validation 
079         */
080        void associate(Object element, Validation validation);
081        
082        
083        /**
084         * Associates, e.g., a property value with validation results
085         * by using the 'parent' element to find it. This allows for
086         * errors to be reported while model is being read,
087         * maybe before the child element is added to the model.
088         * 
089         * @param parent BioPAX element
090         * @param child value
091         */
092        void indirectlyAssociate(Object parent, Object child);
093
094        
095        /**
096         * Given the object, finds
097         * the corresponding validations
098         * (that it's been associated with).
099         * 
100         * @param obj a BioPAX element, Model, or even SimpleIOHandler
101         * @return
102         */
103        Collection<Validation> findValidation(Object obj);
104        
105        /**
106         * Post-model validation. 
107         * 
108         * The result also includes other  
109         * problems that might occur during 
110         * the model initialization and creation.
111         * 
112         * @param validation
113         */
114        void validate(Validation validation);
115        
116        /**
117     * Adds the validation error (with proper attributes)
118     * to registered validation objects associated with the object.
119         * 
120         * @param obj object associated with a validation result (can be even InputStream, during import, but usually is a BioPAX element)
121         * @param errorCode error code
122         * @param reportedBy class name of a validation rule or a name of another BioPAX validating class, method (e.g., AOP aspect's method/joinpoint name).
123         * @param isFixed if true, - find and set the attribute
124         * @param args
125         */
126        void report(Object obj, String errorCode, String reportedBy, boolean isFixed, Object... args);
127        
128}
