001package org.biopax.validator.api;
002
003/*
004 *
005 */
006
007import java.util.Collection;
008import java.util.Set;
009
010import org.biopax.psidev.ontology_manager.OntologyTermI;
011
012
013public interface CvValidator {
014
015        /**
016         * Gets valid ontology term names
017         * using the constraints from the rule bean.
018         * 
019         * @param cvRule controlled vocabulary validation rule
020         * @return recommended terms
021         */
022        Set<String> getValidTermNames(CvRule<?> cvRule);
023
024        /**
025         * Gets valid ontology terms
026         * using the constraints from the rule bean.
027         * 
028         * @param cvRule controlled vocabulary validation rule
029         * @return a set of ontology terms (beans)
030         */
031        Set<OntologyTermI> getValidTerms(CvRule<?> cvRule);
032
033        /**
034         * Gets the set of terms (including synonyms) 
035         * that satisfy all the restrictions.
036         * 
037         * @param restrictions - objects that specify required ontology terms
038         * @return set of names (strings)
039         */
040        Set<String> getValidTermNames(Collection<CvRestriction> restrictions);
041
042        /**
043         * Similar to getValidTermNames method, 
044         * but the term names in the result set are all in lower case.
045         * 
046         * @see #getValidTermNames(Collection)
047         * 
048         * @param restrictions controlled vocabulary restrictions (on type, context, ontology terms)
049         * @return set of valid ontology terms (turned into lower case)
050         */
051        Set<String> getValidTermNamesLowerCase(
052                        Collection<CvRestriction> restrictions);
053
054        /**
055         * Gets term names and synonyms using the 
056         * restriction bean to filter the data.
057         * (restriction's 'NOT' property is ignored here)
058         * 
059         * @param restriction a restriction (on type, context, ontology terms)
060         * @return recommended ontology terms (names)
061         */
062        Set<String> getTermNames(CvRestriction restriction);
063
064        /**
065         * Gets a restricted set of CVs 
066         * (including for synonyms) that satisfy
067         * all the restrictions in the set.
068         * 
069         * @param restrictions - set of beans that together define the required constraint
070         * @return set of ontology terms
071         */
072        Set<OntologyTermI> getValidTerms(Collection<CvRestriction> restrictions);
073
074        /**
075         * Gets CVs (including for synonyms) using the 
076         * criteria defined by the bean
077         * ('NOT' property, if set 'true', is ignored)
078         * 
079         * @param restriction restriction (on type, context, ontology terms)
080         * @return ontology terms (objects)
081         */
082        Set<OntologyTermI> getTerms(CvRestriction restriction);
083
084}
