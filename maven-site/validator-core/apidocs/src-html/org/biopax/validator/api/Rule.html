001package org.biopax.validator.api;
002
003import org.biopax.validator.api.beans.Validation;
004
005
006/**
007 * Validation rule interface.
008 * Although at run time T is a BioPAXElement
009 * (this is 100% true for the current version),
010 * at compile time, when implementing a rule, one may
011 * prefer to use other interfaces as well, e.g.,
012 * Process or Named. Therefore, "T extends BioPAXElement"
013 * is not used here.
014 *
015 * @author rodche
016 */
017public interface Rule<T> {
018
019  /**
020   * Validates the object.
021   *
022   * @param validation the object where the model, validation settings and errors are stored
023   * @param thing      to validate
024   */
025  void check(Validation validation, T thing);
026
027  /**
028   * Can check it?
029   *
030   * @param thing an object to validate
031   * @return True when this rule can check the object.
032   */
033  boolean canCheck(Object thing);
034
035
036  /**
037   * Saves the error or warning that occurred or was fixed.
038   * Call this method from a validation rule implementation
039   * every time after a problem is found and/or fixed.
040   *
041   * @param validation the object where the model, validation settings and errors are stored
042   * @param object     that is invalid or caused the error
043   * @param code       error code, e.g., 'illegal.value'
044   * @param setFixed   true/false - whether the issue was auto-fixed or not
045   * @param args       extra parameters for the error message template
046   */
047  void error(Validation validation, Object object, String code, boolean setFixed, Object... args);
048
049}
