001package org.biopax.validator.api;
002
003/*
004 *
005 */
006
007import java.io.PrintWriter;
008import java.io.StringWriter;
009
010import javax.xml.stream.XMLStreamException;
011
012import org.apache.commons.logging.Log;
013import org.apache.commons.logging.LogFactory;
014import org.springframework.beans.factory.annotation.Autowired;
015
016/**
017 * Basic class for the framework classes 
018 * that use AOP (and LTW) to check BioPAX
019 * and report errors.
020 *
021 * @author rodche
022 */
023public abstract class AbstractAspect {
024        private static final Log log = LogFactory.getLog(AbstractAspect.class);
025        
026        @Autowired
027        protected Validator validator;
028
029        @Autowired
030        protected ValidatorUtils utils;
031        
032        
033    /**
034     * Registers the error in the validator.
035     * 
036     * @param object a model element or parser/reader (e.g., InputStream) associated with the issue
037     * @param errorCode error code
038     * @param reportedBy validation rule class name
039     * @param setFixed whether the problem was auto-fixed or not
040     * @param msgArgs additional error message parameters
041     */
042    public void report(Object object, String errorCode, 
043                String reportedBy, boolean setFixed, Object... msgArgs) 
044    {   
045        validator.report(object, errorCode, reportedBy, setFixed, msgArgs);
046        }
047    
048    
049        /**
050         * Registers other (external) exceptions.
051         * 
052         * The exception class, i.e., simple name in lower case, 
053         * is used as the error code, and the 'object' is to
054         * find the corresponding validation result where this 
055         * problem should be added.
056         * 
057         * This must be public method (for unclear reason, otherwise causes an AOP exception...)
058         * 
059         * @param t the exception
060         * @param obj model, element, or another related to the BioPAX data object
061         * @param errorCode validator error code
062         * @param reportedBy validator rule name
063         * @param details extra message to be added at the end of the original error message if not null
064         */
065    public void reportException(Throwable t, Object obj, String errorCode, String reportedBy, String details) {
066        
067                StringBuilder msg = new StringBuilder(t.toString());
068                
069                if(t instanceof XMLStreamException) {
070                        XMLStreamException ex = (XMLStreamException) t;
071                        msg.append("; ").append(ex.getLocation().toString());
072                } 
073                else {
074                        if("exception".equals(errorCode)) //catch a bug
075                                msg.append(" - stack:").append(getStackTrace(t)).append(" - ");
076                }
077                
078                if(details != null) 
079                        msg.append("; ").append(details);
080                
081                if (validator != null) {
082                        validator.report(obj, errorCode, reportedBy, false, msg.toString());
083                } else {
084                        log.error("utils is null (not initialized?); skipping " +
085                                        "an intercepted 'syntax.error': " + msg.toString() 
086                                        + " reported by: " + reportedBy);
087                }
088        }
089    
090    private static String getStackTrace(Throwable aThrowable) {
091        StringWriter strw = new StringWriter();
092        PrintWriter printWriter = new PrintWriter(strw);
093        aThrowable.printStackTrace(printWriter);
094        return strw.toString();
095    }
096}
