025import java.io.PrintWriter;
026import java.io.StringWriter;
027
028import javax.xml.stream.XMLStreamException;
029
030import org.apache.commons.logging.Log;
031import org.apache.commons.logging.LogFactory;
032import org.springframework.beans.factory.annotation.Autowired;
033
034/**
035 * Basic class for the framework classes 
036 * that use AOP (and LTW) to check BioPAX
037 * and report errors.
038 *
039 * @author rodche
040 */
041public abstract class AbstractAspect {
042        private static final Log log = LogFactory.getLog(AbstractAspect.class);
043        
044        @Autowired
045        protected Validator validator;
046
047        @Autowired
048        protected ValidatorUtils utils;
049        
050        
051    /**
052     * Registers the error in the validator.
053     * 
054     * @param object a model element or parser/reader (e.g., InputStream) associated with the issue
055     * @param errorCode
056     * @param reportedBy validation rule class name
057     * @param setFixed
058     * @param msgArgs
059     */
060    public void report(Object object, String errorCode, 
061                String reportedBy, boolean setFixed, Object... msgArgs) 
062    {   
063        validator.report(object, errorCode, reportedBy, setFixed, msgArgs);
064        }
065    
066    
067        /**
068         * Registers other (external) exceptions.
069         * 
070         * The exception class, i.e., simple name in lower case, 
071         * is used as the error code, and the 'object' is to
072         * find the corresponding validation result where this 
073         * problem should be added.
074         * 
075         * This must be public method (for unclear reason, otherwise causes an AOP exception...)
076         * 
077         * @param t
078         * @param obj model, element, or another related to the BioPAX data object
079         * @param errorCode
080         * @param reportedBy 
081         * @param details extra message to be added at the end of the original error message if not null
082         */
083    public void reportException(Throwable t, Object obj, String errorCode, String reportedBy, String details) {
084        
085                StringBuilder msg = new StringBuilder(t.toString());
086                
087                if(t instanceof XMLStreamException) {
088                        XMLStreamException ex = (XMLStreamException) t;
089                        msg.append("; ").append(ex.getLocation().toString());
090                } 
091                else {
092                        if("exception".equals(errorCode)) //catch a bug
093                                msg.append(" - stack:").append(getStackTrace(t)).append(" - ");
094                }
095                
096                if(details != null) 
097                        msg.append("; ").append(details);
098                
099                if (validator != null) {
100                        validator.report(obj, errorCode, reportedBy, false, msg.toString());
101                } else {
102                        log.error("utils is null (not initialized?); skipping " +
103                                        "an intercepted 'syntax.error': " + msg.toString() 
104                                        + " reported by: " + reportedBy);
105                }
106        }
107    
108    private static String getStackTrace(Throwable aThrowable) {
109        StringWriter strw = new StringWriter();
110        PrintWriter printWriter = new PrintWriter(strw);
111        aThrowable.printStackTrace(printWriter);
112        return strw.toString();
113    }
114}
