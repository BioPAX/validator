047/**
048 * Simple (example) BioPAX Validator client 
049 * to upload and check BioPAX OWL files.
050 * 
051 * @author rodche
052 *
053 */
054public class BiopaxValidatorClient {
055        private static final Log log = LogFactory.getLog(BiopaxValidatorClient.class);
056        
057        /**
058         * Default BioPAX Validator's URL
059         */
060        public static final String 
061                DEFAULT_VALIDATOR_URL = "http://www.biopax.org/validator/check.html";
062        
063        /**
064         * The Java Option to set a BioPAX Validator URL 
065         * (if set, it overrides the default as well as a value set via the Constructor)
066         */
067        public static final String JVM_PROPERTY_URL = "biopax.validator.url";
068        
069        public static enum RetFormat {
070                HTML,// errors as HTML/Javascript 
071                XML, // errors as XML
072                OWL; // modified BioPAX only (when 'autofix' or 'normalize' is true)
073        }
074        
075        private String url;
076
077        
078    /**
079     * Main Constructor
080     * 
081     * It configures for the validator's URL
082     * (defined by DEFAULT_VALIDATOR_URL constant)
083     * and result format ().
084     * 
085     * @param url - validator's file-upload form address
086     */
087    public BiopaxValidatorClient(String url) {
088                // 1) use the arg (if not empty/null) or the default URL
089        this.url = (url == null || url.isEmpty())
090                        ? DEFAULT_VALIDATOR_URL : url;
091        
092        // 2) override if the JVM option is set to another value
093        this.url = System.getProperty(JVM_PROPERTY_URL, this.url);
094                
095        // 3) get actual location (force through redirects, if any)
096                try {
097                        this.url = location(this.url);
098                } catch (IOException e) {
099                        log.warn("Failed to resolve to actual web service " +
100                                "URL using: " + url + " (if there is a 301/302/307 HTTP redirect, " +
101                                        "then validation requests (using HTTP POST method) will probably fail...)", e);
102                }
103        }
104

106    /**
107     * Default Constructor
108     * 
109     * It configures for the default validator URL.
110     */
111    public BiopaxValidatorClient() {
112        this(null);
113        }

116    /**
117     * Checks a BioPAX OWL file(s) or resource 
118     * using the online BioPAX Validator 
119     * and prints the results to the output stream.
120     * 
121     * @param autofix true/false (experimental)
122     * @param profile validation profile name
123     * @param retFormat xml, html, or owl (no errors, just modified owl, if autofix=true)
124     * @param biopaxUrl check the BioPAX at the URL
125     * @param biopaxFiles an array of BioPAX files to validate
126     * @param out
127     * @throws IOException
128     */
129    public void validate(boolean autofix, String profile, RetFormat retFormat, Behavior filterBy,
130                Integer maxErrs, String biopaxUrl, File[] biopaxFiles, OutputStream out) throws IOException 
131    {
132        MultipartEntityBuilder meb = MultipartEntityBuilder.create();
133        meb.setCharset(Charset.forName("UTF-8"));
134        
135        if(autofix)
136                meb.addTextBody("autofix", "true");
137
138                //TODO add extra options (normalizer.fixDisplayName, normalizer.xmlBase)?
139
140                if(profile != null && !profile.isEmpty())
141                meb.addTextBody("profile", profile);
142        if(retFormat != null)
143                meb.addTextBody("retDesired", retFormat.toString().toLowerCase());
144        if(filterBy != null)
145                meb.addTextBody("filter", filterBy.toString());
146        if(maxErrs != null && maxErrs > 0)
147                meb.addTextBody("maxErrors", maxErrs.toString());
148        if(biopaxFiles != null && biopaxFiles.length > 0)
149                for (File f : biopaxFiles) //important: use MULTIPART_FORM_DATA content-type
150                        meb.addBinaryBody("file", f, ContentType.MULTIPART_FORM_DATA, f.getName());
151        else if(biopaxUrl != null) {
152                meb.addTextBody("url", biopaxUrl);
153        } else {
154                log.error("Nothing to do (no BioPAX data specified)!");
155                return;
156        }
157        
158        HttpEntity httpEntity = meb.build();
159//      if(log.isDebugEnabled()) httpEntity.writeTo(System.err);
160        String content = Executor.newInstance()
161                        .execute(Request.Post(url).body(httpEntity))
162                                .returnContent().asString();    
163
164        //save: append to the output stream (file)
165                BufferedReader res = new BufferedReader(new StringReader(content));
166                String line;
167                PrintWriter writer = new PrintWriter(out);
168                while((line = res.readLine()) != null) {
169                        writer.println(line);
170                }
171                writer.flush();
172                res.close();
173    }

175    public void setUrl(String url) {
176                this.url = url;
177        }
178    
179    public String getUrl() {
180                return url;
181        }

183    /**
184     * Converts a biopax-validator XML response to the java object.
185     * 
186     * 
187     * @param xml
188     * @return
189     * @throws JAXBException
190     */
191    public static ValidatorResponse unmarshal(String xml) throws JAXBException {
192        JAXBContext jaxbContext = JAXBContext.newInstance("org.biopax.validator.jaxb");
193                Unmarshaller un = jaxbContext.createUnmarshaller();
194                Source src = new StreamSource(new StringReader(xml));
195                ValidatorResponse resp = un.unmarshal(src, ValidatorResponse.class).getValue();
196                return resp;
197    }

200    /**
201     * Checks BioPAX files using the online BioPAX Validator. 
202     * 
203     * @see <a href="http://www.biopax.org/validator">BioPAX Validator Webservice</a>
204     * 
205     * @param argv
206     * @throws IOException
207     */
208    public static void main(String[] argv) throws IOException 
209    {
210        if (argv.length == 0) {
211            System.err.println("Available parameters: \n" + 
212                "<path> <output> [xml|html|biopax] [auto-fix] [only-errors] [maxerrors=n] [notstrict]\n" +
213                "\t- validate a BioPAX file/directory (up to ~25MB in total size, -\n" +
214                "\totherwise, please use the biopax-validator.jar instead)\n" +
215                "\tin the directory using the online BioPAX Validator service\n" +
216                "\t(generates html or xml report, or gets the processed biopax\n" +
217                "\t(cannot fix all errros though) see http://www.biopax.org/validator)");
218            System.exit(-1);
219        }
220                
221        final String input = argv[0];
222        final String output = argv[1];
223        
224        File fileOrDir = new File(input);
225        if (!fileOrDir.canRead()) {
226            System.err.println("Cannot read from " + input);
227            System.exit(-1);
228        }        
229        if (output == null || output.isEmpty()) {
230            System.err.println("No output file specified (for the validation report).");
231            System.exit(-1);
232        }
233        
234        // default options
235        RetFormat outf = RetFormat.HTML;
236        boolean fix = false;
237        Integer maxErrs = null;
238        Behavior level = null; //will report both errors and warnings
239        String profile = null;
240        
241        // match optional arguments
242                for (int i = 2; i < argv.length; i++) {
243                        if ("html".equalsIgnoreCase(argv[i])) {
244                                outf = RetFormat.HTML;
245                        } else if ("xml".equalsIgnoreCase(argv[i])) {
246                                outf = RetFormat.XML;
247                        } else if ("biopax".equalsIgnoreCase(argv[i])) {
248                                outf = RetFormat.OWL;
249                        } else if ("auto-fix".equalsIgnoreCase(argv[i])) {
250                                fix = true;
251                        } else if ("only-errors".equalsIgnoreCase(argv[i])) {
252                                level = Behavior.ERROR;
253                        } else if ((argv[i]).toLowerCase().startsWith("maxerrors=")) {
254                                String num = argv[i].substring(10);
255                                maxErrs = Integer.valueOf(num);
256                        } else if ("notstrict".equalsIgnoreCase(argv[i])) {
257                                profile = "notstrict";
258                        }
259                }
260
261        // collect files
262        Collection<File> files = new HashSet<File>();
263        
264        if (fileOrDir.isDirectory()) {
265            // validate all the OWL files in the folder
266            FilenameFilter filter = new FilenameFilter() {
267                public boolean accept(File dir, String name) {
268                    return (name.endsWith(".owl"));
269                }
270            };
271            
272            for (String s : fileOrDir.list(filter)) {
273                files.add(new File(
274                        fileOrDir.getCanonicalPath() + File.separator + s));
275            }
276        } else {
277            files.add(fileOrDir);
278        }
279
280        // upload and validate using the default URL: http://www.biopax.org/biopax-validator/check.html        
281        if (!files.isEmpty()) {
282                BiopaxValidatorClient val = new BiopaxValidatorClient();
283                val.validate(fix, profile, outf, level, maxErrs, 
284                        null, files.toArray(new File[]{}), new FileOutputStream(output));
285        }
286    }
287    
288    private String location(final String url) throws IOException {
289        String location = url; //initially the same
290        // discover actual location, avoid going in circles:
291        int i=0;
292        for(String loc = url; loc != null && i<5; i++ ) 
293        {       
294                //do POST for location (Location header present if there's a 301/302/307 redirect on the way)
295                loc = Request.Post(loc).execute()
296                                .handleResponse(new ResponseHandler<String>() {
297                                                public String handleResponse(HttpResponse httpResponse)
298                                                                throws ClientProtocolException, IOException {
299                                                        Header header = httpResponse.getLastHeader("Location");
300//                                                      System.out.println("header=" + header);
301                                                        return (header != null) ? header.getValue().trim() : null;
302                                                }
303                        });
304                 
305                if(loc != null) {
306                        location = loc;         
307                        log.info("BioPAX Validator location: " + loc);
308                }
309        }
310        
311        return location;
312    }
