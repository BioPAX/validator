025import org.biopax.paxtools.model.level3.BindingFeature;
026import org.biopax.paxtools.model.level3.Complex;
027import org.biopax.paxtools.model.level3.ComplexAssembly;
028import org.biopax.paxtools.model.level3.Entity;
029import org.biopax.paxtools.model.level3.PhysicalEntity;
030import org.biopax.paxtools.util.ClassFilterSet;
031import org.biopax.validator.api.AbstractRule;
032import org.biopax.validator.api.beans.Validation;
033import org.springframework.stereotype.Component;
034
035import java.util.Collections;
036import java.util.Set;
037
038/**
039 * This rule checks if either ComplexAssembly conversion has a complex on one side,
040 * or, at least, several participants are bound to each other on either left or right side. 
041 */
042@Component
043public class ComplexAssemblyHasComplexParticipantRule extends AbstractRule<ComplexAssembly>
044{
045    public void check(final Validation validation, ComplexAssembly complexAssembly)
046    {
047                // check if there are any complexes on either side
048        Set<Complex> complexes = 
049                        new ClassFilterSet<Entity,Complex>(complexAssembly.getParticipant(), Complex.class);
050                if(complexes.isEmpty()) {
051                        boolean bound = isBound(complexAssembly.getRight());
052                        if(!bound)
053                                bound = isBound(complexAssembly.getLeft());
054                        if(!bound)
055                                error(validation, complexAssembly, "complex.not.present", false);
056                }
057    }
058
059    /* check "whether participants acquire a binding feature" 
060     * would be much more difficult than - "several participants are bound"
061     * (e.g., because the former is subject to "the same entityReference" cond...)
062     */
063    private boolean isBound(Set<PhysicalEntity> participants) {
064        for(PhysicalEntity pe : participants) {
065                //if(pe instanceof Complex) continue; // it's impossible as far isBound called from where it is called
066                if(pe.getFeature() != null 
067                                && pe.getFeature() instanceof BindingFeature) {
068                        BindingFeature bf1 = (BindingFeature) pe.getFeature();
069                        BindingFeature bf2 = bf1.getBindsTo();
070                        if(bf2 != null) {
071                                Set<PhysicalEntity> bf2Of = bf2.getFeatureOf();
072                                //if these phys. entities are (same side) participants as well
073                                if(!Collections.disjoint(participants, bf2Of))
074                                        return true;
075                        }
076                }
077        }
078        return false;
079        }
080
081        public boolean canCheck(Object thing) {
082        return thing instanceof ComplexAssembly;
083    }
084
085}
