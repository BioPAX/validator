025import org.biopax.paxtools.model.level3.Catalysis;
026import org.biopax.paxtools.model.level3.Control;
027import org.biopax.paxtools.model.level3.ControlType;
028import org.biopax.paxtools.model.level3.TemplateReactionRegulation;
029import org.biopax.validator.api.AbstractRule;
030import org.biopax.validator.api.beans.Validation;
031import org.springframework.stereotype.Component;
032
033/**
034 * Checks:
035 * 
036 * Catalysis.controlType - only "ACTIVATION"
037 * TemplateReactionRegulation.controlType - either "ACTIVATION" or "INHIBITION"
038 *
039 * @author rodche
040 */
041@Component
042public class ControlTypeRule extends AbstractRule<Control> {
043
044        private void fix(Control t, Object... values) {
045                if(t instanceof Catalysis) {
046                        ((Catalysis)t).setControlType(ControlType.ACTIVATION);
047                } else if (t instanceof TemplateReactionRegulation){
048                        if(values.length >0 && values[0] instanceof ControlType) {
049                                ControlType ct = (ControlType) values[0];
050                                ((TemplateReactionRegulation)t).setControlType(ct);
051                        } else {
052                                ((TemplateReactionRegulation)t).setControlType(null);
053                                if(logger.isInfoEnabled()) 
054                                        logger.info(t.getUri() +
055                                                        " - invalid controlType value deleted");
056                        }
057                } else {
058                        logger.error("This does not auto-fix " +
059                                        " controlType property of "
060                                        + t.getModelInterface().getSimpleName());
061                }
062        }
063
064        public boolean canCheck(Object thing) {
065                return thing instanceof Control;
066        }
067
068        public void check(final Validation validation, Control thing) {
069                if(thing.getControlType() != null) 
070                {                       
071                        if(thing instanceof Catalysis) {
072                                Catalysis cat = (Catalysis) thing;
073                                if(cat.getControlType() != ControlType.ACTIVATION) {
074                                        error(validation, thing, "range.violated", 
075                                                        validation.isFix(), "controlType",
076                                                        cat.getControlType().name(), "", ControlType.ACTIVATION.name()
077                                                        + " (or empty)");
078                                        if(validation.isFix()) {
079                                                fix(thing);
080                                        }
081                                }
082                        } else if(thing instanceof TemplateReactionRegulation) {
083                                TemplateReactionRegulation trr = (TemplateReactionRegulation) thing;
084                                if(! (trr.getControlType() == ControlType.ACTIVATION
085                                                || trr.getControlType() == ControlType.INHIBITION) ) 
086                                {
087                                        error(validation, thing, "range.violated", 
088                                                        validation.isFix(), "controlType",
089                                                        trr.getControlType().name(), "", ControlType.ACTIVATION.name() + " or " 
090                                                        + ControlType.INHIBITION.name()
091                                                        + " (or empty)");
092                                        if(validation.isFix()) {
093                                                fix(thing);
094                                        }
095                                }
096                        }
097                }
098        }
099        
100}
