025import java.util.Collection;
026import java.util.HashSet;
027import java.util.Set;
028
029import org.biopax.paxtools.model.level3.Complex;
030import org.biopax.paxtools.model.level3.PhysicalEntity;
031import org.biopax.validator.api.AbstractRule;
032import org.biopax.validator.api.beans.Validation;
033import org.springframework.stereotype.Component;
034
035
036/**
037 * Checks:
038 * Component's cellularLocation is empty
039 * when complex'es is not.
040 * 
041 * @author rodche
042 *
043 * TODO check for "same location" taking into account the cell compartments hierarchy?
044 */
045@Component
046public class ComplexComponentsSameLocationRule extends AbstractRule<Complex> {
047
048        public boolean canCheck(Object thing) {
049                return thing instanceof Complex;
050        }
051
052        public void check(final Validation validation, Complex thing) {
053                if(thing.getCellularLocation() != null) {
054                        Set<PhysicalEntity> ents = thing.getComponent();
055                        if (ents != null) {
056                                Collection<String> comps = new HashSet<String>();
057                                for (PhysicalEntity e : ents) {
058                                        if (e.getCellularLocation() == null) {
059                                                comps.add(e + " " + e.getCellularLocation());           
060                                        }
061                                }
062                                
063                                if(!comps.isEmpty()) {
064                                        error(validation, thing, 
065                                                        "component.no.location", false, thing.getCellularLocation().toString(), comps);
066                                }
067                        }
068                }
069        }
070
071}
