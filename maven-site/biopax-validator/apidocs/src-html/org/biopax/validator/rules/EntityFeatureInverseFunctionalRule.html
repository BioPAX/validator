001package org.biopax.validator.rules;
002
003import org.biopax.paxtools.controller.ShallowCopy;
004import org.biopax.paxtools.model.Model;
005import org.biopax.paxtools.model.level3.EntityFeature;
006import org.biopax.paxtools.model.level3.EntityReference;
007import org.biopax.paxtools.model.level3.SimplePhysicalEntity;
008import org.biopax.paxtools.normalizer.Normalizer;
009import org.biopax.validator.api.AbstractRule;
010import org.biopax.validator.api.beans.Validation;
011import org.springframework.stereotype.Component;
012
013import java.util.HashSet;
014import java.util.Set;
015
016/**
017 *
018 * A rule that can check and fix the error cases when an EntityFeature 
019 * is owned by multiple EntityReferences, which violates the entityFeature 
020 * property's 'inverse functional' BioPAX L3 OWL constraint.
021 *
022 */
023@Component
024public class EntityFeatureInverseFunctionalRule extends AbstractRule<Model>{
025
026    public void check(final Validation validation, Model model) {
027        Set<EntityFeature> efs =  new HashSet<EntityFeature>(model.getObjects(EntityFeature.class));//defensive copy
028        Set<EntityReference> ers =  new HashSet<EntityReference>(model.getObjects(EntityReference.class));
029        
030        for(EntityFeature ef: efs) {                    
031                EntityReference efOf = ef.getEntityFeatureOf(); //the last assigned or the only owner ER of this EF
032                //if the model does not contain the efOf object, no worries, we'll check this rule anyway                               
033                Set<EntityReference> ownerEntityRefs = new HashSet<EntityReference>();
034                for(EntityReference er : ers) {
035                        if(!er.getEntityFeature().contains(ef))
036                                continue; //skip
037                        
038                        ownerEntityRefs.add(er); //owner ER found!
039                        
040                        if(er.equals(efOf)) {
041                                continue; //no fix required for this one
042                        }
043                        
044                        if(validation.isFix()) {                        
045                                //do fix (make a new EF copy, replace in the properties)
046                                String newUri = Normalizer.uri(er.getUri() + "_", null, ef.getUri(), ef.getModelInterface());
047                                EntityFeature newEf = (new ShallowCopy()).copy(ef, newUri);
048                                model.add(newEf);
049                                er.removeEntityFeature(ef);
050                                er.addEntityFeature(newEf);
051                        for(SimplePhysicalEntity spe : er.getEntityReferenceOf()) {
052                                //replace the ef with newEf in the ER and in all owner SPEs
053                                if(spe.getFeature().contains(ef)) {
054                                        spe.removeFeature(ef);
055                                        spe.addFeature(newEf);
056                                } 
057                                if(spe.getNotFeature().contains(ef)) {
058                                        spe.removeNotFeature(ef);
059                                        spe.addNotFeature(newEf);
060                                }
061                        }                       
062                        }
063                }
064                
065                //if ersToFix.isEmpty()==true - dangling EF - the other rule can fix (if this EF belongs to a PE)
066                
067                if(ownerEntityRefs.size()>1) {
068                        error(validation, ef, "inverse.functional.violated", validation.isFix(), "entityFeature", ef, ownerEntityRefs.toString());      
069                } 
070                else if(ownerEntityRefs.size()==1) {  
071                        EntityReference er = ownerEntityRefs.iterator().next();
072                        if(efOf == null || !efOf.equals(er)) { 
073                                //provided ersToFix.size()==1, this here should never happen unless Paxtools API's hacked, or the model's incomplete...
074                                if(validation.isFix()) {
075                                        //easy and quiet fix (er.entityFeature has ef, whereas ef.entityFeatureOf() is null)
076                                        er.removeEntityFeature(ef);
077                                        er.addEntityFeature(ef);
078                                }
079                                
080                                if(efOf!=null)
081                                        error(validation, efOf, "inverse.functional.violated", validation.isFix(), "entityFeature", ef, er);    
082                        }
083                }
084        }
085    }
086
087        public boolean canCheck(Object thing) {
088                return thing instanceof Model;
089        }
090    
091}
