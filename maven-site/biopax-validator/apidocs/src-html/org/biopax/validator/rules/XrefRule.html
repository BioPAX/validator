025import org.biopax.paxtools.model.level3.Xref;
026import org.biopax.validator.api.AbstractRule;
027import org.biopax.validator.api.beans.Validation;
028import org.biopax.validator.utils.XrefHelper;
029import org.springframework.beans.factory.annotation.Autowired;
030import org.springframework.stereotype.Component;
031
032/**
033 * Validates Xrefs 
034 * (db/id properties)
035 *
036 * @author rodche
037 */
038@Component
039public class XrefRule extends AbstractRule<Xref>{
040
041    @Autowired
042    XrefHelper xrefHelper;
043
044        public boolean canCheck(Object thing) {
045                return (thing instanceof Xref);
046        }
047        
048    public void check(final Validation validation, Xref x) {
049        String db = x.getDb();
050                if (db != null) { 
051                        // check db
052                        String preferedDbName = xrefHelper.getPrimaryDbName(db); 
053                        if (preferedDbName == null) {
054                                error(validation, x, "unknown.db", false, db);
055                                return;
056                        }
057                        
058                        //it's always in uppercase (by XrefHelper design)!
059                        assert preferedDbName.equals(preferedDbName.toUpperCase());
060
061                        // check id
062                        String id = x.getId();
063                        if (id != null) {
064                                if (!xrefHelper.canCheckIdFormatIn(preferedDbName)) {
065                                        logger.info("Can't check IDs (no regexp) for " 
066                                                        + db + " (" + preferedDbName + ")");
067                                } else if (!xrefHelper.checkIdFormat(preferedDbName, id)) {
068                                        
069                                        String regxp = xrefHelper.getRegexpString(preferedDbName);
070                                        // report error with fixed=false 
071                                        error(validation, x, "invalid.id.format", false, db, preferedDbName, id, regxp);
072                                        
073                                        // Now - try to fix (in some cases...) 
074                                        while(validation.isFix()) { //- no worries - will use 'break' to escape the infinite loop
075                                                // guess it's a "UniProt Isoform" ID -  
076                                                // (do before we next will try splitting it into id and idVersion parts)
077                                                if (preferedDbName.startsWith("UNIPROT")) {
078                                                        if (id.contains("-")
079                                                                && xrefHelper.checkIdFormat("UniProt Isoform",id.toUpperCase())) 
080                                                        {
081                                                                x.setDb("UniProt Isoform");
082                                                                x.setId(id.toUpperCase());
083                                                                // update the error case, set fixed=true
084                                                                error(validation, x, "invalid.id.format", true);
085                                                                break;
086                                                        } 
087                                                } // guess it's in fact a PSI-MOD despite PSI-MI is used
088                                                else if (preferedDbName.equalsIgnoreCase("MOLECULAR INTERACTIONS ONTOLOGY")) {
089                                                        if (id.toUpperCase().startsWith("MOD")
090                                                                && xrefHelper.checkIdFormat("MOD", id.toUpperCase())) 
091                                                        {
092                                                                x.setDb("PSI-MOD");
093                                                                x.setId(id.toUpperCase());
094                                                                // update the error case, set fixed=true
095                                                                error(validation, x, "invalid.id.format", true);
096                                                                break;
097                                                        } 
098                                                }
099                                                        
100                                                // guess, the illegal id is like 'id_ver' and split then -
101                                                int i = id.lastIndexOf('-');
102                                                if(i<0) 
103                                                        i = id.lastIndexOf('_');
104                                                if(i<0) 
105                                                        i = id.lastIndexOf('.');
106                                                if(i > 0 && i < id.length()) {
107                                                        String newId = id.substring(0, i);
108                                                        if (xrefHelper.checkIdFormat(preferedDbName, newId)) {
109                                                                x.setId(newId);
110                                                                x.setIdVersion(id.substring(i + 1));
111                                                                // update the error case, set fixed=true there
112                                                                error(validation, x, "invalid.id.format", true);
113                                                                break;
114                                                        }
115                                                }
116                                                
117                                                /* 
118                                                 * Fix if MI:, GO:, MOD:, etc., prefixes were simply missing/forgotten -
119                                                 */
120                                                i = regxp.lastIndexOf(':');
121                                                if(i>0) {
122                                                        // guess, regexp looks like "^GO:%d{7}", and we want to get "GO"
123                                                        String prefix = regxp.substring(1, i).toUpperCase();
124                                                        if (logger.isDebugEnabled())
125                                                                logger.debug("Trying to fix id with missing prefix: " + prefix);
126                                                        if(preferedDbName.equalsIgnoreCase(xrefHelper.getPrimaryDbName(prefix))
127                                                                        && !id.toUpperCase().startsWith(prefix)) 
128                                                        {
129                                                                String newId = prefix + ':' + id;
130                                                                if (xrefHelper.checkIdFormat(preferedDbName, newId)) {
131                                                                        x.setId(newId);
132                                                                        error(validation, x, "invalid.id.format", true);
133                                                                        if (logger.isDebugEnabled())
134                                                                                logger.debug(x.getModelInterface()
135                                                                                        .getSimpleName() + " " + x
136                                                                                        + " 'id' auto-fixed! (was: " + id + ")");
137                                                                        break;
138                                                                }
139                                                        }
140                                                }
141                                                
142                                                
143                                                /*
144                                                 * Turning ID to upper case can sometimes help (e.g., KEGG, - c00022 to C00022 helps!) - 
145                                                 * because most identifier patterns corresp. to MIRIAM data collections are case sensitive and 
146                                                 * use upper-case symbols (e.g., Uniport's begin with P, Q, O; also - GO:, MOD:, and NP_ - same idea)
147                                                 */
148                                                String newId = id.toUpperCase();
149                                                if (xrefHelper.checkIdFormat(preferedDbName, newId)) {
150                                                        x.setId(newId);
151                                                        error(validation, x, "invalid.id.format", true);
152                                                        if (logger.isDebugEnabled())
153                                                                logger.debug(x.getModelInterface()
154                                                                        .getSimpleName() + " " + x
155                                                                        + " 'id' auto-fixed! (was: " + id + ")");
156                                                        break;
157                                                }                       
158                                                
159                                                
160                                                break; //breaks this loop anyway
161                                        } //end while
162                                }
163                        } 
164                } 
165    }
166
167}
