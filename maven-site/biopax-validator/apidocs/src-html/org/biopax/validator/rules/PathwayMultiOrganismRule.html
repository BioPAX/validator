025import java.util.Collection;
026import java.util.HashSet;
027
028import org.biopax.paxtools.controller.AbstractTraverser;
029import org.biopax.paxtools.controller.PropertyEditor;
030import org.biopax.paxtools.util.Filter;
031import org.biopax.paxtools.model.BioPAXElement;
032import org.biopax.paxtools.model.Model;
033import org.biopax.paxtools.model.level3.BioSource;
034import org.biopax.paxtools.model.level3.Pathway;
035import org.biopax.validator.api.AbstractRule;
036import org.biopax.validator.api.beans.Validation;
037import org.biopax.paxtools.controller.SimpleEditorMap;
038import org.springframework.stereotype.Component;
039
040/**
041 * Warn if a pathway and its component have different (not null) 'organism' values.
042 * What to do? (ignore, delete this value, or override nested organism properties with pathway's value)
043 * 
044 * @author rodche
045 */
046@Component
047public class PathwayMultiOrganismRule extends AbstractRule<Pathway> 
048{       
049        private final static Filter<PropertyEditor> filter = new Filter<PropertyEditor>() {
050
051                public boolean filter(PropertyEditor editor) {
052                        return !"nextStep".equals(editor.getProperty());
053                }
054        };
055        
056    public void check(final Validation validation, final Pathway pathway) {
057        final Collection<BioPAXElement> organisms = new HashSet<BioPAXElement>();
058        final BioSource organism = pathway.getOrganism(); // not null - due to the canCheck method!
059        //but..
060        if(organism==null) return; // we do not care
061        
062        AbstractTraverser runner = new AbstractTraverser(
063                        SimpleEditorMap.L3, filter) 
064        {
065                @Override
066                        protected void visit(Object value, BioPAXElement parent, 
067                                        Model model, PropertyEditor editor) 
068                {
069                                if(value instanceof BioSource) {        
070                                        if(!((BioPAXElement) value).isEquivalent(organism)) {
071                                                organisms.add((BioPAXElement) value);
072                                        }
073                                } 
074                                else if (value instanceof BioPAXElement) {
075                                        logger.trace("Traverse into " + value + " "
076                                                        + value.getClass().getSimpleName());
077                                        traverse((BioPAXElement) value, model);
078                                }
079                        }
080        };
081        
082                runner.traverse(pathway, null);
083                
084                if(organisms.size()>0) {
085                        error(validation, pathway, "multi.organism.pathway",
086                                false, organism, organisms);
087                }
088    }
089
090    public boolean canCheck(Object thing) {
091        return thing instanceof Pathway
092                && ((Pathway)thing).getOrganism() != null;
093    }
094
095}
