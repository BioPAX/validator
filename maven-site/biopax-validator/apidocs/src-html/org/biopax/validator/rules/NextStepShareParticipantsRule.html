040/**
041 * Checks:
042 * PathwayStep (in Pathway.pathwayOrder), if listed in the 
043 * nextStep property of another PathwayStep, must have 
044 * not empty intersection of participants of their 
045 * stepProcess-es (Process is an interaction or pathway) 
046 *
047 * @author rodche
048 *
049 */
050@Component
051public class NextStepShareParticipantsRule extends AbstractRule<PathwayStep>  
052{
053        public boolean canCheck(Object thing) {
054                return thing instanceof PathwayStep
055                        && !((PathwayStep) thing).getNextStepOf().isEmpty();
056        }
057
058        public void check(final Validation validation, PathwayStep step) {
059                if(step.getNextStepOf().isEmpty())
060                        return;
061                
062                // get all the participants
063                Collection<Entity> thisStepParticipants = getParticipants(step);
064                
065                // now find participants intersection with each previous step:
066                for (PathwayStep prevStep : step.getNextStepOf()) {     
067                        Collection<Entity> participants = getParticipants(prevStep);
068                        // first set becomes the intersection of the two:
069                        participants.retainAll(thisStepParticipants);
070                        if (participants.isEmpty()) {
071                                error(validation, step, "empty.participants.intersection", false, prevStep);
072                        }
073                }
074        }
075
076        Collection<Entity> getParticipants(PathwayStep step) {
077                // this is a protection from an infinite loop 
078                //(which normally should never occur; another rule will check that.)
079                Collection<Process> processes = new HashSet<Process>();
080                Set<Entity> ret = new HashSet<Entity>();
081                if(step instanceof BiochemicalPathwayStep) {
082                        Conversion c = ((BiochemicalPathwayStep) step).getStepConversion();
083                        ret.addAll( getParticipants(c, processes) );
084                }
085                for(Process p : step.getStepProcess()) {
086                        ret.addAll( getParticipants(p, processes) );
087                }
088                return ret;
089        }
090
091        Collection<Entity> getParticipants(Process process, Collection<Process> visited) 
092        {
093                Collection<Entity> ret = new HashSet<Entity>();
094                
095                // escape infinite loop
096                if(visited.contains(process)) {
097                        //"Step Processes Form a Loop!"
098                        return ret; // empty
099                } 
100                visited.add(process);
101                
102                if(process instanceof Interaction) {
103                        for(Entity pat : ((Interaction) process).getParticipant()) 
104                        {
105                                if(pat instanceof PhysicalEntity || pat instanceof Gene) 
106                                        ret.add(pat);
107                        }
108                } else { // a pathway
109                        for (Process p : ((Pathway) process).getPathwayComponent()) 
110                        {
111                                ret.addAll(getParticipants(p, visited));
112                        }
113                }
114                
115                return ret;
116        }
117        
118        
119        private void fix(PathwayStep t, Object... values) {
120                // is very difficult ;)
121        }
