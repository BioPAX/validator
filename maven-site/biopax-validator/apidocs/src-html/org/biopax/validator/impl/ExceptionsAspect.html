025import java.lang.reflect.Method;
026
027import org.apache.commons.logging.Log;
028import org.apache.commons.logging.LogFactory;
029import org.aspectj.lang.JoinPoint;
030import org.aspectj.lang.ProceedingJoinPoint;
031import org.aspectj.lang.annotation.*;
032import org.springframework.beans.factory.annotation.Configurable;
033import org.biopax.paxtools.controller.PropertyEditor;
034import org.biopax.paxtools.io.SimpleIOHandler;
035import org.biopax.paxtools.io.SimpleIOHandler.Triple;
036import org.biopax.paxtools.model.BioPAXElement;
037import org.biopax.paxtools.model.Model;
038import org.biopax.validator.api.AbstractAspect;
039
040/**
041 * This is the central aspect to report 
042 * all the validation exceptions together with 
043 * external exceptions that may happen in the PaxTools
044 * and other libraries. 
045 *
046 * @author rodche
047 */
048@Configurable
049@Aspect
050public class ExceptionsAspect extends AbstractAspect {
051        
052        private static final Log log = LogFactory.getLog(ExceptionsAspect.class);
053    
054    /**
055     * This captures the exceptions that occur 
056     * during the model build and, more important,
057     * associates the just created model
058     * with the corresponding validation result
059     * (this is the earliest possibility to do so)!
060     * 
061     * @param jp
062     */
063    @Around("execution(void org.biopax.paxtools.io.SimpleIOHandler.createAndBind(*)) " +
064                "&& args(model)")
065    public void adviseCreateAndBind(ProceedingJoinPoint jp, Model model) {
066        SimpleIOHandler reader = (SimpleIOHandler) jp.getTarget();
067        
068        // associate the model with the reader (and validation results)
069        validator.indirectlyAssociate(reader, model);
070        
071        try {
072            jp.proceed();
073        } catch (Throwable ex) {
074                reportException(ex, reader, "syntax.error", "SimpleIOHandler.createAndBind interceptor", null);
075        }
076    }
077    
078    
079    @Around("execution(* org.biopax.paxtools.io.SimpleIOHandler.processIndividual(*)) " 
080                + "&& args(model)")
081    public String adviseProcessIndividual(ProceedingJoinPoint jp, Model model) {  
082        String id = null;
083        SimpleIOHandler reader = (SimpleIOHandler) jp.getThis();
084        
085        try {
086            id = (String) jp.proceed();
087        } catch (Throwable ex) {
088                reportException(ex, reader, "syntax.error", "SimpleIOHandler.processIndividual interceptor", null);
089        }
090        return id;
091    }
092    
093    
094    @Around("execution(private void org.biopax.paxtools.io.SimpleIOHandler.bindValue(..))" +
095                " && args(triple, model)")
096    public void adviseBindValue(ProceedingJoinPoint jp, Triple triple, Model model) {
097        if(log.isDebugEnabled())
098                log.debug("adviseBindValue, triple: " + triple);
099        
100        SimpleIOHandler reader = (SimpleIOHandler) jp.getTarget();
101        // try to find the best object to report about...
102        Object o = reader;
103        BioPAXElement el = model.getByID(triple.domain);
104        if(el != null) {
105                o =  el;
106                        PropertyEditor<?,?> editor = reader.getEditorMap()
107                                .getEditorForProperty(triple.property, el.getModelInterface());
108                        if (editor == null) {
109                                // auto-fix (for some)
110                                if(triple.property.equals("taxonXref")) {
111                                        report(el, "unknown.property", 
112                                                        "SimpleIOHandler.bindValue interceptor", 
113                                                        true, triple.property + 
114                                                        " - replaced with 'xref'");
115                                        triple.property = "xref";
116                                } else {
117                                        report(el, "unknown.property", 
118                                                        "SimpleIOHandler.bindValue interceptor", 
119                                                        false, triple.property + 
120                                                        " - skipped");
121                                }
122                        }
123        } 
124                
125                try {
126                        jp.proceed();
127                } catch (Throwable t) {
128                        reportException(t, o, "syntax.error", "SimpleIOHandler.bindValue interceptor", triple.toString());
129                }
130    }
131    
132    @Around("execution(protected void org.biopax.paxtools.controller.PropertyEditor*+.checkRestrictions(..)) " +
133                "&& args(value, bean)")
134        public void adviseCheckRestrictions(ProceedingJoinPoint jp,  Object value, BioPAXElement bean) {
135        try {
136                jp.proceed();
137        } catch (Throwable ex) {
138                reportException(ex, bean, "syntax.error", "PropertyEditor.checkRestrictions interceptor", String.valueOf(value));
139        }
140        }    
141    
142    
143    @Around("execution(protected void org.biopax.paxtools.controller.PropertyEditor*+.invokeMethod(..)) " +
144                                "&& args(method, bean, value)")
145        public void adviseInvokeMethod(ProceedingJoinPoint jp, Method method, BioPAXElement bean, Object value) {
146        try {
147                jp.proceed();
148        } catch (Throwable ex) {
149                reportException(ex, bean, "syntax.error", "PropertyEditor.invokeMethod interceptor", "method: "+ method + ", value: " + value);
150        }
151        }
152    
153    
154    @Around("execution(* org.biopax.paxtools.io.BioPAXIOHandler*+.convertFromOWL(*))")
155        public Object adviseConvertFromOwl(ProceedingJoinPoint jp) {
156        Object model = null;
157        try {
158                model = jp.proceed();
159        } catch (Throwable ex) {
160                reportException(ex, jp.getTarget(), "syntax.error", "BioPAXIOHandler.convertFromOWL interceptor", null);
161        }
162        
163        return model;
164        }
165    
166    
167    @Before("execution(* org.biopax.paxtools.io.SimpleIOHandler.skip(..))")
168        public void adviseUnknownClass(JoinPoint jp) {
169        SimpleIOHandler reader = (SimpleIOHandler) jp.getTarget();
170        String loc = reader.getXmlStreamInfo();
171                report(reader, "unknown.class", "SimpleIOHandler.skip interceptor", false, loc);        
172        }
173        
174}
