018/**
019 * This helps validate xref's 'db', where different database synonymous 
020 * or misspelled names are often used, and 'id' that should match the 
021 * datasource's pattern.
022 *
023 * @author rodche
024 */
025public class XrefHelper {
026    private static final Log log = LogFactory.getLog(XrefHelper.class);
027    
028    private Map<String, Pattern> dataPatterns;
029    private CompositeCollection<String> allSynonyms; //set in init()!
030    private BiopaxOntologyManager ontologyManager;
031    private final CompositeCollection<String> extraGroups; //set in Constructor
032    private Set<String> unofficialDbNames; // to be generated   
033    
034    /**
035     * Constructor.
036     * 
037     * @param extraSynonyms a set of lists of unofficial names or common typos
038     * @param ontologyManager the object that gives access to some of controlled vocabularies/ontologies (OBO)
039     */
040    public XrefHelper(Set<List<String>> extraSynonyms, 
041        BiopaxOntologyManager ontologyManager) 
042    {           
043                // all database names and ID patterns go here
044                this.dataPatterns = new ConcurrentHashMap<String, Pattern>();
045                
046                // copy/normalize provided extra synonyms
047                this.extraGroups = new CompositeCollection<String>();
048                if(extraSynonyms != null) {
049                        for(List<String> group : extraSynonyms) {
050                                Collection<String> newG = new ArrayList<String>();
051                                for(String s : group) 
052                                        newG.add(dbName(s)); //trim, uppercase, add
053                                this.extraGroups.addComposited(newG);
054                        }
055                }
056                        
057                this.ontologyManager = ontologyManager;
058    }
059
060    /**
061     * Post-construct initializer.
062     * 
063     * Loads and merges bio database/datasource names and synonyms
064     * from Miriam resource, PSI-MI ontology ("database citation" branch),
065     * and extra (configured by user) names.
066     * 
067     */
068    @PostConstruct //vital thing
069    synchronized void init() {
070                this.allSynonyms =  new CompositeCollection<String>();
071                this.unofficialDbNames = Collections.newSetFromMap(new ConcurrentHashMap<String, Boolean>());   
072        
073        // need a temporaty, local "all synonyms" collection
074        // (hide the member one)
075        CompositeCollection<String> allSynonyms = new CompositeCollection<String>();
076        
077        // first, we simple prepare lists of db synonyms,
078        // from  Miriam, MI, and "extra" set (already done - constructor arg.)
079                for (String dt : MiriamLink.getDataTypesName()) {
080                        String regexp = MiriamLink.getDataTypePattern(dt);
081                        
082                        Pattern pattern = null;
083                        try {
084                                pattern= Pattern.compile(regexp);
085                        } catch (PatternSyntaxException e) {
086                                log.error(
087                                        "Pattern compilation failed for MIRIAM " +
088                                        "db=" + dt + "; regexp=" + dt + "; " + e);
089                        }
090                        String db = dbName(dt); //uppercase
091                        List<String> synonyms = new ArrayList<String>();
092                        synonyms.add(db);
093                        String[] names = MiriamLink.getNames(dt);
094                        if (names != null)
095                                for(String name : names) {
096                                        String s = dbName(name);
097                                        if(!synonyms.contains(s))
098                                                synonyms.add(s); 
099                                }
100                        //save
101                        allSynonyms.addComposited(synonyms);
102                        
103                        // also associate primary name with ID patterns
104                        if(pattern != null) //null value is not allowed for ConcurrentHashMap
105                                dataPatterns.put(db, pattern); // will be used with all synonyms
106                }
107        
108                // load all names from MI 'database citation'
109                OntologyAccess mi = ontologyManager.getOntology("MI");
110                Collection<OntologyTermI> terms = mi.getAllChildren(new OntologyTermImpl("MI:0444")); 
111                for (OntologyTermI term : terms) {
112                        List<String> synonyms = new ArrayList<String>();
113                        synonyms.add(dbName(term.getPreferredName()));
114                        for(String name : term.getNameSynonyms()) {
115                                String s = dbName(name);
116                                if(!synonyms.contains(s))
117                                        synonyms.add(s); 
118                        }
119                        allSynonyms.addComposited(synonyms);
120                }
121        
122        // second, we populate unofficialDbNames by comparing 
123        // allSynonyms (found so far) vs. extraSynonyms (provided)
124                for(String db : extraGroups) {
125                        if(!allSynonyms.contains(db))
126                                unofficialDbNames.add(db);
127                }
128        
129                allSynonyms.addComposited((Collection<? extends String>[]) extraGroups.getCollections().toArray());
130                
131        // find synonyms groups that overlap,
132                Cluster<Collection<String>> clus = new Cluster<Collection<String>>() {
133                        @Override
134                        public boolean match(Collection<String> a, Collection<String> b) {
135                                return !Collections.disjoint(a, b);
136                        }
137                };
138
139                // find all overlapping groups of names (can be single-group "clusters" as well))
140                // get the set of clusters (sets) of overlapping synonym lists! (i.e., we're in fact clustering List<String> objects)
141                Set<Set<Collection<String>>> groupsOfGroups = clus.cluster(allSynonyms.getCollections(), Integer.MAX_VALUE);
142                
143                // perform overlapping groups merging
144                // so that Miriam's preferred name goes first 
145                for(Set<Collection<String>> groupsToMerge : groupsOfGroups) {
146                        if (groupsToMerge.size() > 1) {
147                                List<String> merged = new ArrayList<String>();
148                                String primary = null;
149                                for (Collection<String> group : groupsToMerge) {
150                                        for (String name : group) {
151                                                if (dataPatterns.containsKey(name))
152                                                        primary = name;
153                                                if (!merged.contains(name))
154                                                        merged.add(name);
155                                        }
156                                }
157
158                                if (primary != null) {
159                                        merged.remove(primary);
160                                        merged.add(0, primary);
161                                }
162                                this.allSynonyms.addComposited(merged);
163                        } else {
164                                assert !groupsToMerge.isEmpty(); //one!
165                                //not overlapping (with other) single group
166                                this.allSynonyms.addComposited(groupsToMerge.iterator().next());
167                        }
168                }
169        
170                //sanity checks
171                assert(!getSynonymsForDbName("EntrezGene").isEmpty());  
172                assert(!getSynonymsForDbName("chebi").isEmpty());
173                assert xcheck();                
174    }
175    
176
177        private boolean xcheck() {
178                Object[] lists = allSynonyms.getCollections().toArray();
179                for(int i=0; i < lists.length; i++) {
180                        List<String> li = (List<String>) lists[i];
181                        for(int j=i+1; j < lists.length; j++) {
182                                if(!Collections.disjoint(li, (Collection<?>) lists[j])) {
183                                        log.error("different synonyms groups intercection found: " 
184                                                + li.toString() + " has names in common with " 
185                                                + lists[j].toString());
186                                        return false;
187                                }
188                        }
189                }
190                return true; // now intersection found :)
191        }

194        /**
195         * Removes tail spaces and converts to upper case
196         * 
197         * @param name
198         * @return
199         */
200        public String dbName(String name) {
201                return name.trim().toUpperCase();
202        }
203        
204        
205    /**
206     * Gets database name and its variants.
207     * The first in the list is the recommended one.
208     *
209     * @param name case insensitive
210     * @return set of equivalent database names
211     */
212    public List<String> getSynonymsForDbName(String name) {
213        return getSynonymsForName(name, allSynonyms);
214    }
215
216    
217    private List<String> getSynonymsForName(String name, CompositeCollection<String> groups) {
218        String dbName = dbName(name);
219                
220                for(Collection<String> group : groups.getCollections()) {
221                        if (group.contains(dbName)) {
222                                return (List<String>) group; //copy to protect
223                        }
224                }
225                return Collections.emptyList();
226    }

229    /**
230     * Gets the primary name for the DB.
231     * It returns NULL for "unknown" database name. 
232     * 
233     * @param name case insensitive
234     * @return
235     */
236    public String getPrimaryDbName(String name) {
237        String dbName = dbName(name);
238                
239                for(Collection<String> group : allSynonyms.getCollections()) {
240                        if (group.contains(dbName)) {
241                                return group.iterator().next(); //get the first
242                        }
243                }
244                
245                return null;
246    }

249    /**
250     * Checks whether the ID format is valid for the database.
251     * Always use {@link #canCheckIdFormatIn(String)} before this method,
252     * because it may throw an exception if you do not.
253     * 
254     * @param db case insensitive
255     * @param id
256     * @return 'false' if matcher fails, 'true' otherwise
257     * @throws NullPointerException when no pattern available
258     */
259    public boolean checkIdFormat(String db, String id) {
260        String name = getPrimaryDbName(db);
261        Pattern p = dataPatterns.get(name);
262        return p.matcher(id).find();
263    }
264
265    /**
266     * @param name a database name (used in xrefs), case insensitive
267     * @return true if it's possible to check the format.
268     */
269    public boolean canCheckIdFormatIn(String name) {
270        String db = getPrimaryDbName(name);
271        return (db == null) ? false : dataPatterns.get(db) != null;
272    }

275    /**
276     * Gets the regular expression corresponding 
277     * to the database.
278     * 
279     * @param db a database name, case insensitive
280     * @return regular expression to check its ID
281     */
282    public String getRegexpString(String db) {
283        String name = getPrimaryDbName(db);
284        Pattern p = dataPatterns.get(name);
285        return (p != null) ? p.pattern() : null;
286    }

289    //case insensitive
290    public boolean isSynonyms(String db1, String db2) {
291        // dbName function is used here also to get right 
292        // name capitalization
293        return getSynonymsForDbName(db1).contains(dbName(db2));
294    }

298     * Get the immutable list of 
299     * DB misspellings and unofficial names,
300     * which the Validator recognizes, reports 
301     * warnings, and is able to replace with
302     * corresponding official names; and which
303     * otherwise, using MI or Miriam, 
304     * cannot be resolved.
305     * 
306     * @return
307     */
308    Set<String> getUnofficialDbNames() {
309                return unofficialDbNames;
310        }
311    
312    /**
313     * Checks whether the db name is known (configured) 
314     * misspellings or unofficial name,
315     * which the Validator can recognize, report 
316     * warning, and replace with official names; and which
317     * otherwise, using MI or Miriam, cannot be resolved.
318     * 
319     * @param db case insensitive
320     * @return
321     */
322        public boolean isUnofficialOrMisspelledDbName(final String db) {
323                return getUnofficialDbNames().contains(dbName(db));
324        }
