001package org.biopax.validator.utils;
002
003import java.util.*;
004
005import org.apache.commons.lang.StringEscapeUtils;
006import org.apache.commons.logging.Log;
007import org.apache.commons.logging.LogFactory;
008import org.biopax.psidev.ontology_manager.OntologyAccess;
009import org.biopax.psidev.ontology_manager.OntologyTermI;
010import org.biopax.psidev.ontology_manager.impl.OntologyManagerImpl;
011import org.biopax.validator.api.CvRestriction;
012import org.biopax.validator.api.CvRule;
013import org.biopax.validator.api.CvValidator;
014import org.biopax.validator.api.CvRestriction.UseChildTerms;
015
016
017/**
018 * Access to BioPAX controlled vocabularies.
019 * This component is built from a modified PSIDEV tool, OntologyAccess Manager, 
020 * by extending it and adding several "proxy" methods that allow to extract 
021 * validator-specific data only once and free the memory after it, if required.
022 * However, it does not hide base class's methods.
023 * 
024 * REQUIRES INTERNET CONNECTION (when configured to use OBO URLs)
025 * 
026 * @author rodche
027 *
028 */
029public class BiopaxOntologyManager extends OntologyManagerImpl implements CvValidator {
030        private final static Log log = LogFactory.getLog(BiopaxOntologyManager.class);
031        
032        /**
033         * Constructor
034         * @param ontologiesConfig OBO ontologies configuration properties
035         * 
036         */
037        public BiopaxOntologyManager(Properties ontologiesConfig)
038        {
039                try {
040                        loadOntologies(ontologiesConfig);
041                } catch (Throwable e) {
042                        throw new RuntimeException("Failed to load or parse all required biological ontologies!", e);
043                }
044        }
045
046        /* (non-Javadoc)
047         * @see org.biopax.validator.utils.CvValidator#getValidTermNames(org.biopax.validator.CvRule)
048         */
049        public Set<String> getValidTermNames(CvRule<?> cvRule) {
050                return getValidTermNamesLowerCase(cvRule.getRestrictions());
051        }
052        
053        /* (non-Javadoc)
054         * @see org.biopax.validator.utils.CvValidator#getValidTerms(org.biopax.validator.CvRule)
055         */
056        public Set<OntologyTermI> getValidTerms(CvRule<?> cvRule) {
057                return getValidTerms(cvRule.getRestrictions());
058        }
059        
060        
061        /* (non-Javadoc)
062         * @see org.biopax.validator.utils.CvValidator#getValidTermNames(java.util.Collection)
063         */
064        public Set<String> getValidTermNames(Collection<CvRestriction> restrictions) {
065                Set<String> names = new HashSet<String>();
066                Set<OntologyTermI> terms = getValidTerms(restrictions);
067                names.addAll(getTermNames(terms));
068                return names;
069        }
070                
071
072        /* (non-Javadoc)
073         * @see org.biopax.validator.utils.CvValidator#getValidTermNamesLowerCase(java.util.Collection)
074         */
075        public Set<String> getValidTermNamesLowerCase(Collection<CvRestriction> restrictions) {
076                Set<String> names = new HashSet<String>();
077                for(String name : getValidTermNames(restrictions)) {
078                        names.add(name.toLowerCase());
079                }
080                return names;
081        }
082        
083        
084        /* (non-Javadoc)
085         * @see org.biopax.validator.utils.CvValidator#getTermNames(org.biopax.validator.impl.CvTermRestriction)
086         */
087        public Set<String> getTermNames(CvRestriction restriction) {
088                Set<OntologyTermI> terms = getTerms(restriction);
089                Set<String> names = new HashSet<String>();
090                names.addAll(getTermNames(terms));
091                return names;
092        }
093   
094        
095        
096        /* (non-Javadoc)
097         * @see org.biopax.validator.utils.CvValidator#getValidTerms(java.util.Collection)
098         */
099        public Set<OntologyTermI> getValidTerms(Collection<CvRestriction> restrictions) {
100                Set<OntologyTermI> terms = new HashSet<OntologyTermI>();
101                
102                // first, collect all the valid terms
103                for(CvRestriction restriction : restrictions) {
104                        if(!restriction.isNot()) {
105                                terms.addAll(getTerms(restriction));
106                        }
107                }
108                
109                // now remove all those where restriction 'not' property set to true
110                for(CvRestriction restriction : restrictions) {
111                        if(restriction.isNot()) {
112                                terms.removeAll(getTerms(restriction));
113                        }
114                }
115                
116                return terms;
117        }
118        
119        
120        /* (non-Javadoc)
121         * @see org.biopax.validator.utils.CvValidator#getTerms(org.biopax.validator.impl.CvTermRestriction)
122         */
123        public Set<OntologyTermI> getTerms(CvRestriction restriction) {
124                Set<OntologyTermI> terms = new HashSet<OntologyTermI>();
125                OntologyAccess ontologyAccess = getOntology(restriction.getOntologyId());
126                if(ontologyAccess == null) {
127                        throw new IllegalArgumentException(
128                                        "Cannot get access to the ontology using ID: " 
129                                        + restriction.getOntologyId() +
130                                        "; I know the following IDs: " 
131                                        + getOntologyIDs().toString());
132                }
133                OntologyTermI term = ontologyAccess.getTermForAccession(restriction.getId());
134                if(term == null) {
135                        log.error("Cannot Get " + restriction.getOntologyId()
136                                        + " OntologyAccess Term for the Accession: " + restriction.getId());
137                        return terms;
138                }
139                
140                if(restriction.isTermAllowed()) {
141                        terms.add(term);
142                }
143                
144                if (restriction.getChildrenAllowed() == UseChildTerms.ALL) {
145                        terms.addAll(ontologyAccess.getAllChildren(term));
146                } else if (restriction.getChildrenAllowed() == UseChildTerms.DIRECT) {
147                        terms.addAll(ontologyAccess.getDirectChildren(term));
148                }
149                
150                // FIX xml escape symbols that come from the OntologyManagerImpl (a bug?)
151                for(OntologyTermI t : terms) {
152                        t.setPreferredName(StringEscapeUtils.unescapeXml(t.getPreferredName()));
153                        Set<String> synonyms = new HashSet<String>();
154                        for(String s : t.getNameSynonyms()) {
155                                synonyms.add(StringEscapeUtils.unescapeXml(s));
156                        }
157                        t.setNameSynonyms(synonyms);
158                }
159                
160                return terms;
161        }
