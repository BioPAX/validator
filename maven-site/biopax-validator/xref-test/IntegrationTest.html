105 	/*
106 	 * Tests:
107 	 * 
108 	 * For 'GO', in the manuallyAddedDbSynonyms bean configuration
109 	 * (see spring-context.xml) we have 
110 	 * 	<list>
111 	 *		<value>GO</value>
112 	 *		<value>GENE ONTOLOGY</value>
113 	 *		<value>GENE_ONTOLOGY</value>
114 	 *	</list>
115 	 * 
116 	 * In the MIRIAM (resources/Miriam.xml) we have:
117 	 *  <datatype id="MIR:00000022" startChar="^GO:\d{7}$">
118 	 * 	<name>Gene OntologyAccess</name>
119 	 *	<synonyms>
120 	 * 		<synonym>GO</synonym>
121 	 * 	</synonyms>
122 	 * 		
123 	 *  So, we want to test here that
124 	 *  1. all three synonyms are present there;
125 	 *  2. the primary one should be 'GENE ONTOLOGY';
126 	 *  3. all three are assigned with the regexp: "^GO:\d{7}$".
127 	 *  
128 	 */
129     @Test
130     public void testSynonymsWereRead() {
131     	List<String> gs = xrefHelper.getSynonymsForDbName("go"); 	
132     	assertTrue(gs.contains("GO"));
133     	assertTrue(gs.contains("GENE ONTOLOGY"));
134     	assertTrue(gs.contains("GENE_ONTOLOGY"));
135     	
136     	assertTrue(xrefHelper.isUnofficialOrMisspelledDbName("GENE_ONTOLOGY"));
137     	assertFalse(xrefHelper.isUnofficialOrMisspelledDbName("GO"));
138     	assertTrue(xrefHelper.isUnofficialOrMisspelledDbName("medline"));
139     }

282     @Test
283     public void testInteractionTypeRule() {
284         InteractionTypeCvRule instance =  
285                 (InteractionTypeCvRule) context.getBean("interactionTypeCvRule");    
286         
287         Model m = factory3.createModel();
288         
289         InteractionVocabulary iv = factory3.create(InteractionVocabulary.class, "preferredCVTerm");
290         iv.addTerm("phosphorylation reaction");
291         iv.addComment("Preferred term");
292         
293         UnificationXref ux = factory3.create(UnificationXref.class, "UnificationXref_MI_0217");
294     	ux.setDb("MI");
295     	ux.setId("MI:0217");
296     	iv.addXref(ux);
297     	
298         m.add(ux);
299         m.add(iv);
300         
301         Validation v = new Validation(new IdentifierImpl());
302         instance.check(v, iv);
303         assertTrue(v.getError().isEmpty());
304         
305         iv = factory3.create(InteractionVocabulary.class, "synonymCVTerm");
306         iv.addTerm("phosphorylation");
307         iv.addComment("Valid term");
308         m.add(iv);
309         iv.addXref(ux);
310         
311         v = new Validation(new IdentifierImpl());
312         instance.check(v, iv);
313         assertTrue(v.getError().isEmpty());
314         
315         iv = factory3.create(InteractionVocabulary.class, "okCVTerm");
316         iv.addTerm("Phosphorylation");
317         iv.addComment("Ok term: upper or lower case letters do not matter.");
318         m.add(iv);
319         iv.addXref(ux);
320         
321         v = new Validation(new IdentifierImpl());
322         instance.check(v, iv);
323         assertTrue(v.getError().isEmpty());
324         
325         iv = factory3.create(InteractionVocabulary.class, "invalidCVTerm");
326         iv.addTerm("phosphorylated residue");
327         iv.addComment("Invalid term (very similar, however)");
328         ux = factory3.create(UnificationXref.class, "UnificationXref_MOD_00696");
329     	ux.setDb("MOD");
330     	ux.setId("MOD:00696"); 
331     	/* Note: in fact, both MOD:00696 and MI:0217 have synonym name "Phosphorylation"!
332     		TODO Validator (currently) checks names in the CV 'term' property only, but also should check what can be inferred from the xref.id!
333     	*/
334     	iv.addXref(ux);
335     	m.add(ux);
336         m.add(iv);
337         
338         v = new Validation(new IdentifierImpl());
339 		instance.check(v, iv);
340 		assertEquals(1, v.countErrors(iv.getUri(), null, "illegal.cv.term", null, false, false));
341         
342         writeExample("testInteractionTypeRule.owl", m);
343     }
