16  import org.springframework.stereotype.Component;

18  /**
19   * Checks:
20   * Dna's or Rna's modificationFeature.modoficationType has valid SequenceModificationVocabulary term
21   * 
22   * @author rodche
23   */
24  @Component
25  public class NucleicAcidModificationFeatureCvRule extends CvTermsRule<ModificationFeature> {
26  		
27  	public NucleicAcidModificationFeatureCvRule() {
28  		super(ModificationFeature.class, "modificationType", 
29  				new CvRestriction("SO:1000132","SO", true, 
30  						UseChildTerms.ALL, false),
31  				new CvRestriction("SO:0001059","SO", true, 
32  						UseChildTerms.ALL, false));
33  	}
34  	
35  	// This rule is for Dna*/Rna* and corresp.  entity references's features
36  	@Override
37  	public boolean canCheck(Object thing) {
38  		if ( thing instanceof ModificationFeature
39  			&& ((ModificationFeature) thing).getModificationType() != null)
40  		{
41  			EntityReference er = ((ModificationFeature) thing).getEntityFeatureOf();
42  			if (er instanceof NucleicAcidReference || er instanceof NucleicAcidRegionReference)
43  				return true;
44  			
45  			for (PhysicalEntity pe : ((ModificationFeature) thing).getFeatureOf()) {
46  				if (pe instanceof NucleicAcid)
47  					return true;
48  			}
49  		}
50  		return false;
51  	}
52  
53   }
