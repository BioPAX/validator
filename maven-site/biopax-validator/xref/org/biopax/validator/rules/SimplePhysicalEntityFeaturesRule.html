25  import org.biopax.paxtools.controller.ShallowCopy;
26  import org.biopax.paxtools.model.level3.EntityFeature;
27  import org.biopax.paxtools.model.level3.EntityReference;
28  import org.biopax.paxtools.model.level3.SimplePhysicalEntity;
29  import org.biopax.paxtools.normalizer.Normalizer;
30  import org.biopax.validator.api.AbstractRule;
31  import org.biopax.validator.api.beans.Validation;
32  import org.springframework.stereotype.Component;
33  
34  import java.util.HashSet;
35  import java.util.Set;
36  
37  /**
38   *
39   * A rule to check if a Simple Physical Entity Feature is also a feature of
40   * its Entity Reference. If not, fixes it by adding the missing feature
41   * to the Entity Reference.
42   *
43   */
44  @Component
45  public class SimplePhysicalEntityFeaturesRule extends AbstractRule<SimplePhysicalEntity>{
46  
47      public void check(final Validation validation, SimplePhysicalEntity thing) {
48          EntityReference er = thing.getEntityReference();
49          //sync: this er can belong to many SPEs, and this rule can be called from another thread for the other SPE
50          synchronized (er) {	
51          	Set<EntityFeature> erFeatures =  new HashSet<EntityFeature>(er.getEntityFeature());//defensive copy
52          	
53          	Set<EntityFeature> peFeaturesAndNotFeatures = new HashSet<EntityFeature>();
54          	peFeaturesAndNotFeatures.addAll(thing.getFeature());
55          	peFeaturesAndNotFeatures.addAll(thing.getNotFeature());
56  
57          	for(EntityFeature ef: peFeaturesAndNotFeatures) {
58          		if(!erFeatures.contains(ef)) { //the ER does not have this EF...
59          			if(validation.isFix()) {
60          				if(ef.getEntityFeatureOf() != null) {//it belongs to the other ER
61          					//trying to generate a new unique URI for the EF copy, for any ER,EF pair is unique 
62          					//(considering one-to-many, inverse functional constraints of the entityFeature prop.):
63          					String uri = Normalizer.uri(er.getUri() + "_", null, ef.getUri(), ef.getModelInterface());
64          					EntityFeature newEf = null; //check if there is one with this URI already; use that one then
65          					for(EntityFeature f : er.getEntityFeature()) {
66          						if(uri.equals(f.getUri())) {
67          							newEf = f;
68          							break;
69          						}
70          					}
71          					if(newEf == null) {
72          						//make a copy
73          						newEf = (new ShallowCopy()).copy(ef, uri);
74          						assert newEf.getEntityFeatureOf() == null 
75          							: "getEntityFeatureOf of a shallow copy of EF is not null...";
76          					}
77          					//replace the EF in the SPE's feature or notFeature set
78          					if(thing.getFeature().contains(ef)) {
79          						thing.removeFeature(ef);
80          						thing.addFeature(newEf);
81          					} 
82          					if(thing.getNotFeature().contains(ef)) {
83          						thing.removeNotFeature(ef);
84          						thing.addNotFeature(newEf);
85          					}
86          					// add the new one to the ER
87          					er.addEntityFeature(newEf);
88          				} else {
89          					// add the one to the ER
90          					er.addEntityFeature(ef);
91          				}
92          			}
93  
94          			error(validation, thing, "improper.feature.use", validation.isFix(), ef.getUri(), er.getUri());
95          		}
96          	}        
97          }             
98      }
99  
100     public boolean canCheck(Object thing) {
101         return (thing instanceof SimplePhysicalEntity) 
102         		&& (((SimplePhysicalEntity) thing).getEntityReference() != null);
103     }
104 }
