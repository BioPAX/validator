19  /**
20   * This class warns on instances of too general (top-level) BioPAX classes
21   *
22   * @author rodche
23   *
24   */
25  @Component
26  public class NotAdvisedInstancesRule extends AbstractRule<BioPAXElement> {
27  
28      final Class[] NOT_ADVISED = {
29      	Control.class, 
30      	Conversion.class,
31          Interaction.class,
32      	EntityFeature.class, 
33      	PhysicalEntity.class, 
34      	physicalEntity.class,
35      	conversion.class,
36      };
37      
38      private boolean notAllowed(Object o) {
39  		if (o instanceof BioPAXElement) {
40  			for (int i = 0; i < NOT_ADVISED.length; i++) {
41  				if (((BioPAXElement) o).getModelInterface().equals(
42  						NOT_ADVISED[i])) {
43  					return true; // found
44  				}
45  			}
46  		}
47  		return false;
48  	}
49      
50  	public boolean canCheck(Object thing) {
51  		return notAllowed(thing);
52  	}
53  
54  	public void check(final Validation validation, BioPAXElement thing) {
55  		error(validation, thing, "not.specific.element", false, thing.getModelInterface().getSimpleName());
56  	}
57  
58  }
