36  import java.util.HashSet;

39  /**
40   *
41   * A rule to check if the Conversion can be converted
42   * to a ComplexAssembly.
43   *
44   * Some Conversions are not cast as ComplexAssembly,
45   * although there is no modification to the PEs
46   * and there is a clear Complex formation
47   * throughout the process.
48   *
49   */
50  @Component
51  public class ConversionToComplexAssemblyRule extends AbstractRule<Conversion> {
52      public void check(final Validation validation, Conversion thing) {
53      	//for thread safety (concurrency) we're using a new fetcher here rather than static one 
54      	Fetcher fetcher = new Fetcher(
55      			SimpleEditorMap.L3, new Filter<PropertyEditor>() {
56      				//complex.component only
57      				public boolean filter(PropertyEditor editor) {
58      					return editor.getProperty().equals("component");
59      				}
60      			});   	
61          Set<PhysicalEntity> left = new HashSet<PhysicalEntity>(getPEsRecursively(thing.getLeft(), fetcher)); //need a mutable set
62          Set<PhysicalEntity> right = getPEsRecursively(thing.getRight(), fetcher);
63          left.removeAll(right);
64  
65          int complexDiff = getComplexCount(thing.getLeft()) - getComplexCount(thing.getRight());
66          if( left.isEmpty()  // when there are no modifications really, but different no. complexes or participants
67                  && (complexDiff != 0 || thing.getLeft().size() - thing.getRight().size() != 0 ))
68              error(validation, thing, "wrong.conversion.class", false, thing.getModelInterface());
69  
70      }
71  
72      private int getComplexCount(Set<PhysicalEntity> pes) {
73          int count = 0;
74  
75          for(PhysicalEntity pe: pes) {
76              if(pe instanceof Complex)
77                  count++;
78          }
79  
80          return count;
81      }
82  
83      private Set<PhysicalEntity> getPEsRecursively(Set<PhysicalEntity> pes, Fetcher fetcher) {
84      	Model m = BioPAXLevel.L3.getDefaultFactory().createModel();
85      	for(PhysicalEntity pe : pes) {
86      		if(pe instanceof Complex)
87      			fetcher.fetch(pe,m);
88      		else 
89      			if(!m.containsID(pe.getUri()))
90      				m.add(pe);
91      	}   
92  
93      	return m.getObjects(PhysicalEntity.class);
94      }
95  
96      public boolean canCheck(Object thing) {
97          return thing instanceof Conversion
98                  && !(thing instanceof ComplexAssembly);
99      }
100 }
