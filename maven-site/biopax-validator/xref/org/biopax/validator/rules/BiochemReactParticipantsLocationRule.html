25  import org.apache.commons.collections15.CollectionUtils;
26  import org.biopax.paxtools.model.level3.BiochemicalReaction;
27  import org.biopax.paxtools.model.level3.Complex;
28  import org.biopax.paxtools.model.level3.PhysicalEntity;
29  import org.biopax.paxtools.model.level3.SimplePhysicalEntity;
30  import org.biopax.paxtools.model.level3.Transport;
31  import org.biopax.validator.api.AbstractRule;
32  import org.biopax.validator.api.beans.Validation;
33  import org.springframework.stereotype.Component;
34  
35  import java.util.Set;
36  
37  /**
38   * BiochemicalReaction (base) - the participants on one side don't 
39   * change compartments on the other side (otherwise it should be a 
40   * TransportWithBiochemicalReaction)
41   * 
42   * TODO Complexes are matched by name, other physical entities can be "identified" using entity reference.
43   * 
44   * @author rodche
45   */
46  @Component
47  public class BiochemReactParticipantsLocationRule extends AbstractRule<BiochemicalReaction>
48  {
49  
50      public void check(final Validation validation, BiochemicalReaction react)
51      {
52         Set<PhysicalEntity> left = react.getLeft();
53         Set<PhysicalEntity> right = react.getRight();
54         for (PhysicalEntity lefty : left)
55         {
56              for (PhysicalEntity righty : right)
57              {
58              	// PhysicalEntity is either Complex or SimplePhysicalEntity (NucleicAcid, Protein or SmallMolecule)
59              	
60              	if(!lefty.getModelInterface().equals(righty.getModelInterface())) {
61                  	continue;
62                  }
63              	
64              	boolean sameComplex = (lefty instanceof Complex) // means - righty is also Complex
65              		&& CollectionUtils.containsAny(lefty.getName(), righty.getName());
66              	// Complex does not have entityReference to match the same kind on both sides...
67              	
68              	boolean sameSimplePhysicalEntity = (lefty instanceof SimplePhysicalEntity) 
69              		&& ( ((SimplePhysicalEntity)lefty).getEntityReference() == null 
70              			? ((SimplePhysicalEntity)righty).getEntityReference() == null 
71              			: ((SimplePhysicalEntity)lefty).getEntityReference()
72              				.isEquivalent(((SimplePhysicalEntity)righty).getEntityReference()) );
73              	
74              	if(sameComplex || sameSimplePhysicalEntity) {
75                  	boolean sameLoc = lefty.hasEquivalentCellularLocation(righty);
76                  	if(!sameLoc	&& !(react instanceof Transport)) {
77                  		error(validation, react, "participant.location.changed", false, lefty, righty); 
78                  	} else if(sameLoc && react instanceof Transport) {
79                  		error(validation, react, "transport.location.same", false, lefty, righty); 
80                  	}
81                  } 
82              }
83          }
84      }
85  
86      public boolean canCheck(Object thing) {
87          return thing instanceof BiochemicalReaction;
88      }
89  }
