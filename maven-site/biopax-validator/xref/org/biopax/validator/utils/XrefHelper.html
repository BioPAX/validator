18  /**
19   * This helps validate xref's 'db', where different database synonymous 
20   * or misspelled names are often used, and 'id' that should match the 
21   * datasource's pattern.
22   *
23   * @author rodche
24   */
25  public class XrefHelper {
26      private static final Log log = LogFactory.getLog(XrefHelper.class);
27      
28      private Map<String, Pattern> dataPatterns;
29      private CompositeCollection<String> allSynonyms; //set in init()!
30      private BiopaxOntologyManager ontologyManager;
31      private final CompositeCollection<String> extraGroups; //set in Constructor
32      private Set<String> unofficialDbNames; // to be generated	
33      
34      /**
35       * Constructor.
36       * 
37       * @param extraSynonyms a set of lists of unofficial names or common typos
38       * @param ontologyManager the object that gives access to some of controlled vocabularies/ontologies (OBO)
39       */
40      public XrefHelper(Set<List<String>> extraSynonyms, 
41      	BiopaxOntologyManager ontologyManager) 
42      {   	
43  		// all database names and ID patterns go here
44  		this.dataPatterns = new ConcurrentHashMap<String, Pattern>();
45  		
46  		// copy/normalize provided extra synonyms
47  		this.extraGroups = new CompositeCollection<String>();
48  		if(extraSynonyms != null) {
49  			for(List<String> group : extraSynonyms) {
50  				Collection<String> newG = new ArrayList<String>();
51  				for(String s : group) 
52  					newG.add(dbName(s)); //trim, uppercase, add
53  				this.extraGroups.addComposited(newG);
54  			}
55  		}
56  			
57  		this.ontologyManager = ontologyManager;
58      }
59  
60      /**
61       * Post-construct initializer.
62       * 
63       * Loads and merges bio database/datasource names and synonyms
64       * from Miriam resource, PSI-MI ontology ("database citation" branch),
65       * and extra (configured by user) names.
66       * 
67       */
68      @PostConstruct //vital thing
69      synchronized void init() {
70  		this.allSynonyms =  new CompositeCollection<String>();
71  		this.unofficialDbNames = Collections.newSetFromMap(new ConcurrentHashMap<String, Boolean>());	
72      	
73      	// need a temporaty, local "all synonyms" collection
74      	// (hide the member one)
75      	CompositeCollection<String> allSynonyms = new CompositeCollection<String>();
76      	
77      	// first, we simple prepare lists of db synonyms,
78      	// from  Miriam, MI, and "extra" set (already done - constructor arg.)
79  		for (String dt : MiriamLink.getDataTypesName()) {
80  			String regexp = MiriamLink.getDataTypePattern(dt);
81  			
82  			Pattern pattern = null;
83  			try {
84  				pattern= Pattern.compile(regexp);
85  			} catch (PatternSyntaxException e) {
86  				log.error(
87  					"Pattern compilation failed for MIRIAM " +
88  					"db=" + dt + "; regexp=" + dt + "; " + e);
89  			}
90  			String db = dbName(dt); //uppercase
91  			List<String> synonyms = new ArrayList<String>();
92  			synonyms.add(db);
93  			String[] names = MiriamLink.getNames(dt);
94  			if (names != null)
95  				for(String name : names) {
96  					String s = dbName(name);
97  					if(!synonyms.contains(s))
98  						synonyms.add(s); 
99  				}
100 			//save
101 			allSynonyms.addComposited(synonyms);
102 			
103 			// also associate primary name with ID patterns
104 			if(pattern != null) //null value is not allowed for ConcurrentHashMap
105 				dataPatterns.put(db, pattern); // will be used with all synonyms
106 		}
107     	
108 		// load all names from MI 'database citation'
109 		OntologyAccess mi = ontologyManager.getOntology("MI");
110 		Collection<OntologyTermI> terms = mi.getAllChildren(new OntologyTermImpl("MI:0444")); 
111 		for (OntologyTermI term : terms) {
112 			List<String> synonyms = new ArrayList<String>();
113 			synonyms.add(dbName(term.getPreferredName()));
114 			for(String name : term.getNameSynonyms()) {
115 				String s = dbName(name);
116 				if(!synonyms.contains(s))
117 					synonyms.add(s); 
118 			}
119 			allSynonyms.addComposited(synonyms);
120 		}
121     	
122     	// second, we populate unofficialDbNames by comparing 
123     	// allSynonyms (found so far) vs. extraSynonyms (provided)
124 		for(String db : extraGroups) {
125 			if(!allSynonyms.contains(db))
126 				unofficialDbNames.add(db);
127 		}
128     	
129 		allSynonyms.addComposited((Collection<? extends String>[]) extraGroups.getCollections().toArray());
130 		
131     	// find synonyms groups that overlap,
132 		Cluster<Collection<String>> clus = new Cluster<Collection<String>>() {
133 			@Override
134 			public boolean match(Collection<String> a, Collection<String> b) {
135 				return !Collections.disjoint(a, b);
136 			}
137 		};
138 
139 		// find all overlapping groups of names (can be single-group "clusters" as well))
140 		// get the set of clusters (sets) of overlapping synonym lists! (i.e., we're in fact clustering List<String> objects)
141 		Set<Set<Collection<String>>> groupsOfGroups = clus.cluster(allSynonyms.getCollections(), Integer.MAX_VALUE);
142 		
143 		// perform overlapping groups merging
144 		// so that Miriam's preferred name goes first 
145 		for(Set<Collection<String>> groupsToMerge : groupsOfGroups) {
146 			if (groupsToMerge.size() > 1) {
147 				List<String> merged = new ArrayList<String>();
148 				String primary = null;
149 				for (Collection<String> group : groupsToMerge) {
150 					for (String name : group) {
151 						if (dataPatterns.containsKey(name))
152 							primary = name;
153 						if (!merged.contains(name))
154 							merged.add(name);
155 					}
156 				}
157 
158 				if (primary != null) {
159 					merged.remove(primary);
160 					merged.add(0, primary);
161 				}
162 				this.allSynonyms.addComposited(merged);
163 			} else {
164 				assert !groupsToMerge.isEmpty(); //one!
165 				//not overlapping (with other) single group
166 				this.allSynonyms.addComposited(groupsToMerge.iterator().next());
167 			}
168 		}
169     	
170 		//sanity checks
171 		assert(!getSynonymsForDbName("EntrezGene").isEmpty());	
172 		assert(!getSynonymsForDbName("chebi").isEmpty());
173 		assert xcheck();		
174     }

177 	private boolean xcheck() {
178 		Object[] lists = allSynonyms.getCollections().toArray();
179 		for(int i=0; i < lists.length; i++) {
180 			List<String> li = (List<String>) lists[i];
181 			for(int j=i+1; j < lists.length; j++) {
182 				if(!Collections.disjoint(li, (Collection<?>) lists[j])) {
183 					log.error("different synonyms groups intercection found: " 
184 						+ li.toString() + " has names in common with " 
185 						+ lists[j].toString());
186 					return false;
187 				}
188 			}
189 		}
190 		return true; // now intersection found :)
191 	}

194 	/**
195 	 * Removes tail spaces and converts to upper case
196 	 * 
197 	 * @param name original name
198 	 * @return normalized name
199 	 */
200 	public String dbName(String name) {
201 		return name.trim().toUpperCase();
202 	}

205     /**
206      * Gets database name and its variants.
207      * The first in the list is the recommended one.
208      *
209      * @param name case insensitive
210      * @return set of equivalent database names
211      */
212     public List<String> getSynonymsForDbName(String name) {
213     	return getSynonymsForName(name, allSynonyms);
214     }

216     
217     private List<String> getSynonymsForName(String name, CompositeCollection<String> groups) {
218     	String dbName = dbName(name);
219 		
220 		for(Collection<String> group : groups.getCollections()) {
221 			if (group.contains(dbName)) {
222 				return (List<String>) group;
223 			}
224 		}
225 		return Collections.emptyList();
226     }

229     /**
230      * Gets the primary name for the DB.
231      * It returns NULL for "unknown" database name. 
232      * 
233      * @param name case insensitive name (of a bio ID type/resource) name
234      * @return preferred name
235      */
236     public String getPrimaryDbName(String name) {
237 		List<String> names = getSynonymsForDbName(name);
238 		//get the first name
239 		return (names.isEmpty()) ? null : names.iterator().next();
240     }

243     /**
244      * Checks whether the ID format is valid for the database.
245      * Always use {@link #canCheckIdFormatIn(String)} before this method,
246      * because it may throw an exception if you do not.
247      * 
248      * @param db case insensitive
249      * @param id
250      * @return 'false' if matcher fails, 'true' otherwise
251      * @throws NullPointerException when no pattern available
252      */
253     public boolean checkIdFormat(String db, String id) {
254     	String name = getPrimaryDbName(db);
255     	Pattern p = dataPatterns.get(name);
256        	return p.matcher(id).find();
257     }
258 
259     /**
260      * @param name a database name (used in xrefs), case insensitive
261      * @return true if it's possible to check the format.
262      */
263     public boolean canCheckIdFormatIn(String name) {
264         String db = getPrimaryDbName(name);
265         return (db == null) ? false : dataPatterns.get(db) != null;
266     }

269     /**
270      * Gets the regular expression corresponding 
271      * to the database.
272      * 
273      * @param db a database name, case insensitive
274      * @return regular expression to check its ID
275      */
276     public String getRegexpString(String db) {
277     	String name = getPrimaryDbName(db);
278     	Pattern p = dataPatterns.get(name);
279     	return (p != null) ? p.pattern() : null;
280     }

283     //case insensitive
284     public boolean isSynonyms(String db1, String db2) {
285     	// dbName function is used here also to get right 
286     	// name capitalization
287     	return getSynonymsForDbName(db1).contains(dbName(db2));
288     }

306     /**
307      * Checks whether the db name is known (configured) 
308      * misspellings or unofficial name,
309      * which the Validator can recognize, report 
310      * warning, and replace with official names; and which
311      * otherwise, using MI or Miriam, cannot be resolved.
312      *
