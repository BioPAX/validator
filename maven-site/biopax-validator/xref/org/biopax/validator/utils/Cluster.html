1   package org.biopax.validator.utils;
2   
3   import java.util.*;
4   
5   public abstract class Cluster<T> {
6   	
7   	/**
8   	 * Implemented in concrete subclasses.
9   	 * Must be "symmetric"; may be "transitive" or not (you decide) -
10  	 * 
11  	 * @param a
12  	 * @param b
13  	 * @return true/false
14  	 */
15  	public abstract boolean match(T a, T b);
16  	
17  	
18  	
19  	/**
20  	 * Puts elements into sets of "similar" ones. 
21  	 * Every object belongs to exactly one cluster.
22  	 * Clusters contain at least one object.
23  	 * 
24  	 * If concrete {@link #match(Object, Object)} method
25  	 * is not transitive, this produces such groups, where,
26  	 * match(A,B) is not necessarily true for all pairs in
27  	 * the same cluster, but there exists Z, such as,
28  	 * match(X,Z) and match(Y,Z) are true.
29  	 * 
30  	 * @see #match(Object, Object)
31  	 * 
32  	 * @param elements
33  	 * @param clusterMaxSize
34  	 * @return
35  	 */
36  	public final Set<Set<T>> cluster(Collection<T> elements, int clusterMaxSize) {
37  		Set<Set<T>> clusters = new HashSet<Set<T>>();
38  		for (T u : new HashSet<T>(elements)) {
39  			Set<Set<T>> toMerge = new HashSet<Set<T>>();
40  			for (Set<T> clu : clusters) {
41  				if(!clu.isEmpty() && matchAny(u, clu)) {
42  					clu.add(u);
43  					toMerge.add(clu);
44  				}
45  			}
46  			
47  			if(toMerge.isEmpty()) { //create a new cluster 
48  				Set<T> group = new HashSet<T>();
49  				group.add(u);
50  				clusters.add(group);
51  			} else if (toMerge.size() > 1) {
52  				// merge several groups into the first one
53  				Iterator<Set<T>> it = toMerge.iterator();
54  				Set<T> group = it.next();
55  				assert !group.isEmpty();
56  				while(it.hasNext()) {
57  					Set<T> g = it.next();
58  					group.addAll(g);
59  					g.clear();
60  				}
61  			}
62  		}
63  		
64  		//remove empty groups
65  		Set<Set<T>> result = new HashSet<Set<T>>();
66  		for(Set<T> clu : clusters) {
67  			if(!clu.isEmpty())
68  				result.add(clu);
69  		}
70  		
71  		return result;
72  	}
73  	
74  	
75  	private boolean matchAny(T u, Set<T> clu) {
76  		for(T t : clu) {
77  			if(match(u, t) || match(t, u))
78  				return true;
79  		}
80  		return false;
81  	}
82  	
83  }
