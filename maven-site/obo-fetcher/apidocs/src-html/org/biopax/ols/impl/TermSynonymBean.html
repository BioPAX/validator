001package org.biopax.ols.impl;
002
003/*

005 */
006
007
008import java.util.Collection;
009
010import org.biopax.ols.DbXref;
011import org.biopax.ols.Term;
012import org.biopax.ols.TermSynonym;
013
014
015
016/**
017 * @author R. Cote
018 * @version $Id: TermSynonymBean.java,v 1.1 2006/03/23 12:32:41 rglcote Exp $
019 */
020public class TermSynonymBean implements TermSynonym {
021
022    /**
023     * <p>Represents ...</p>
024     */
025    private String parentTermPk;
026
027    /**
028     * <p>Represents ...</p>
029     */
030    private Term parentTerm = null;
031
032    /**
033     * <p>Represents ...</p>
034     */
035    private String synonym = null;
036
037    /**
038     * <p>Represents ...</p>
039     */
040    private String synonymPk;
041
042    /**
043     * <p>Represents ...</p>
044     */
045    private String synonymTypePk = null;
046
047    /**
048     * <p>Represents ...</p>
049     */
050    private Term synonymType = null;
051
052    /**
053     * <p>Represents ...</p>
054     */
055    private Collection<DbXref> synonymXrefs = null;
056
057    /**
058     * @return
059     */
060    public Term getParentTerm() {
061        return parentTerm;
062    }
063
064    /**
065     * @param _parentTerm
066     */
067    public void setParentTerm(final Term _parentTerm) {
068        parentTerm = _parentTerm;
069    }
070
071    /**
072     * @return
073     */
074    public String getSynonym() {
075        return synonym;
076    }
077
078    /**
079     * @param _synonym
080     */
081    public void setSynonym(final String _synonym) {
082        synonym = _synonym;
083    }
084
085    public String getSynonymPk() {
086        return synonymPk;
087    }
088
089    public void setSynonymPk(String synonymPk) {
090        this.synonymPk = synonymPk;
091    }
092
093    public String getSynonymTypePk() {
094        return synonymTypePk;
095    }
096
097    public void setSynonymTypePk(String synonymTypePk) {
098        this.synonymTypePk = synonymTypePk;
099    }
100
101    /**
102     * @return
103     */
104    public Term getSynonymType() {
105        return synonymType;
106    }
107
108    /**
109     * @param _synonymType
110     */
111    public void setSynonymType(final Term _synonymType) {
112        synonymType = _synonymType;
113    }
114
115    public Collection<DbXref> getSynonymXrefs() {
116        return synonymXrefs;
117    }
118
119    public void setSynonymXrefs(Collection<DbXref> synonymXrefs) {
120        this.synonymXrefs = synonymXrefs;
121    }
122
123    public String toString() {
124        String syntype;
125        if (synonymType != null)
126            syntype = synonymType.getName();
127        else
128            syntype = "unknown";
129        return "[Synonym]: " + synonym + "(" + syntype  + ")";
130    }
131
132    public String getParentTermPk() {
133        return parentTermPk;
134    }
135
136    public void setParentTermPk(String parentTermPk) {
137        this.parentTermPk = parentTermPk;
138    }
139
140    public boolean equals(Object o) {
141        if (this == o) return true;
142        if (!(o instanceof TermSynonymBean)) return false;
143
144        TermSynonymBean that = (TermSynonymBean) o;
145
146        if (parentTermPk != null ? !parentTermPk.equals(that.parentTermPk) : that.parentTermPk != null) return false;
147        if (synonym != null ? !synonym.equals(that.synonym) : that.synonym != null) return false;
148        if (synonymPk != null ? !synonymPk.equals(that.synonymPk) : that.synonymPk != null) return false;
149        if (synonymTypePk != null ? !synonymTypePk.equals(that.synonymTypePk) : that.synonymTypePk != null)
150            return false;
151
152        return true;
153    }
154
155    public int hashCode() {
156        int result;
157        result = (parentTermPk != null ? parentTermPk.hashCode() : 0);
158        result = 31 * result + (synonym != null ? synonym.hashCode() : 0);
159        result = 31 * result + (synonymPk != null ? synonymPk.hashCode() : 0);
160        result = 31 * result + (synonymTypePk != null ? synonymTypePk.hashCode() : 0);
161        return result;
162    }
163}
