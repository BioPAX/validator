026import java.util.Collection;
027
028import org.biopax.ols.DbXref;
029import org.biopax.ols.Term;
030import org.biopax.ols.TermSynonym;
031
032
033
034/**
035 * @author R. Cote
036 * @version $Id: TermSynonymBean.java,v 1.1 2006/03/23 12:32:41 rglcote Exp $
037 */
038public class TermSynonymBean implements TermSynonym {
039
040    /**
041     * <p>Represents ...</p>
042     */
043    private String parentTermPk;
044
045    /**
046     * <p>Represents ...</p>
047     */
048    private Term parentTerm = null;
049
050    /**
051     * <p>Represents ...</p>
052     */
053    private String synonym = null;
054
055    /**
056     * <p>Represents ...</p>
057     */
058    private String synonymPk;
059
060    /**
061     * <p>Represents ...</p>
062     */
063    private String synonymTypePk = null;
064
065    /**
066     * <p>Represents ...</p>
067     */
068    private Term synonymType = null;
069
070    /**
071     * <p>Represents ...</p>
072     */
073    private Collection<DbXref> synonymXrefs = null;
074
075    /**
076     * @return
077     */
078    public Term getParentTerm() {
079        return parentTerm;
080    }
081
082    /**
083     * @param _parentTerm
084     */
085    public void setParentTerm(final Term _parentTerm) {
086        parentTerm = _parentTerm;
087    }
088
089    /**
090     * @return
091     */
092    public String getSynonym() {
093        return synonym;
094    }
095
096    /**
097     * @param _synonym
098     */
099    public void setSynonym(final String _synonym) {
100        synonym = _synonym;
101    }
102
103    public String getSynonymPk() {
104        return synonymPk;
105    }
106
107    public void setSynonymPk(String synonymPk) {
108        this.synonymPk = synonymPk;
109    }
110
111    public String getSynonymTypePk() {
112        return synonymTypePk;
113    }
114
115    public void setSynonymTypePk(String synonymTypePk) {
116        this.synonymTypePk = synonymTypePk;
117    }
118
119    /**
120     * @return
121     */
122    public Term getSynonymType() {
123        return synonymType;
124    }
125
126    /**
127     * @param _synonymType
128     */
129    public void setSynonymType(final Term _synonymType) {
130        synonymType = _synonymType;
131    }
132
133    public Collection<DbXref> getSynonymXrefs() {
134        return synonymXrefs;
135    }
136
137    public void setSynonymXrefs(Collection<DbXref> synonymXrefs) {
138        this.synonymXrefs = synonymXrefs;
139    }
140
141    public String toString() {
142        String syntype;
143        if (synonymType != null)
144            syntype = synonymType.getName();
145        else
146            syntype = "unknown";
147        return "[Synonym]: " + synonym + "(" + syntype  + ")";
148    }
149
150    public String getParentTermPk() {
151        return parentTermPk;
152    }
153
154    public void setParentTermPk(String parentTermPk) {
155        this.parentTermPk = parentTermPk;
156    }
157
158    public boolean equals(Object o) {
159        if (this == o) return true;
160        if (!(o instanceof TermSynonymBean)) return false;
161
162        TermSynonymBean that = (TermSynonymBean) o;
163
164        if (parentTermPk != null ? !parentTermPk.equals(that.parentTermPk) : that.parentTermPk != null) return false;
165        if (synonym != null ? !synonym.equals(that.synonym) : that.synonym != null) return false;
166        if (synonymPk != null ? !synonymPk.equals(that.synonymPk) : that.synonymPk != null) return false;
167        if (synonymTypePk != null ? !synonymTypePk.equals(that.synonymTypePk) : that.synonymTypePk != null)
168            return false;
169
170        return true;
171    }
172
173    public int hashCode() {
174        int result;
175        result = (parentTermPk != null ? parentTermPk.hashCode() : 0);
176        result = 31 * result + (synonym != null ? synonym.hashCode() : 0);
177        result = 31 * result + (synonymPk != null ? synonymPk.hashCode() : 0);
178        result = 31 * result + (synonymTypePk != null ? synonymTypePk.hashCode() : 0);
179        return result;
180    }
181}
