001package org.biopax.psidev.ontology_manager;
002
003/*
004 *
005 */
006
007import java.util.Properties;
008import java.util.Set;
009
010import org.biopax.psidev.ontology_manager.impl.OntologyLoaderException;
011
012
013public interface OntologyManager {
014
015        void putOntology(String ontologyID, OntologyAccess ontologyAccess);
016
017
018        Set<String> getOntologyIDs();
019
020
021        OntologyAccess getOntology(String ontologyID);
022
023        
024        boolean containsOntology(String ontologyID);
025
026
027        void loadOntologies(Properties cfg)
028                        throws OntologyLoaderException;
029
030        /**
031         * Search for terms using a preferred name or synonym.
032         * The search is case insensitive.
033         * It iterates through all loaded ontologies, so use with caution!
034         * 
035         * @param name - term name (not ID)
036         * @return
037         */
038        Set<OntologyTermI> searchTermByName(String name);
039        
040        /**
041         * Search for terms by name or synonym.
042         * The search is case insensitive.
043         * It still iterates over all available ontologies, but
044         * skips ones other than specified in the second parameter.
045         * 
046         * @param name - term name (not ID)
047         * @param ontologies to look into
048         * @return
049         */
050        Set<OntologyTermI> searchTermByName(String name, Set<String> ontologies);
051        
052        
053        /**
054         * Finds an ontology term by its accession.
055         * 
056         * @param acc
057         * @return
058         */
059        OntologyTermI findTermByAccession(String acc);
060}
