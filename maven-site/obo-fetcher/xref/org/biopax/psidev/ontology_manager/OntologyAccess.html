26  import java.util.Collection;
27  import java.util.Set;
28  
29  /**
30   * Defines what can be asked to an ontology.
31   *
32   * @author Florian Reisinger
33   * @author Samuel Kerrien (skerrien@ebi.ac.uk)
34   * @author rodche (baberlab.org) - re-factoring
35   * @since 2.0.0
36   */
37  public interface OntologyAccess {
38  
39      /**
40       * This method builds a set of all allowed terms based on the specified parameters.
41       *
42       * @param accession     the accession number of the term wanted.
43       * @param allowChildren whether child terms are allowed.
44       * @param useTerm       whether to include the specified term ID in the set of allowed IDs.
45       * @return a non null set of allowed ontology terms for the specified parameters.
46       */
47      public Set<OntologyTermI> getValidTerms( String accession, boolean allowChildren, boolean useTerm );
48  
49      /**
50       * Search a term by accession number.
51       *
52       * @param accession the accession to be searched for.
53       * @return an ontology term or null if not found.
54       */
55      public OntologyTermI getTermForAccession( String accession );
56      
57      void setName(String name);
58         
59      String getName();
60      
61      boolean hasTerms();
62  
63      OntologyTermI search( String id );
64  
65      Collection<OntologyTermI> getRoots();
66  
67      Collection<OntologyTermI> getOntologyTerms();
68  
69      Collection<OntologyTermI> getObsoleteTerms();
70  
71      boolean isObsolete( OntologyTermI term );
72  
73      Set<OntologyTermI> getDirectParents( OntologyTermI term );
74  
75      Set<OntologyTermI> getDirectChildren( OntologyTermI term );
76  
77      Set<OntologyTermI> getAllParents( OntologyTermI term );
78  
79      Set<OntologyTermI> getAllChildren( OntologyTermI term );
80      
81      void addLink( String parentId, String childId );
82      
83      void addTerm( OntologyTermI term );
84      
85      void addObsoleteTerm( OntologyTermI term );
86  
87  }
