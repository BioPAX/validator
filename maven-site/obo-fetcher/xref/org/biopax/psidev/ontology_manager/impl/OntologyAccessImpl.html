60      public Set<OntologyTermI> getValidTerms( String accession, boolean allowChildren, boolean useTerm ) {
61          Set<OntologyTermI> collectedTerms = new HashSet<OntologyTermI>();
62  
63          final OntologyTermI term = getTermForAccession( accession );
64          if ( term != null ) {
65              if ( useTerm ) {
66                  collectedTerms.add( term );
67              }
68  
69              if ( allowChildren ) {
70                  collectedTerms.addAll( getAllChildren( term ) );
71              }
72          }
73  
74          return collectedTerms;
75      }

78      /**
79       * Pool of all term contained in that ontology.
80       */
81      private Collection<OntologyTermI> ontologyTerms = new ArrayList<OntologyTermI>( 1024 );

83      /**
84       * Represent the relationship: child -> parents.
85       */
86      private final Map<OntologyTermI, Set<OntologyTermI>> parents = new HashMap<OntologyTermI, Set<OntologyTermI>>();

88      /**
89       * Represent the relationship: parent -> children.
90       */
91      private final Map<OntologyTermI, Set<OntologyTermI>> children = new HashMap<OntologyTermI, Set<OntologyTermI>>();

93      /**
94       * Mapping of all OboTerm by their ID.
95       */
96      private Map<String, OntologyTermI> id2ontologyTerm = new HashMap<String, OntologyTermI>( 1024 );

99       * Collection of root terms of that ontology. A root term is defined as follow: term having no parent.

101     private Collection<OntologyTermI> roots = null;

111     /**
112      * Add a new Term in the pool. It will be indexed by its ID.
113      *
114      * @param term the OntologyTerm to add in that OntologyAccess.
115      */
116     public void addTerm( OntologyTermI term ) {
117 
118         ontologyTerms.add( term );
119         String id = term.getTermAccession();
120         if ( id2ontologyTerm.containsKey( id ) ) {
121             OntologyTermI old = id2ontologyTerm.get( id );
122             log.error( "WARNING: 2 Objects have the same ID (" + id 
123             	+ "), the old one is being replaced. old: " 
124             	+ old.getPreferredName() + " new: " + term.getPreferredName() );
125         }
126 
127         id2ontologyTerm.put( id, term );
128 
129         flushRootsCache();
130     }
131 
132     /**
133      * Create a relashionship parent to child between two OntologyTerm.
134      *
135      * @param parentId The parent term.
136      * @param childId  The child term.
137      */
138 	public void addLink(String parentId, String childId) {
139 
140 		OntologyTermI child = id2ontologyTerm.get(childId);
141 		OntologyTermI parent = id2ontologyTerm.get(parentId);
142 
143 		if (child == null || parent == null) {
144 			throw new NullPointerException("You must give a non null " +
145 				"child/parent for addLink method!");
146 		} else {
147 
148 			if (!children.containsKey(parent)) {
149 				children.put(parent, new HashSet<OntologyTermI>());
150 			}
151 
152 			if (!parents.containsKey(child)) {
153 				parents.put(child, new HashSet<OntologyTermI>());
154 			}
155 
156 			children.get(parent).add(child);
157 			parents.get(child).add(parent);
158 
159 			flushRootsCache();
160 		}
161 	}
162 
163     /**
164      * Remove the Root cache from memory.<br/> That method should be called every time the collection of OntologyTerm is
165      * altered.
166      */
167     private void flushRootsCache() {
168         if ( roots != null ) {
169             // flush roots cache
170             roots.clear();
171             roots = null;
172         }
173     }

194     public boolean hasParent( OntologyTermI term ) {
195         return parents.containsKey( term );
196     }
197 
198     /**
199      * Get the Root terms of the ontology. The way to get it is as follow: pick a term at random, and go to his highest
200      * parent.
201      *
202      * @return a collection of Root term.
203      */
204     public Collection<OntologyTermI> getRoots() {
205 
206         if ( roots != null ) {
207             return roots;
208         }
209 
210         // it wasn't precalculated, then do it here...
211         roots = new HashSet<OntologyTermI>();
212 
213         for ( Iterator iterator = ontologyTerms.iterator(); iterator.hasNext(); ) {
214             OntologyTermI ontologyTerm = ( OntologyTermI ) iterator.next();
215 
216             if ( !hasParent( ontologyTerm ) ) {
217                 roots.add( ontologyTerm );
218             }
219         }
220 
221         if ( roots.isEmpty() ) {
222             return Collections.emptyList();
223         }
224 
225         return roots;
226     }

237     public void addObsoleteTerm( OntologyTermI term ) {
238         if ( term == null ) {
239             throw new IllegalArgumentException( "You must give a non null term" );
240         }
241 
242         log.debug( "Adding obsolete term: " + term.getTermAccession() + " " + term.getPreferredName() );
243         
244         obsoleteTerms.add( term );
245     }

260     public Set<OntologyTermI> getDirectParents( OntologyTermI term ) {
261         final Set<OntologyTermI> directParents = parents.get( term );
262         if ( directParents == null ) {
263             return Collections.emptySet();
264         } else {
265             return directParents;
266         }
267     }
268 
269     public Set<OntologyTermI> getDirectChildren( OntologyTermI term ) {
270         final Set<OntologyTermI> directChildren = children.get( term );
271         if ( directChildren == null ) {
272             return Collections.emptySet();
273         } else {
274             return directChildren;
275         }
276     }
277 
278     public Set<OntologyTermI> getAllParents( OntologyTermI term ) {
279         Set<OntologyTermI> parents = new HashSet<OntologyTermI>();
280         getAllParents( term, parents );
281         return parents;
282     }
283 
284     private void getAllParents( OntologyTermI term, Set<OntologyTermI> parents ) {
285         final Collection<OntologyTermI> directParents = getDirectParents( term );
286         parents.addAll( directParents );
287         for ( OntologyTermI parent : directParents ) {
288             getAllParents( parent, parents );
289         }
290     }
291 
292     public Set<OntologyTermI> getAllChildren( OntologyTermI term ) {
293         Set<OntologyTermI> children = new HashSet<OntologyTermI>();
294         getAllChildren( term, children );
295         return children;
296     }
297 
298     private void getAllChildren( OntologyTermI term, Set<OntologyTermI> children ) {
299          getAllChildren( "", term, children, new HashSet(512) );
300     }
301 
302     private void getAllChildren( String prefix, OntologyTermI term, Set<OntologyTermI> children, Set<String> traversedChildren ) {
303         if( traversedChildren.contains( term.getTermAccession() ) ) {
304 //            System.out.println( prefix.replaceAll( " ", "#" )+" > "+ term.getTermAccession() +" / "+ term.getPreferredName() +" )" );
305             return;
306         } else {
307 //            System.out.println( prefix +" > "+ term.getTermAccession() +" / "+ term.getPreferredName() +" )" );
308         }
309         final Collection<OntologyTermI> directChildren = getDirectChildren( term );
310         traversedChildren.add( term.getTermAccession() );
311         children.addAll( directChildren );
312         for ( OntologyTermI child : directChildren ) {
313             getAllChildren( prefix+"     ", child, children, traversedChildren );
314         }
315     }

320     public void print() {
321         log.info( ontologyTerms.size() + " terms to display." );
322         final Collection<OntologyTermI> roots = getRoots();
323 
324         log.info( this.roots.size() + " root(s) found." );
325 
326         for ( OntologyTermI root : roots ) {
327             print( root );
328         }
329     }
330 
331     private void print( OntologyTermI term, String indent ) {
332 
333         log.info( indent + term.getTermAccession() + "   " + term.getPreferredName() );
334         for ( OntologyTermI child : getDirectChildren( term ) ) {
335             print( child, indent + "  " );
336         }
337     }
338 
339     public void print( OntologyTermI term ) {
340         print( term, "" );
341     }
