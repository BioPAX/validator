102     /**
103      * will return a collection of TermBean accesssion strings that correspond to the root term
104      *
105      * @throws IllegalStateException if the parser has not been initialized
106      */
107     protected Collection<String> getRootTerms() {
108         HashSet<String> retval = new HashSet<String>();
109         if (parser != null) {
110             Set<OBOObject> roots = parser.getRootTerms(useGreedy);
111             if (roots != null) {
112                 for (OBOObject root : roots) {
113                     retval.add(safeTrim(root.getID()));
114                 }
115             }
116         } else {
117             throw new IllegalStateException("Parser has not been initialized. Did you run configure()?");
118         }
119         return retval;
120     }

122     /**
123      * Will iterate through all parsed terms and generate memory structure suitable for loading to db
124      *
125      * @throws IllegalStateException if the parser has not been initialized
126      */
127     protected void process() {
128 
129         //returns unmodifiable set, so need to create a modifiable one
130         Set<OBOObject> terms = new HashSet<OBOObject>();
131 
132         //sanity check
133         if (parser == null) {
134             throw new IllegalStateException("parser has not been initialized. Did you run configure()?");
135         }
136 
137         //tmp collection to store terms
138         Collection<OBOObject> toAdd;
139 
140         //add all terms - this will include obsolete and roots
141         //sanity check to avoid NPE
142         toAdd = parser.getTerms();
143         if (toAdd != null) {
144             terms.addAll(toAdd);
145         }
146 
147         //create common objects needed to build references and synonyms
148         initializeCommonObjects();
149 
150         //process all terms
151         logger.info("Total Terms to process: " + terms.size());
152 
153         int count = 0;
154         for (Object obj : terms) {
155             count++;
156             if (obj instanceof OBOObject) {
157                 processTerm((OBOObject) obj);
158                 if (count % 1000 == 0) {
159                     logger.debug("Terms Processed: " + count);
160                 }
161             } else {
162                 logger.info("Ignored object: " + obj.toString());
163             }
164         }
165         logger.info("Term processing done");
166 
167         //need to iterate again to build relationships
168         logger.info("Creating relationships");
169         count = 0;
170         for (Object obj : terms) {
171             count++;
172             if (obj instanceof OBOObject) {
173                 processTermRelationships((OBOObject) obj);
174                 if (count % 1000 == 0) {
175                     logger.debug("Relationships Processed: " + count);
176                 }
177             } else {
178                 logger.info("Ignored object: " + obj.toString());
179             }
180         }
181         logger.info("Relationship processing done");
182 
183         //add to OntologyAccess
184         ontBean.setTerms(ontologyTerms.values());
185 
186     }

347     /**
348      * helper method to create a TermBean given a name and a definition. This is mostly for terms
349      * associated with an ontology but not defined by it (eg relations types, synonym types, etc)
350      *
351      * @param name - this value will be used to set the term.name and term.identifier
352      * @param def  - the term definition.
353      * @return a valid TermBean object
354      * @throws IllegalArgumentException if the term name is null.
355      */
356     private TermBean initializeTermBean(String name, String def) {
357         if (name != null && !"".equals(name.trim())) {
358             TermBean bean = new TermBean();
359             bean.setName(name.trim());
360             bean.setIdentifier(SHORT_NAME + ":" + name.toUpperCase().trim());
361             bean.setDefinition(safeTrim(def));
362             bean.setParentOntology(ontBean);
363             //must set PK here because OJB will now not set it automatically
364             //PK will be term_ac+ont+fully_loaded_false
365             bean.setTermPk(bean.getIdentifier() + SHORT_NAME + "0");
366             return bean;
367         } else {
368             throw new IllegalArgumentException("Can't have a non-null term name!");
369         }
370     }

375      * @param name      - this value will be used to set the term.name

388             throw new IllegalArgumentException("Can't have a non-null term name!");

392     /**
393      * This method will convert an OBOEdit model term into a valid TermBean, while creating synonyms,
394      * xrefs and annotations. The valid TermBean generated will be added to a global HashMap that will
395      * be used at a later stage.
396      *
397      * @param obj being an OBOObject object obtained from the parser
398      */
399     protected void processTerm(OBOObject obj) {
400 
401         if (obj.getID().startsWith("obo:")) {
402             /*
403                 obo:datatype
404                 obo:property
405                 obo:class
406             */
407             logger.debug("bogus term: " + obj.getID());
408             return;
409         }
410 
411         TermBean trm = new TermBean();
412         //must set PK here because OJB will now not set it automatically
413         //PK will be term_ac+ont+fully_loaded_false
414         trm.setTermPk(safeTrim(obj.getID()) + SHORT_NAME + "0");
415         if (trm.getTermPk().length() > 255) {
416             throw new IllegalStateException("term PK longer than allowed database column length: " + trm.getTermPk());
417         }
418 
419         //trim definition 
420         trm.setDefinition(safeTrim(obj.getDefinition()));
421         if (trm.getDefinition() != null && trm.getDefinition().length() > 4000) {
422             logger.warn("term definition longer than allowed database column length - truncating" + trm.getIdentifier());
423             trm.setDefinition(trm.getDefinition().substring(0, 4000));
424         }
425 
426         //trim ID 
427         trm.setIdentifier(safeTrim(obj.getID()));
428         if (trm.getIdentifier() != null && trm.getIdentifier().length() > 255) {
429             logger.warn("term identifier longer than allowed database column length - truncating" + trm.getIdentifier());
430             trm.setIdentifier(trm.getIdentifier().substring(0, 255));
431         }
432         //set as root term if required
433         if (rootTerms.contains(safeTrim(obj.getID()))) {
434             trm.setRootTerm(true);
435             logger.info(obj.getID() + " is a root term");
436         }
437         //trim name 
438         trm.setName(safeTrim(obj.getName()));
439         //trim namespace 
440         if (trm.getName() != null && trm.getName().length() > 2000) {
441             logger.warn("term name longer than allowed database column length - truncating" + trm.getIdentifier());
442             trm.setName(trm.getName().substring(0, 2000));
443         }
444 
445         Namespace nspace = obj.getNamespace();
446         if (nspace != null) {
447             trm.setNamespace(safeTrim(nspace.getID()));
448             if (trm.getNamespace() != null && trm.getNamespace().length() > 255) {
449                 logger.warn("term namespace longer than allowed database column length - truncating" + trm.getIdentifier());
450                 trm.setNamespace(trm.getNamespace().substring(0, 255));
451             }
452         }
453         //set if obsolete
454         trm.setObsolete(obj.isObsolete());
455         //set parent ontology
456         trm.setParentOntology(ontBean);
457         //process synonyms
458         trm.setSynonyms(processSynonyms(obj, trm));
459         //process xrefs
460         trm.setXrefs(processXrefs(obj, trm));
461         //process annotations
462         trm.setAnnotations(processAnnotations(obj, trm));
463 
464         //set number of children
465         int nbChild = 0;
466         if (obj.getChildren() != null) {
467             nbChild = obj.getChildren().size();
468         }
469 
470         //set leaf status
471         if (nbChild > 0) {
472             trm.setLeaf(false);
473         } else {
474             trm.setLeaf(true);
475         }
476 
477 //        if (obj instanceof Instance) {
478 //            trm.setInstance(true);
479 //            //store type (eg objID is_instance_of typeID
480 //            instances.put(obj.getType().getID(), obj.getID());
481 //        } else {
482 //            trm.setInstance(false);
483 //        }
484 
485         //add to global storage
486         ontologyTerms.put(trm.getIdentifier(), trm);
487 
488     }

490     /**
491      * internal method to create AnnotationBeans objects from values extracted from an OBOEdit term object
492      * and properly setup associations to the parent OLS term objet. Annotations can include comments and
493      * replacement term ids for obsolete or misused terms, as well as defined subsets.
494      *
495      * @param obj - the OBOEdit object to extract information from
496      * @param trm - the parent term to link the annotations to
497      * @return a collection of properly created AnnotationBeans
498      */
499     private Collection<Annotation> processAnnotations(OBOObject obj, TermBean trm) {
500 
501         ArrayList<Annotation> retval = new ArrayList<Annotation>();
502 
503         String comment = safeTrim(obj.getComment());
504         if (comment != null) {
505             AnnotationBean annot = new AnnotationBean();
506             annot.setAnnotationStringValue(comment);
507             annot.setAnnotationType(Annotation.OBO_COMMENT);
508             if (comment != null && comment.length() > 2000) {
509                 logger.warn("annotation comment longer than allowed database column length - truncating " + trm.getIdentifier());
510                 annot.setAnnotationStringValue(annot.getAnnotationStringValue().substring(0, 2000));
511             }
512 
513             annot.setParentTerm(trm);
514             retval.add(annot);
515         }
516 
517         Set<ObsoletableObject> considers = obj.getConsiderReplacements();
518         for (ObsoletableObject obsolete : considers) {
519             AnnotationBean annot = new AnnotationBean();
520             annot.setAnnotationType(Annotation.OBO_CONSIDER_REPLACEMENT);
521             String val = obsolete.getID();
522             if (obsolete.getName() != null) {
523                 val += ": " + obsolete.getName();
524             }
525             annot.setAnnotationStringValue(val);
526             if (val != null && val.length() > 2000) {
527                 logger.warn("annotation value longer than allowed database column length - truncating " + trm.getIdentifier());
528                 annot.setAnnotationStringValue(annot.getAnnotationStringValue().substring(0, 2000));
529             }
530             annot.setParentTerm(trm);
531             retval.add(annot);
532         }
533 
534         Set<ObsoletableObject> replacers = obj.getReplacedBy();
535         for (ObsoletableObject replacedby : replacers) {
536             AnnotationBean annot = new AnnotationBean();
537             annot.setAnnotationType(Annotation.OBO_REPLACED_BY);
538             String val = replacedby.getID();
539             if (replacedby.getName() != null) {
540                 val += ": " + replacedby.getName();
541             }
542             annot.setAnnotationStringValue(val);
543             if (val != null && val.length() > 2000) {
544                 logger.warn("annotation value longer than allowed database column length - truncating " + trm.getIdentifier());
545                 annot.setAnnotationStringValue(annot.getAnnotationStringValue().substring(0, 2000));
546             }
547             annot.setParentTerm(trm);
548             retval.add(annot);
549         }
550 
551         Set<TermSubset> subsets = obj.getSubsets();
552         for (TermSubset subset : subsets) {
553             AnnotationBean annot = new AnnotationBean();
554             annot.setAnnotationType(Annotation.SUBSET + "_" + subset.getName());
555             String val = subset.getDesc();
556             annot.setAnnotationStringValue(val);
557             if (val != null && val.length() > 2000) {
558                 logger.warn("annotation value longer than allowed database column length - truncating " + trm.getIdentifier());
559                 annot.setAnnotationStringValue(annot.getAnnotationStringValue().substring(0, 2000));
560             }
561             annot.setParentTerm(trm);
562             retval.add(annot);
563         }
564 
565         Set<PropertyValue> propVal = obj.getPropertyValues();
566         for (PropertyValue pv : propVal) {
567             AnnotationBean annot = new AnnotationBean();
568             //property_value: EFO:definition_editor "James Malone" xsd:string
569             //parses to property = property_value
570             //             value = EFO:definition_editor "James Malone" xsd:string
571             //so manually process the value to something more informative
572             //          property = EFO:definition_editor
573             //             value = "James Malone"
574             try {
575                 if (pv.getValue() == null) {
576                     //invalidly constucted property_value!
577                     logger.warn("Error parsing property_value - Ignoring null value: " + pv.toString());
578                     continue;
579                 }
580 
581                 int ndx;
582 
583                 //parse property type
584                 String tmpStr = pv.getValue();
585                 ndx = tmpStr.indexOf(' ');
586                 if (ndx > 0) {
587                     tmpStr = tmpStr.substring(0, ndx).trim();
588                     if (tmpStr.endsWith(":")) {
589                         tmpStr = tmpStr.substring(0, tmpStr.length() - 1);
590                     }
591                     logger.debug("Setting property type: " + tmpStr);
592                     annot.setAnnotationType(tmpStr.trim());
593                     if (annot.getAnnotationType() != null && annot.getAnnotationType().length() > 2000) {
594                         logger.warn("annotation type longer than allowed database column length - truncating " + trm.getIdentifier());
595                         annot.setAnnotationType(annot.getAnnotationType().substring(0, 2000));
596                     }
597 
598                     //parse property value
599                     tmpStr = pv.getValue();
600 
601                     tmpStr = tmpStr.substring(ndx + 1);
602                     ndx = tmpStr.indexOf("xsd:");
603                     if (ndx > 0) {
604                         tmpStr = tmpStr.substring(0, ndx);
605                     }
606                     tmpStr = tmpStr.trim();
607                     if (tmpStr.startsWith("\"") && tmpStr.endsWith("\"")) {
608                         tmpStr = tmpStr.substring(1, tmpStr.length() - 1);
609                     }
610                     logger.debug("Setting property value: " + tmpStr.trim());
611                     annot.setAnnotationStringValue(tmpStr.trim());
612                     if (tmpStr != null && tmpStr.length() > 2000) {
613                         logger.warn("annotation value longer than allowed database column length - truncating " + trm.getIdentifier());
614                         annot.setAnnotationStringValue(annot.getAnnotationStringValue().substring(0, 2000));
615                     }
616 
617                 } else {
618 
619                     //this will capture occations where people just put random key-value tags
620                     //as annotations
621                     annot.setAnnotationType(pv.getProperty());
622                     if (pv.getProperty() != null && pv.getProperty().length() > 2000) {
623                         logger.warn("annotation type longer than allowed database column length - truncating " + trm.getIdentifier());
624                         annot.setAnnotationType(annot.getAnnotationType().substring(0, 2000));

887     /**
888      * internal helper method to create TermPathBeans for a given term. This method will
889      * precompute all paths from a parent to all its children for the 3 major relationship types:
890      * IS_A, PART_OF and DEVELOPS_FROM. The PART_OF and DEVELOPS_FROM relations can traverse IS_A
891      * relations for maximal completeness and still be semantically correct, but IS_A relationships
892      * cannot traverse other relation types.
893      * <pre>
894      *        term1
895      *            |_ child1        child1 IS_A term1
896      *            |_ child2        child2 IS_A term1
897      *                             subject pred object
898      * </pre>
899      *
900      * @param obj - the OBOEdit term object to extract information from
901      * @param trm - the OLS parent term to link to
902      * @return a Collection of valid TermRelationshipBeans
903      */
904     private Collection<TermPath> processPaths(OBOObject obj, TermBean trm) {
905 
906         HashSet<TermPath> retval = new HashSet<TermPath>();
907 
908         HashMap<String, Integer> paths = parser.computeChildPaths(1, IS_A_SET, obj);
909         retval.addAll(createTermPathBeans(paths, Constants.IS_A_RELATION_TYPE_ID, IS_A, trm));
910 
911         //the part_of relation can traverse is_a relations to generate term_paths
912         //so the set passed to computeChildPaths needs to contain both PART_OF and IS_A labels.
913         HashSet<String> traversingSet = new HashSet<String>();
914         traversingSet.addAll(PART_OF_SET);
915         traversingSet.addAll(IS_A_SET);
916         paths = parser.computeChildPaths(1, traversingSet, obj);
917         retval.addAll(createTermPathBeans(paths, Constants.PART_OF_RELATION_TYPE_ID, PART_OF, trm));
918 
919         //the dev_from relation can traverse is_a relations to generate term_paths
920         //so the set passed to computeChildPaths needs to contain both DEV_FROM and IS_A labels.
921         traversingSet.clear();
922         traversingSet.addAll(DEV_FROM_SET);
923         traversingSet.addAll(IS_A_SET);
924         paths = parser.computeChildPaths(1, traversingSet, obj);
925         retval.addAll(createTermPathBeans(paths, Constants.DEVELOPS_FROM_RELATION_TYPE_ID, DEVELOPS_FROM, trm));
926 
927         return retval;
928     }

932      */

934                                                      TermBean relationBean, TermBean trm)
935     {
936         HashSet<TermPath> retval = new HashSet<TermPath>();
937

946         */
947         Term objTrm;

963                 //set relationshipType
964                 tpb.setRelationshipTypeId((long) relationTypeId);
965                 //set ontology
966                 tpb.setParentOntology(ontBean);
967                 //add to retval
968                 retval.add(tpb);
969             } else {
