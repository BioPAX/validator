093    /**
094     * JSP pages and RESTful web services controller, the main one that checks BioPAX data.
095     * All parameter names are important, i.e., these are part of public API (for clients)
096     * 
097     * Framework's built-in parameters:
098     * @param request the web request object (may contain multi-part data, i.e., multiple files uploaded)
099     * @param response 
100     * @param mvcModel Spring MVC Model
101     * @param writer HTTP response writer
102     * 
103     * BioPAX Validator/Normalizer query parameters:
104     * @param url
105     * @param retDesired
106     * @param autofix
107     * @param filter
108     * @param maxErrors
109     * @param profile
110     * @param normalizer binds to three boolean options: normalizer.fixDisplayName, normalizer.inferPropertyOrganism, normalizer.inferPropertyDataSource
111     * @return
112     * @throws IOException
113     */
114    @RequestMapping(value="/check", method=RequestMethod.POST)
115        public String check(
116                HttpServletRequest request, HttpServletResponse response, 
117                Model mvcModel, Writer writer,
118                @RequestParam(required=false) String url, 
119                @RequestParam(required=false) String retDesired,
120                @RequestParam(required=false) Boolean autofix,
121                @RequestParam(required=false) Behavior filter,
122                @RequestParam(required=false) Integer maxErrors,
123                @RequestParam(required=false) String profile, 
124                //normalizer!=null when called from the JSP; 
125                //but it's usually null when from the validator-client or a web script
126                @ModelAttribute("normalizer") Normalizer normalizer
127        ) throws IOException
128        {
129        Resource resource = null; // a resource to validate
130        
131        // create the response container
132        ValidatorResponse validatorResponse = new ValidatorResponse();
133        
134        if(url != null && url.length()>0) {
135                if(url != null) 
136                        log.info("url : " + url);
137                try {
138                        resource = new UrlResource(url);
139                } catch (MalformedURLException e) {
140                        mvcModel.addAttribute("error", e.toString());
141                        return "check";
142                        }               
143                
144                try {
145                                Validation v = execute(resource, resource.getDescription(), maxErrors, autofix, filter, profile, normalizer);
146                                validatorResponse.addValidationResult(v);
147                        } catch (Exception e) {
148                                return errorResponse(mvcModel, "check", "Exception: " + e);
149                        }               
150                
151        } else if (request instanceof MultipartHttpServletRequest) {
152                        MultipartHttpServletRequest multiRequest = (MultipartHttpServletRequest) request;
153                        Map files = multiRequest.getFileMap();
154                        Assert.state(!files.isEmpty(), "No files to validate");
155                        for (Object o : files.values()) {
156                                MultipartFile file = (MultipartFile) o;
157                                String filename = file.getOriginalFilename();
158                                // a workaround (for some reason there is always a no-name-file;
159                                // this might be a javascript isue)
160                                if(file.getBytes().length==0 || filename==null || "".equals(filename)) 
161                                        continue;
162
163                                log.info("check : " + filename);
164                                
165                                resource = new ByteArrayResource(file.getBytes());                              
166                                
167                                try {
168                                        Validation v = execute(resource, filename, maxErrors, autofix, filter, profile, normalizer);
169                                        validatorResponse.addValidationResult(v);
170                                } catch (Exception e) {
171                                return errorResponse(mvcModel, "check", "Exception: " + e);
172                                }               
173                        
174                        }                       
175                } else {
176                        return errorResponse(mvcModel, 
177                                "check", "No BioPAX input source provided!");
178                }
179                
180        if("xml".equalsIgnoreCase(retDesired)) {
181                response.setContentType("application/xml");
182                ValidatorUtils.write(validatorResponse, writer, null);
183        } else if("html".equalsIgnoreCase(retDesired)) {
184                /* could also use ValidatorUtils.write with a xml-to-html xslt source
185                 but using JSP here makes it easier to keep the same style, header, footer*/
186                        mvcModel.addAttribute("response", validatorResponse);
187                        return "groupByCodeResponse";
188                } else { // owl only
189                        response.setContentType("text/plain");
190                        // write all the OWL results one after another TODO any better solution?
191                        for(Validation result : validatorResponse.getValidationResult()) 
192                        {
193                                if(result.getModelData() != null)
194                                        writer.write(result.getModelData() + NEWLINE);
195                                else
196                                        // empty result
197                                        writer.write("<?xml version=\"1.0\" encoding=\"UTF-8\"?>" +
198                                                "<rdf:RDF></rdf:RDF>" + NEWLINE);
199                        }
200                }
201        
202        return null; // (Writer is used instead)
203    }
204        
205
206        private Validation execute(Resource biopaxResource, String resultName, Integer maxErrors,
207                        Boolean autofix, Behavior errorLevel, String profile, Normalizer normalizer) 
208                                        throws IOException 
209        {       
210        int errMax = 0;
211        if(maxErrors != null) {
212                errMax = maxErrors.intValue();
213                log.info("Limiting max no. errors to " + maxErrors);
214        }
215    
216        boolean isFix = Boolean.TRUE.equals(autofix);
217        
218        Validation validationResult = 
219                new Validation(new IdentifierImpl(), resultName, isFix, errorLevel, errMax, profile);
220        
221                //run the biopax-validator (this updates the validationResult object)
222        validator.importModel(validationResult, biopaxResource.getInputStream());
223                validator.validate(validationResult);
224        validator.getResults().remove(validationResult);                
225        
226        if(isFix) { // do normalize too
227                if(normalizer == null) {//e.g., when '/check' called from a client/script, not JSP
228                        normalizer = new Normalizer();
229                }
230                org.biopax.paxtools.model.Model m = (org.biopax.paxtools.model.Model) validationResult.getModel();
231                        normalizer.normalize(m);//this further modifies the validated and auto-fixed model
232                        //update the serialized model (BioPAX RDF/XML)
233                        //for the client to get it (to possibly, unmarshall)
234                        validationResult.setModel(m);
235                        validationResult.setModelData(SimpleIOHandler.convertToOwl(m));
236        } else {
237                validationResult.setModelData(null);
238                validationResult.setModel(null);
239        }
240        
241        return validationResult;
242        }

245    private String errorResponse(Model model, String viewName, String msg) {
246        model.addAttribute("error", msg);
247                return viewName;
248    }

251    /**
252     * Prints the XML schema.
253     * 
254     * @param writer
255     * @throws IOException
256     */
257    @RequestMapping(value="/schema")
258    public void getSchema(Writer writer, HttpServletResponse response) 
259                throws IOException 
260    {
261                log.debug("XML Schema requested.");
262        
263        BufferedReader bis = new BufferedReader(new InputStreamReader(
264                LOADER.getResource("classpath:org/biopax/validator/api/schema/schema1.xsd")
265                        .getInputStream(), "UTF-8"));
266        
267        response.setContentType("application/xml");
268        
269        String line = null;
270        while((line = bis.readLine()) != null) {
271                writer.write(line + NEWLINE);
272        }
273    }
