025import java.util.Collection;
026
027/**
028 * @author R. Cote
029 * @version $Id: Term.java,v 1.13 2006/03/23 12:34:17 rglcote Exp $
030 */
031public interface Term {
032    /**
033     * <p>Does ...</p>
034     *
035     * @return
036     */
037    public String getDefinition();
038
039    /**
040     * <p>Does ...</p>
041     *
042     * @return
043     */
044    public String getIdentifier();
045
046    /**
047     * <p>Does ...</p>
048     *
049     * @return
050     */
051    public boolean isObsolete();
052
053    /**
054     * <p>Does ...</p>
055     *
056     * @return
057     */
058    public boolean isRootTerm();
059
060    /**
061     * <p>Does ...</p>
062     *
063     * @return
064     */
065    public boolean isLeaf();
066
067    /**
068     * <p>Does ...</p>
069     *
070     * @return
071     */
072    public boolean isInstance();
073
074    /**
075     * <p>Does ...</p>
076     *
077     * @return
078     */
079    public String getName();
080
081    /**
082     * <p>Does ...</p>
083     *
084     * @return
085     */
086    public Ontology getParentOntology();
087
088    /**
089     * <p>Does ...</p>
090     *
091     * @return
092     */
093    public Collection<TermSynonym> getSynonyms();
094
095    /**
096     * <p>Does ...</p>
097     *
098     * @return
099     */
100    public Collection<TermRelationship> getRelationships();
101
102    /**
103     * <p>Does ...</p>
104     *
105     * @return
106     */
107    public Collection<TermPath> getPaths();
108
109    /**
110     * <p>Does ...</p>
111     *
112     * @return
113     */
114    public String getNamespace();
115
116    /**
117     * <p>Does ...</p>
118     *
119     * @return
120     */
121    public String getTermPk();
122
123    /**
124     * <p>Does ...</p>
125     *
126     * @return
127     */
128    public Collection<Annotation> getAnnotations();
129
130    /**
131     * <p>Does ...</p>
132     *
133     * @return
134     */
135    public Collection<DbXref> getXrefs();
136
137
138}
