027import java.util.Collection;
028import java.util.ArrayList;
029
030import org.biopax.psidev.ontology_manager.OntologyTermI;
031
032/**
033 * Representation of a cv term.
034 *
035 * @author Florian Reisinger (florian@ebi.ac.uk)
036 * @author Samuel Kerrien (skerrien@ebi.ac.uk)
037 * @version $Id$
038 * @since 2.0.0
039 */
040public class OntologyTermImpl implements OntologyTermI {
041
042    private String acc;
043
044    private String name;
045
046    private Collection<String> nameSynonyms;
047    
048    private String ontologyId;
049
050    //////////////////////////
051    // Constructors
052
053    public OntologyTermImpl( String acc ) {
054        setTermAccession( acc );
055    }
056
057    public OntologyTermImpl( String ont, String acc, String name ) {
058        this(acc);
059        this.name = name;
060        this.ontologyId = ont;
061    }
062    
063    ///// ///// ///// ///// /////
064    // Getter & Setter
065
066    public void setTermAccession( String accession ) {
067        if ( accession == null || accession.trim().length() == 0 ) {
068            throw new IllegalArgumentException( "You must give a non null/empty term accession" );
069        }
070        acc = accession;
071    }
072
073    public String getTermAccession() {
074        return acc;
075    }
076
077    public void setPreferredName( String preferredName ) {
078        name = preferredName;
079    }
080
081    public Collection<String> getNameSynonyms() {
082        if ( nameSynonyms == null ) {
083            nameSynonyms = new ArrayList();
084        }
085        return nameSynonyms;
086    }
087
088    public void setNameSynonyms( Collection<String> nameSynonyms ) {
089        this.nameSynonyms = nameSynonyms;
090    }
091
092    public String getPreferredName() {
093        return name;
094    }
095
096    @Override
097    public String toString() {
098        final StringBuilder sb = new StringBuilder();
099        sb.append( "OntologyTermImpl" );
100        sb.append( "{acc='" ).append( acc ).append( '\'' );
101        sb.append( ", name='" ).append( name ).append( '\'' );
102        sb.append( '}' );
103        return sb.toString();
104    }
105
106    @Override
107    public boolean equals( Object o ) {
108        if ( this == o ) return true;
109        if ( o == null || getClass() != o.getClass() ) return false;
110
111        OntologyTermImpl that = ( OntologyTermImpl ) o;
112
113        if ( !acc.equals( that.acc ) ) return false;
114
115        return true;
116    }
117
118    @Override
119    public int hashCode() {
120        int result;
121        result = acc.hashCode();
122        return result;
123    }
124
125        public String getOntologyId() {
126                return this.ontologyId;
127        }
128
129        public void setOntologyId(String ontologyId) {
130                this.ontologyId = ontologyId;
131        }
132}
