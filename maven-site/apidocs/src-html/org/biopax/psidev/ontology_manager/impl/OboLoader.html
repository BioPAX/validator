053    /**
054     * Parse the given OBO file and build a representation of the DAG into an IntactOntology.
055     *
056     * @param file the input file. It has to exist and to be readable, otherwise it will break.
057     * @return a non null IntactOntology.
058     */
059    public OntologyAccess parseOboFile( File file, String ontologyID) {
060
061        if ( !file.exists() ) {
062            throw new IllegalArgumentException( file.getAbsolutePath() + " doesn't exist." );
063        }
064
065        if ( !file.canRead() ) {
066            throw new IllegalArgumentException( file.getAbsolutePath() + " could not be read." );
067        }
068
069        try {
070            setParser(new OBO2FormatParser(file.getAbsolutePath()));
071            process();
072        } catch ( Exception e ) {
073            log.fatal( "Parse failed: " + e.getMessage(), e );
074        }
075
076        return buildOntology(ontologyID);
077    }
078    
079    
080    /**
081     * Load an OBO file from an URL.
082     *
083     * @param url the URL to load (must not be null)
084     * @return an ontology
085     * @see #parseOboFile(File, String)
086     */
087    public OntologyAccess parseOboFile( URL url, String ontologyID ) throws OntologyLoaderException {
088        // load config file (ie. a map)
089        // check if that URL has already been loaded
090        // if so, get the associated temp file and check if available
091        // if available, then load it and skip URL load
092        // if any of the above failed, load it from the network.
093
094        if ( url == null ) {
095            throw new IllegalArgumentException( "Please give a non null URL." );
096        }
097
098        File ontologyFile = null;
099
100        try {
101            if ( ontologyFile == null || !ontologyFile.exists() || !ontologyFile.canRead() ) {
102                // if it is not defined, not there or not readable...
103                // Read URL content
104                log.info( "Loading URL: " + url );
105                URLConnection con = url.openConnection();
106                long size = con.getContentLength(); // -1 if not stat available
107                log.info( "size = " + size );
108
109                InputStream is = url.openStream();
110
111                // make the temporary file name specific to the URL
112                String name = null;
113                String filename = url.getFile();
114                int idx = filename.lastIndexOf( '/' );
115                if ( idx != -1 ) {
116                    name = filename.substring( idx+1, filename.length() );
117                    name = name.replaceAll("[.,;:&^%$@*?=]", "_");
118                } else {
119                    name = "unknown";
120                }
121
122                ontologyFile = Files.createTempFile(name + "_", ".obo").toFile();
123                ontologyFile.deleteOnExit();
124                log.debug( "The OBO file will be temporary stored as: " + ontologyFile.getAbsolutePath() );
125
126                FileOutputStream out = new FileOutputStream( ontologyFile );
127                if(size == -1) size = 1024 * 1024 * 1024; //Integer.MAX_VALUE;
128                ReadableByteChannel source = Channels.newChannel(is);
129                                size = out.getChannel().transferFrom(source, 0, size);
130                                log.info(size + " bytes downloaded");
131
132                is.close();
133                out.flush();
134                out.close();
135            }
136
137            if ( ontologyFile == null ) {
138                log.error( "The ontology file is still null..." );
139            }
140
141            // Parse file
142            return parseOboFile( ontologyFile, ontologyID );
143
144        } catch ( IOException e ) {
145            throw new OntologyLoaderException( "Error while loading URL (" + url + ")", e );
146        }
147    }

150    private OntologyAccess buildOntology(String ontologyID) {
151
152        OntologyAccess ontologyAccess = new OntologyAccessImpl();
153
154        // 1. convert and index all terms (note: at this stage we don't handle the hierarchy)
155        for ( Iterator iterator = ontBean.getTerms().iterator(); iterator.hasNext(); ) {
156            TermBean term = ( TermBean ) iterator.next();
157
158            /*
159             * Quick workaround for that
160             * we want to ignore the PSI-MOD terms that are included into PSI-MI files!
161             */
162            if("PSI-MOD".equals(term.getNamespace()) 
163                        && ("PSI-MI".equals(ontologyID) || "MI".equals(ontologyID)))
164                continue; // skip
165            
166            // convert term into a OboTerm
167            OntologyTermI ontologyTerm = new OntologyTermImpl(ontologyID, term.getIdentifier(), term.getName());
168// this workaround is not required anymore (since obo-fetcher impl. has changed).
169//          StringEscapeUtils.unescapeXml(term.getName()) );
170//          // unescapeXml above and below was a workaround a bug in ols-1.18 OBO parser (org.biopax.ols.loader..),
171//          // which returned, e.g., "O4&;apos;-phospho-L-tyrosine" instead "O4'-phospho-L-tyrosine")
172            
173            final Collection<TermSynonym> synonyms = (Collection<TermSynonym>) term.getSynonyms();
174            if( synonyms != null ) {
175                for ( TermSynonym synonym : synonyms ) {
176//                    ontologyTerm.getNameSynonyms().add( StringEscapeUtils.unescapeXml(synonym.getSynonym()) );
177                    ontologyTerm.getNameSynonyms().add( synonym.getSynonym() );
178                }
179            }
180
181            ontologyAccess.addTerm( ontologyTerm );
182
183            if ( term.isObsolete() ) {
184                ontologyAccess.addObsoleteTerm( ontologyTerm );
185            }
186        }
187
188        // 2. build hierarchy based on the relations of the Terms
189        for ( Iterator iterator = ontBean.getTerms().iterator(); iterator.hasNext(); )
190        {
191            TermBean term = ( TermBean ) iterator.next();
192            // a quick workaround an issue that we want to ignore PSI-MOD included in PSI-MI
193            if("PSI-MOD".equals(term.getNamespace()) && ("PSI-MI".equals(ontologyID) || "MI".equals(ontologyID)))
194                continue; // skip
195            
196            if ( term.getRelationships() != null ) {
197                for ( Iterator iterator1 = term.getRelationships().iterator(); iterator1.hasNext(); ) {
198                    TermRelationship relation = ( TermRelationship ) iterator1.next();
199
200                    // to ignore PSI-MOD included in PSI-MI, simply to check for NPE -
201                    try {
202                        ontologyAccess.addLink( relation.getObjectTerm().getIdentifier(),
203                                relation.getSubjectTerm().getIdentifier() );
204                    } catch (NullPointerException e) {
205                                log.warn("Skipping terms relationship " + relation + "; " + e);
206                                        }
207                }
208            }
209        }
210
211        return ontologyAccess;
212    }
