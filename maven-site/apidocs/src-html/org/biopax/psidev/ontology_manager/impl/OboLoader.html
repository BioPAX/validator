063    public static final Log log = LogFactory.getLog( OboLoader.class );

067    public OboLoader( ) {
068    }

071    /**
072     * Parse the given OBO file and build a representation of the DAG into an IntactOntology.
073     *
074     * @param file the input file. It has to exist and to be readable, otherwise it will break.
075     * @return a non null IntactOntology.
076     */
077    public OntologyAccess parseOboFile( File file, String ontologyID) {
078
079        if ( !file.exists() ) {
080            throw new IllegalArgumentException( file.getAbsolutePath() + " doesn't exist." );
081        }
082
083        if ( !file.canRead() ) {
084            throw new IllegalArgumentException( file.getAbsolutePath() + " could not be read." );
085        }
086
087        try {
088            setParser(new OBO2FormatParser(file.getAbsolutePath()));
089            process();
090        } catch ( Exception e ) {
091            log.fatal( "Parse failed: " + e.getMessage(), e );
092        }
093
094        return buildOntology(ontologyID);
095    }
096    
097    
098    /**
099     * Load an OBO file from an URL.
100     *
101     * @param url the URL to load (must not be null)
102     * @return an ontology
103     * @see #parseOboFile(File, String)
104     */
105    public OntologyAccess parseOboFile( URL url, String ontologyID ) throws OntologyLoaderException {
106        // load config file (ie. a map)
107        // check if that URL has already been loaded
108        // if so, get the associated temp file and check if available
109        // if available, then load it and skip URL load
110        // if any of the above failed, load it from the network.
111
112        if ( url == null ) {
113            throw new IllegalArgumentException( "Please give a non null URL." );
114        }
115
116        File ontologyFile = null;
117
118        try {
119            if ( ontologyFile == null || !ontologyFile.exists() || !ontologyFile.canRead() ) {
120                // if it is not defined, not there or not readable...
121                // Read URL content
122                log.info( "Loading URL: " + url );
123                URLConnection con = url.openConnection();
124                long size = con.getContentLength(); // -1 if not stat available
125                log.info( "size = " + size );
126
127                InputStream is = url.openStream();
128
129                // make the temporary file name specific to the URL
130                String name = null;
131                String filename = url.getFile();
132                int idx = filename.lastIndexOf( '/' );
133                if ( idx != -1 ) {
134                    name = filename.substring( idx+1, filename.length() );
135                    name = name.replaceAll("[.,;:&^%$@*?=]", "_");
136                } else {
137                    name = "unknown";
138                }
139
140                ontologyFile = Files.createTempFile(name + "_", ".obo").toFile();
141                ontologyFile.deleteOnExit();
142                log.debug( "The OBO file will be temporary stored as: " + ontologyFile.getAbsolutePath() );
143
144                FileOutputStream out = new FileOutputStream( ontologyFile );
145                if(size == -1) size = 1024 * 1024 * 1024; //Integer.MAX_VALUE;
146                ReadableByteChannel source = Channels.newChannel(is);
147                                size = out.getChannel().transferFrom(source, 0, size);
148                                log.info(size + " bytes downloaded");
149
150                is.close();
151                out.flush();
152                out.close();
153            }
154
155            if ( ontologyFile == null ) {
156                log.error( "The ontology file is still null..." );
157            }
158
159            // Parse file
160            return parseOboFile( ontologyFile, ontologyID );
161
162        } catch ( IOException e ) {
163            throw new OntologyLoaderException( "Error while loading URL (" + url + ")", e );
164        }
165    }

168    private OntologyAccess buildOntology(String ontologyID) {
169
170        OntologyAccess ontologyAccess = new OntologyAccessImpl();
171
172        // 1. convert and index all terms (note: at this stage we don't handle the hierarchy)
173        for ( Iterator iterator = ontBean.getTerms().iterator(); iterator.hasNext(); ) {
174            TermBean term = ( TermBean ) iterator.next();
175
176            /*
177             * Quick workaround for that
178             * we want to ignore the PSI-MOD terms that are included into PSI-MI files!
179             */
180            if("PSI-MOD".equals(term.getNamespace()) 
181                        && ("PSI-MI".equals(ontologyID) || "MI".equals(ontologyID)))
182                continue; // skip
183            
184            // convert term into a OboTerm
185            OntologyTermI ontologyTerm = new OntologyTermImpl(ontologyID, term.getIdentifier(), term.getName());
186// this workaround is not required anymore (since obo-fetcher impl. has changed).
187//          StringEscapeUtils.unescapeXml(term.getName()) );
188//          // unescapeXml above and below was a workaround a bug in ols-1.18 OBO parser (org.biopax.ols.loader..),
189//          // which returned, e.g., "O4&;apos;-phospho-L-tyrosine" instead "O4'-phospho-L-tyrosine")
190            
191            final Collection<TermSynonym> synonyms = (Collection<TermSynonym>) term.getSynonyms();
192            if( synonyms != null ) {
193                for ( TermSynonym synonym : synonyms ) {
194//                    ontologyTerm.getNameSynonyms().add( StringEscapeUtils.unescapeXml(synonym.getSynonym()) );
195                    ontologyTerm.getNameSynonyms().add( synonym.getSynonym() );
196                }
197            }
198
199            ontologyAccess.addTerm( ontologyTerm );
200
201            if ( term.isObsolete() ) {
202                ontologyAccess.addObsoleteTerm( ontologyTerm );
203            }
204        }
205
206        // 2. build hierarchy based on the relations of the Terms
207        for ( Iterator iterator = ontBean.getTerms().iterator(); iterator.hasNext(); )
208        {
209            TermBean term = ( TermBean ) iterator.next();
210            // a quick workaround an issue that we want to ignore PSI-MOD included in PSI-MI
211            if("PSI-MOD".equals(term.getNamespace()) && ("PSI-MI".equals(ontologyID) || "MI".equals(ontologyID)))
212                continue; // skip
213            
214            if ( term.getRelationships() != null ) {
215                for ( Iterator iterator1 = term.getRelationships().iterator(); iterator1.hasNext(); ) {
216                    TermRelationship relation = ( TermRelationship ) iterator1.next();
217
218                    // to ignore PSI-MOD included in PSI-MI, simply to check for NPE -
219                    try {
220                        ontologyAccess.addLink( relation.getObjectTerm().getIdentifier(),
221                                relation.getSubjectTerm().getIdentifier() );
222                    } catch (NullPointerException e) {
223                                log.warn("Skipping terms relationship " + relation + "; " + e);
224                                        }
225                }
226            }
227        }
228
229        return ontologyAccess;
230    }
