008import java.util.Collection;
009import java.util.Set;
010
011/**
012 * Defines what can be asked to an ontology.
013 *
014 * @author Florian Reisinger
015 * @author Samuel Kerrien (skerrien@ebi.ac.uk)
016 * @author rodche (baberlab.org) - re-factoring
017 * @since 2.0.0
018 */
019public interface OntologyAccess {
020
021    /**
022     * This method builds a set of all allowed terms based on the specified parameters.
023     *
024     * @param accession     the accession number of the term wanted.
025     * @param allowChildren whether child terms are allowed.
026     * @param useTerm       whether to include the specified term ID in the set of allowed IDs.
027     * @return a non null set of allowed ontology terms for the specified parameters.
028     */
029    public Set<OntologyTermI> getValidTerms( String accession, boolean allowChildren, boolean useTerm );
030
031    /**
032     * Search a term by accession number.
033     *
034     * @param accession the accession to be searched for.
035     * @return an ontology term or null if not found.
036     */
037    public OntologyTermI getTermForAccession( String accession );
038    
039    void setName(String name);
040       
041    String getName();
042    
043    boolean hasTerms();
044
045    OntologyTermI search( String id );
046
047    Collection<OntologyTermI> getRoots();
048
049    Collection<OntologyTermI> getOntologyTerms();
050
051    Collection<OntologyTermI> getObsoleteTerms();
052
053    boolean isObsolete( OntologyTermI term );
054
055    Set<OntologyTermI> getDirectParents( OntologyTermI term );
056
057    Set<OntologyTermI> getDirectChildren( OntologyTermI term );
058
059    Set<OntologyTermI> getAllParents( OntologyTermI term );
060
061    Set<OntologyTermI> getAllChildren( OntologyTermI term );
062    
063    void addLink( String parentId, String childId );
064    
065    void addTerm( OntologyTermI term );
066    
067    void addObsoleteTerm( OntologyTermI term );
068
069}
