025import org.biopax.paxtools.controller.ShallowCopy;
026import org.biopax.paxtools.model.level3.EntityFeature;
027import org.biopax.paxtools.model.level3.EntityReference;
028import org.biopax.paxtools.model.level3.SimplePhysicalEntity;
029import org.biopax.paxtools.normalizer.Normalizer;
030import org.biopax.validator.api.AbstractRule;
031import org.biopax.validator.api.beans.Validation;
032import org.springframework.stereotype.Component;
033
034import java.util.HashSet;
035import java.util.Set;
036
037/**
038 *
039 * A rule to check if a Simple Physical Entity Feature is also a feature of
040 * its Entity Reference. If not, fixes it by adding the missing feature
041 * to the Entity Reference.
042 *
043 */
044@Component
045public class SimplePhysicalEntityFeaturesRule extends AbstractRule<SimplePhysicalEntity>{
046
047    public void check(final Validation validation, SimplePhysicalEntity thing) {
048        EntityReference er = thing.getEntityReference();
049        //sync: this er can belong to many SPEs, and this rule can be called from another thread for the other SPE
050        synchronized (er) {     
051                Set<EntityFeature> erFeatures =  new HashSet<EntityFeature>(er.getEntityFeature());//defensive copy
052                
053                Set<EntityFeature> peFeaturesAndNotFeatures = new HashSet<EntityFeature>();
054                peFeaturesAndNotFeatures.addAll(thing.getFeature());
055                peFeaturesAndNotFeatures.addAll(thing.getNotFeature());
056
057                for(EntityFeature ef: peFeaturesAndNotFeatures) {
058                        if(!erFeatures.contains(ef)) { //the ER does not have this EF...
059                                if(validation.isFix()) {
060                                        if(ef.getEntityFeatureOf() != null) {//it belongs to the other ER
061                                                //trying to generate a new unique URI for the EF copy, for any ER,EF pair is unique 
062                                                //(considering one-to-many, inverse functional constraints of the entityFeature prop.):
063                                                String uri = Normalizer.uri(er.getUri() + "_", null, ef.getUri(), ef.getModelInterface());
064                                                EntityFeature newEf = null; //check if there is one with this URI already; use that one then
065                                                for(EntityFeature f : er.getEntityFeature()) {
066                                                        if(uri.equals(f.getUri())) {
067                                                                newEf = f;
068                                                                break;
069                                                        }
070                                                }
071                                                if(newEf == null) {
072                                                        //make a copy
073                                                        newEf = (new ShallowCopy()).copy(ef, uri);
074                                                        assert newEf.getEntityFeatureOf() == null 
075                                                                : "getEntityFeatureOf of a shallow copy of EF is not null...";
076                                                }
077                                                //replace the EF in the SPE's feature or notFeature set
078                                                if(thing.getFeature().contains(ef)) {
079                                                        thing.removeFeature(ef);
080                                                        thing.addFeature(newEf);
081                                                } 
082                                                if(thing.getNotFeature().contains(ef)) {
083                                                        thing.removeNotFeature(ef);
084                                                        thing.addNotFeature(newEf);
085                                                }
086                                                // add the new one to the ER
087                                                er.addEntityFeature(newEf);
088                                        } else {
089                                                // add the one to the ER
090                                                er.addEntityFeature(ef);
091                                        }
092                                }
093
094                                error(validation, thing, "improper.feature.use", validation.isFix(), ef.getUri(), er.getUri());
095                        }
096                }        
097        }             
098    }
099
100    public boolean canCheck(Object thing) {
101        return (thing instanceof SimplePhysicalEntity) 
102                        && (((SimplePhysicalEntity) thing).getEntityReference() != null);
103    }
104}
