025import org.biopax.paxtools.model.level3.ComplexAssembly;
026import org.biopax.paxtools.model.level3.Conversion;
027import org.biopax.paxtools.model.level3.Degradation;
028import org.biopax.paxtools.model.level3.Interaction;
029import org.biopax.paxtools.model.level3.PhysicalEntity;
030import org.biopax.paxtools.model.level3.SimplePhysicalEntity;
031import org.biopax.paxtools.model.level3.SmallMolecule;
032import org.biopax.paxtools.util.ClassFilterSet;
033import org.biopax.validator.api.AbstractRule;
034import org.biopax.validator.api.beans.Validation;
035import org.springframework.stereotype.Component;
036
037import java.util.HashSet;
038import java.util.Set;
039
040/**
041 * This rule checks that a biopolymer on one side of a conversion 
042 * is that modified/relocated on the other side, and not new one;
043 * (two exceptions: ComplexAssembly and Degradation)
044 * 
045 * User: rodche
046 */
047@Component
048public class SimplePhysicalEntityConversionRule extends AbstractRule<SimplePhysicalEntity>
049{
050        
051    public void check(final Validation validation, SimplePhysicalEntity spe)
052    {           
053        Set<Conversion> conversions = new HashSet<Conversion>(
054                        new ClassFilterSet<Interaction,Conversion>(
055                                spe.getParticipantOf(), Conversion.class));
056       
057        for(Conversion conversion : conversions) {
058           if(conversion instanceof ComplexAssembly 
059                           || conversion instanceof Degradation) {
060                   continue; //ignore these conversion types
061           }
062           
063           Set<SimplePhysicalEntity> side = 
064                           new ClassFilterSet<PhysicalEntity,SimplePhysicalEntity>(
065                                           conversion.getLeft(), SimplePhysicalEntity.class);
066           
067           if(side.contains(spe)) //then compare with the other side
068                   side = new ClassFilterSet<PhysicalEntity,SimplePhysicalEntity>(
069                                           conversion.getRight(), SimplePhysicalEntity.class);
070  
071           if(!sameKindEntityExists(spe, side))
072                   error(validation, spe, "illegal.conversion", false, conversion);
073        }
074    }
075    
076    boolean sameKindEntityExists(SimplePhysicalEntity spe, Set<SimplePhysicalEntity> side) 
077    {           
078        assert !(spe instanceof SmallMolecule);
079        
080        boolean ret = false;
081        
082        for (SimplePhysicalEntity value : side) 
083        {
084                        if (!(value instanceof SmallMolecule)) 
085                        {
086                                if(value.getEntityReference() != null
087                                        && value.getEntityReference().isEquivalent(spe.getEntityReference())) 
088                                                return true;
089                        }
090                }
091        
092                return ret;
093    }
094    
095    public boolean canCheck(Object thing) {
096        return thing instanceof SimplePhysicalEntity 
097         && !(thing instanceof SmallMolecule);
098    }
099
100}
