025import org.biopax.validator.api.AbstractRule;
026import org.biopax.validator.api.beans.Validation;
027import org.springframework.stereotype.Component;
028import org.biopax.paxtools.model.*;
029import org.biopax.paxtools.model.level3.Entity;
030import org.biopax.paxtools.model.level3.UtilityClass;
031import org.biopax.paxtools.model.level3.Xref;
032import org.biopax.paxtools.model.level2.entity;
033import org.biopax.paxtools.model.level2.utilityClass;
034import org.biopax.paxtools.model.level2.xref;
035import org.biopax.paxtools.model.level2.externalReferenceUtilityClass;
036
037/**
038 * This class advises on instances of too general (top-level) BioPAX classes
039 * (Entity, UtilityClass, etc.)
040 *
041 * @author rodche
042 *
043 */
044@Component
045public class NotAllowedInstancesRule extends AbstractRule<BioPAXElement> {
046
047   final Class[] NOT_ALLOWED = {
048        Entity.class,
049        UtilityClass.class,
050        Xref.class,
051        xref.class,
052        entity.class,
053        utilityClass.class,
054        externalReferenceUtilityClass.class
055    };
056
057    private boolean notAllowed(BioPAXElement bp) {
058                for (int i = 0; i < NOT_ALLOWED.length; i++) {
059                        if (bp.getModelInterface().equals(NOT_ALLOWED[i])) {
060                                return true; // found
061                        }
062                }
063                return false;
064        }
065    
066        public boolean canCheck(Object thing) {
067                if(thing instanceof BioPAXElement) {
068                        return notAllowed((BioPAXElement) thing);
069                } else {
070                        return false;
071                }
072        }
073
074        public void check(final Validation validation, BioPAXElement thing) {
075                error(validation, thing, "not.allowed.element", false, thing.getModelInterface().getSimpleName());
076        }
077    
078}
