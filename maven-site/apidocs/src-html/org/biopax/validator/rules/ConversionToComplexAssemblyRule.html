025import org.biopax.paxtools.controller.Fetcher;
026import org.biopax.paxtools.controller.PropertyEditor;
027import org.biopax.paxtools.controller.SimpleEditorMap;
028import org.biopax.paxtools.model.BioPAXLevel;
029import org.biopax.paxtools.model.Model;
030import org.biopax.paxtools.model.level3.*;
031import org.biopax.paxtools.util.Filter;
032import org.biopax.validator.api.AbstractRule;
033import org.biopax.validator.api.beans.Validation;
034import org.springframework.stereotype.Component;
035
036import java.util.HashSet;
037import java.util.Set;
038
039/**
040 *
041 * A rule to check if the Conversion can be converted
042 * to a ComplexAssembly.
043 *
044 * Some Conversions are not cast as ComplexAssembly,
045 * although there is no modification to the PEs
046 * and there is a clear Complex formation
047 * throughout the process.
048 *
049 */
050@Component
051public class ConversionToComplexAssemblyRule extends AbstractRule<Conversion> {
052    public void check(final Validation validation, Conversion thing) {
053        //for thread safety (concurrency) we're using a new fetcher here rather than static one 
054        Fetcher fetcher = new Fetcher(
055                        SimpleEditorMap.L3, new Filter<PropertyEditor>() {
056                                //complex.component only
057                                public boolean filter(PropertyEditor editor) {
058                                        return editor.getProperty().equals("component");
059                                }
060                        });     
061        Set<PhysicalEntity> left = new HashSet<PhysicalEntity>(getPEsRecursively(thing.getLeft(), fetcher)); //need a mutable set
062        Set<PhysicalEntity> right = getPEsRecursively(thing.getRight(), fetcher);
063        left.removeAll(right);
064
065        int complexDiff = getComplexCount(thing.getLeft()) - getComplexCount(thing.getRight());
066        if( left.isEmpty()  // when there are no modifications really, but different no. complexes or participants
067                && (complexDiff != 0 || thing.getLeft().size() - thing.getRight().size() != 0 ))
068            error(validation, thing, "wrong.conversion.class", false, thing.getModelInterface());
069
070    }
071
072    private int getComplexCount(Set<PhysicalEntity> pes) {
073        int count = 0;
074
075        for(PhysicalEntity pe: pes) {
076            if(pe instanceof Complex)
077                count++;
078        }
079
080        return count;
081    }
082
083    private Set<PhysicalEntity> getPEsRecursively(Set<PhysicalEntity> pes, Fetcher fetcher) {
084        Model m = BioPAXLevel.L3.getDefaultFactory().createModel();
085        for(PhysicalEntity pe : pes) {
086                if(pe instanceof Complex)
087                        fetcher.fetch(pe,m);
088                else 
089                        if(!m.containsID(pe.getUri()))
090                                m.add(pe);
091        }   
092
093        return m.getObjects(PhysicalEntity.class);
094    }
095
096    public boolean canCheck(Object thing) {
097        return thing instanceof Conversion
098                && !(thing instanceof ComplexAssembly);
099    }
100}
