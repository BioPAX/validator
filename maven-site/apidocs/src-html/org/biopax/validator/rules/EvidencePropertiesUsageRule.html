025import org.biopax.paxtools.model.level3.Evidence;
026import org.biopax.paxtools.model.level3.EvidenceCodeVocabulary;
027import org.biopax.paxtools.model.level3.ExperimentalForm;
028import org.biopax.paxtools.model.level3.Score;
029import org.biopax.validator.api.AbstractRule;
030import org.biopax.validator.api.beans.Validation;
031import org.springframework.stereotype.Component;
032
033/**
034 * Checks:
035 * Evidence properties and cardinality constraints.
036 * 
037 * This is not a trivial "cardinality and range rule" that
038 * cannot simply extend the CardinalityAndRangeRule class.
039 * 
040 * @author rodche
041 */
042@Component
043public class EvidencePropertiesUsageRule extends AbstractRule<Evidence> {
044
045    public void check(final Validation validation, Evidence ev) {
046
047                // check it has AT LEAST one of the things
048                if ((ev.getEvidenceCode() == null || ev.getEvidenceCode().isEmpty())
049                                && (ev.getExperimentalForm() == null || ev.getExperimentalForm().isEmpty())
050                                && (ev.getConfidence() == null || ev.getConfidence().isEmpty())) {
051                        error(validation, ev,
052                                        "min.cardinality.violated", false, "'evidenceCode' or 'confidence' or 'experimantalForm'", 1);
053                } else {
054
055                        // evidenceCode range
056                        if (ev.getEvidenceCode() != null) {
057                                for (Object cv : ev.getEvidenceCode()) {
058                                        if (!EvidenceCodeVocabulary.class.isInstance(cv)) {
059                                                error(validation, ev, "range.violated", 
060                                                                false, "evidenceCode",
061                                                                cv, cv.getClass().getSimpleName(), "EvidenceCodeVocabulary");
062                                        }
063                                }
064                        }
065
066                        // experimentalForm
067                        if (ev.getExperimentalForm() != null) {
068                                for (Object cv : ev.getExperimentalForm()) {
069                                        if (!ExperimentalForm.class.isInstance(cv)) {
070                                                error(validation, ev, "range.violated", 
071                                                                false, "experimentalForm", cv, cv.getClass().getSimpleName(), "ExperimentalForm");
072                                        }
073                                }
074                        }
075
076                        // confidence
077                        if (ev.getConfidence() != null) {
078                                for (Object cv : ev.getConfidence()) {
079                                        if (!Score.class.isInstance(cv)) {
080                                                error(validation, ev, "range.violated", 
081                                                                false, "confidence", cv, cv.getClass().getSimpleName(), "Score");
082                                        }
083                                }
084                        }
085                }
086        }
087
088        public boolean canCheck(Object thing) {
089                return thing instanceof Evidence;
090        }
091
092}
