025import org.apache.commons.collections15.CollectionUtils;
026import org.biopax.paxtools.model.level3.BiochemicalReaction;
027import org.biopax.paxtools.model.level3.Complex;
028import org.biopax.paxtools.model.level3.PhysicalEntity;
029import org.biopax.paxtools.model.level3.SimplePhysicalEntity;
030import org.biopax.paxtools.model.level3.Transport;
031import org.biopax.validator.api.AbstractRule;
032import org.biopax.validator.api.beans.Validation;
033import org.springframework.stereotype.Component;
034
035import java.util.Set;
036
037/**
038 * BiochemicalReaction (base) - the participants on one side don't 
039 * change compartments on the other side (otherwise it should be a 
040 * TransportWithBiochemicalReaction)
041 * 
042 * TODO Complexes are matched by name, other physical entities can be "identified" using entity reference.
043 * 
044 * @author rodche
045 */
046@Component
047public class BiochemReactParticipantsLocationRule extends AbstractRule<BiochemicalReaction>
048{
049
050    public void check(final Validation validation, BiochemicalReaction react)
051    {
052       Set<PhysicalEntity> left = react.getLeft();
053       Set<PhysicalEntity> right = react.getRight();
054       for (PhysicalEntity lefty : left)
055       {
056            for (PhysicalEntity righty : right)
057            {
058                // PhysicalEntity is either Complex or SimplePhysicalEntity (NucleicAcid, Protein or SmallMolecule)
059                
060                if(!lefty.getModelInterface().equals(righty.getModelInterface())) {
061                        continue;
062                }
063                
064                boolean sameComplex = (lefty instanceof Complex) // means - righty is also Complex
065                        && CollectionUtils.containsAny(lefty.getName(), righty.getName());
066                // Complex does not have entityReference to match the same kind on both sides...
067                
068                boolean sameSimplePhysicalEntity = (lefty instanceof SimplePhysicalEntity) 
069                        && ( ((SimplePhysicalEntity)lefty).getEntityReference() == null 
070                                ? ((SimplePhysicalEntity)righty).getEntityReference() == null 
071                                : ((SimplePhysicalEntity)lefty).getEntityReference()
072                                        .isEquivalent(((SimplePhysicalEntity)righty).getEntityReference()) );
073                
074                if(sameComplex || sameSimplePhysicalEntity) {
075                        boolean sameLoc = lefty.hasEquivalentCellularLocation(righty);
076                        if(!sameLoc     && !(react instanceof Transport)) {
077                                error(validation, react, "participant.location.changed", false, lefty, righty); 
078                        } else if(sameLoc && react instanceof Transport) {
079                                error(validation, react, "transport.location.same", false, lefty, righty); 
080                        }
081                } 
082            }
083        }
084    }
085
086    public boolean canCheck(Object thing) {
087        return thing instanceof BiochemicalReaction;
088    }
089}
