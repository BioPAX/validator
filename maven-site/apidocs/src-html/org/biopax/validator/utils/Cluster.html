001package org.biopax.validator.utils;
002
003import java.util.*;
004
005public abstract class Cluster<T> {
006        
007        /**
008         * Implemented in concrete subclasses.
009         * Must be "symmetric"; may be "transitive" or not (you decide) -
010         * 
011         * @param a
012         * @param b
013         * @return true/false
014         */
015        public abstract boolean match(T a, T b);
016        
017        
018        
019        /**
020         * Puts elements into sets of "similar" ones. 
021         * Every object belongs to exactly one cluster.
022         * Clusters contain at least one object.
023         * 
024         * If concrete {@link #match(Object, Object)} method
025         * is not transitive, this produces such groups, where,
026         * match(A,B) is not necessarily true for all pairs in
027         * the same cluster, but there exists Z, such as,
028         * match(X,Z) and match(Y,Z) are true.
029         * 
030         * @see #match(Object, Object)
031         * 
032         * @param elements
033         * @param clusterMaxSize
034         * @return
035         */
036        public final Set<Set<T>> cluster(Collection<T> elements, int clusterMaxSize) {
037                Set<Set<T>> clusters = new HashSet<Set<T>>();
038                for (T u : new HashSet<T>(elements)) {
039                        Set<Set<T>> toMerge = new HashSet<Set<T>>();
040                        for (Set<T> clu : clusters) {
041                                if(!clu.isEmpty() && matchAny(u, clu)) {
042                                        clu.add(u);
043                                        toMerge.add(clu);
044                                }
045                        }
046                        
047                        if(toMerge.isEmpty()) { //create a new cluster 
048                                Set<T> group = new HashSet<T>();
049                                group.add(u);
050                                clusters.add(group);
051                        } else if (toMerge.size() > 1) {
052                                // merge several groups into the first one
053                                Iterator<Set<T>> it = toMerge.iterator();
054                                Set<T> group = it.next();
055                                assert !group.isEmpty();
056                                while(it.hasNext()) {
057                                        Set<T> g = it.next();
058                                        group.addAll(g);
059                                        g.clear();
060                                }
061                        }
062                }
063                
064                //remove empty groups
065                Set<Set<T>> result = new HashSet<Set<T>>();
066                for(Set<T> clu : clusters) {
067                        if(!clu.isEmpty())
068                                result.add(clu);
069                }
070                
071                return result;
072        }
073        
074        
075        private boolean matchAny(T u, Set<T> clu) {
076                for(T t : clu) {
077                        if(match(u, t) || match(t, u))
078                                return true;
079                }
080                return false;
081        }
082        
083}
