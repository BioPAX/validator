025import java.util.HashSet;
026import java.util.Set;
027
028import org.biopax.paxtools.controller.EditorMap;
029import org.biopax.paxtools.controller.PropertyEditor;
030import org.biopax.paxtools.model.BioPAXElement;
031import org.biopax.validator.api.AbstractRule;
032import org.biopax.validator.api.ValidatorException;
033import org.biopax.validator.api.beans.Validation;
034
035/**
036 * Base BioPAX Rule for checking cardinality and range restrictions.
037 *
038 * @author rodche
039 *
040 * @param <E> extends BioPAXElement
041 */
042
043public abstract class AbstractCardinalityAndRangeRule<E extends BioPAXElement> 
044        extends AbstractRule<E> 
045{
046        protected EditorMap editorMap;
047        private final String property;
048        private final Class<E> domain;
049        private int minCardinality = 0;
050        private int maxCardinality = Integer.MAX_VALUE;
051        Class<?>[] ranges; // although the editor has range property, this is for stricter checks
052        String rangesAsString = "";
053        
054        // Constructor with arguments
055        public AbstractCardinalityAndRangeRule(
056                        Class<E> domain, String property, 
057                        int min, int max, Class<?>... ranges) {
058                this.domain = domain;
059                this.property = property;
060                this.minCardinality = (min >= 0) ? min : 0;
061                this.maxCardinality = (max > 0) ? max : Integer.MAX_VALUE;
062                this.ranges = ranges;
063                for(Class<?> cl : ranges) {
064                        rangesAsString += cl.getSimpleName() + " ";
065                }
066        }
067        
068        
069        public boolean canCheck(Object thing) {
070                return domain.isInstance(thing);
071        }
072        
073        public void check(Validation validation, E thing) {
074                PropertyEditor editor = 
075                        editorMap.getEditorForProperty(property, thing.getModelInterface());
076                if(editor == null) {
077                        throw new ValidatorException(
078                                "BUG in " + getClass().getSimpleName() +
079                                " rule: no editor exists for property '" + 
080                                                property + "' of " + domain);
081                }
082
083                //get value(s) from the property of the biopax obj; copy to avoid CMEx...
084                Set<?> ret = null;
085                //sync to get the property values (there're other rules in separate threads that might check/fix the same thing and property)
086                synchronized (thing) {
087                        ret = new HashSet<Object>(editor.getValueFromBean(thing));
088                }
089                        
090                int size = ret.size();
091                if (maxCardinality == minCardinality) {
092                        // exact cardinality check
093                        if (maxCardinality  > 0 
094                                        && maxCardinality < Integer.MAX_VALUE 
095                                        && size != maxCardinality) 
096                        {
097                                error(validation, thing, 
098                                                "cardinality.violated", false, editor.getProperty(), maxCardinality);
099                        }
100                } else {
101                        // min. cardinality check
102                        if (minCardinality > 0 && size == 0) {
103                                error(validation, thing, 
104                                                "min.cardinality.violated", false, editor.getProperty(), minCardinality);
105                        }
106                        // max. cardinality check
107                        if (maxCardinality > 0
108                                        && maxCardinality < Integer.MAX_VALUE
109                                        && size > maxCardinality) {
110                                error(validation, thing, 
111                                                "max.cardinality.violated", false, editor.getProperty(), maxCardinality);
112                        }
113                }
114                // check range
115                for (Object val : ret) {
116                        checkRange(validation, thing, val);
117                }
118        }
119        
120        private void checkRange(Validation validation, E thing, Object p) {
121                boolean isViolated = true;
122                for (Class<?> range : ranges) {
123                        if (range.isInstance(p)) {
124                                isViolated = false;
125                                break;
126                        }
127                }
128                if(isViolated) {
129                        error(validation, thing, "range.violated", 
130                                        false, getProperty(), 
131                                        p, p.getClass().getSimpleName(), rangesAsString);
132                }
133        }
134
135
136        public EditorMap getEditorMap() {
137                return editorMap;
138        }
139
140
141        public int getMinCardinality() {
142                return minCardinality;
143        }
144
145
146        public void setMinCardinality(int minCardinality) {
147                this.minCardinality = minCardinality;
148        }
149
150
151        public int getMaxCardinality() {
152                return maxCardinality;
153        }
154
155
156        public void setMaxCardinality(int maxCardinality) {
157                this.maxCardinality = maxCardinality;
158        }
159
160
161        public Class<?>[] getRanges() {
162                return ranges;
163        }
164
165
166        public void setRanges(Class<?>[] ranges) {
167                this.ranges = ranges;
168        }
169
170
171        public String getProperty() {
172                return property;
173        }
174
175
176        public Class<E> getDomain() {
177                return domain;
178        }       
179        
180}
