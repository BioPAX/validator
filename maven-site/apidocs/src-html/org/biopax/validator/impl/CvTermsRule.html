026
027import javax.annotation.PostConstruct;
028
029import org.biopax.paxtools.controller.SimpleEditorMap;
030import org.biopax.paxtools.model.BioPAXLevel;
031import org.biopax.paxtools.model.level3.Level3Element;
032import org.biopax.paxtools.model.level3.ControlledVocabulary;
033import org.biopax.paxtools.model.level3.UnificationXref;
034import org.biopax.paxtools.model.level3.Xref;
035import org.biopax.paxtools.normalizer.Normalizer;
036import org.biopax.paxtools.util.ClassFilterSet;
037import org.biopax.psidev.ontology_manager.*;
038import org.biopax.validator.api.CvRestriction;
039import org.biopax.validator.api.beans.Validation;

042/**
043 * An abstract class for CV terms checking (BioPAX Level3).
044 * 
045 * @author rodch
046 */
047public abstract class CvTermsRule<T extends Level3Element> 
048        extends AbstractCvRule<T> {
049  
050    /**
051     * Constructor.
052     * 
053     * TODO a feature: to allow using a properties path in the 'property' parameter, like "modificationFeature/modificationType"
054     * 
055     * @param domain
056     * @param property
057     * @param restrictions
058     */
059    public CvTermsRule(Class<T> domain, String property, CvRestriction... restrictions)
060    {
061        super(domain, property, restrictions);
062    }
063   
064    @PostConstruct
065    @Override
066    public void init() {
067        super.init();
068                this.editor = (property != null && !ControlledVocabulary.class.isAssignableFrom(domain)) 
069                        ? SimpleEditorMap.L3.getEditorForProperty(property, this.domain)
070                        : null;         
071    };
072    
073    
074        public void check(Validation validation, T thing) {
075                // a set of CVs for this rule to validate
076                Collection<ControlledVocabulary> vocabularies = new HashSet<ControlledVocabulary>();
077                
078                // if the editor is null, we expect a ControlledVocabulary object!
079                if(editor == null) {
080                        vocabularies.add((ControlledVocabulary)thing); 
081                } else {
082                        vocabularies.addAll((Collection<ControlledVocabulary>) editor.getValueFromBean(thing));
083                }
084                
085                // shortcut
086                if(vocabularies.isEmpty()) return;
087                
088                // text to report in any CV error case
089                String cvRuleInfo = ((editor != null) ? " property: "
090                                + property : "") + " " + restrictions.toString();
091                
092                // check each CV terms against the restrictions
093                for (ControlledVocabulary cv : vocabularies) 
094                {
095                        if (cv == null) {
096                                logger.warn(thing
097                                        + " referes to 'null' controlled vocabulary (bug!): "
098                                        + ", domain: " + domain + ", property: " + property);
099                        } else if(cv.getTerm().isEmpty()) {
100                                /* won't report/fix what other rules (e.g., 'controlledVocabularyTermCRRule') 
101                                 * or Normalizercan do! */
102                        } 
103                        else {
104                                //TODO (an advanced feature, a separate rule - ) to check if multiple terms are synonyms (equivalent)...
105                                
106                                final Set<String> badTerms = new HashSet<String>(); // initially - none
107                                final Map<String, Set<OntologyTermI>> noXrefTerms = new HashMap<String, Set<OntologyTermI>>();                          
108                                //original terms set to iterate over (to avoid concurrent modification exceptions - other rules can modify the set simultaneously)
109                                final Set<String> terms = Collections.unmodifiableSet(new HashSet<String>(cv.getTerm()));
110                                
111                                // first, check terms (names) are valid
112                                for(String name : terms) 
113                                {
114                                        if(!getValidTerms().contains(name.toLowerCase())) {
115                                                // save to report/delete/replace the invalid term later
116                                                badTerms.add(name);
117                                        }
118                                }
119                                // report but keep original perhaps illegal terms
120                                if (!badTerms.isEmpty()) {      
121                                        String badTermInfo = badTerms.toString();
122                                        error(validation, thing, "illegal.cv.term", false, badTermInfo, cvRuleInfo);
123                                }                                                               
124                                
125                                /* check if unif. xref.id points to invalid term, 
126                                 * and, if so, report 'illegal.cv.xref' error
127                                 */
128                                final Set<UnificationXref> badXrefs = new HashSet<UnificationXref>();
129                                for (UnificationXref x : new ClassFilterSet<Xref,UnificationXref>(
130                                                cv.getXref(), UnificationXref.class)) 
131                                {
132                                        OntologyTermI ot = ((OntologyManager) ontologyManager).findTermByAccession(x.getId());
133                                        if(ot == null || !getValidTerms().contains(ot.getPreferredName().toLowerCase())) {
134                                                badXrefs.add(x);
135                                        }
136                                }                               
137                                // report wrong uni.xrefs
138                                if(!badXrefs.isEmpty()) {
139                                        String bads = badXrefs.toString();
140                                        // report as not fixed error case (won't fix/remove such xrefs, keep original)
141                                        error(validation, thing, "illegal.cv.xref", false, bads, cvRuleInfo);
142                                }                               
143                                
144                                // check valid terms have a uni.xref
145                                for(String name : terms) 
146                                {
147                                        // only for valid terms
148                                        if(getValidTerms().contains(name.toLowerCase())) {
149                                                // check if there is the corresponding unification xref
150                                                Set<OntologyTermI> ots = ((OntologyManager) ontologyManager)
151                                                                .searchTermByName(name.toLowerCase(), getOntologyIDs());
152                                                assert(!ots.isEmpty()); // shouldn't be, because the above getValidTerms() contains the name
153                                                boolean noXrefsForTermNameFound = true; // next, - prove otherwise is the case
154                                                terms: for(OntologyTermI term : ots) {
155                                                        String id = term.getTermAccession();
156                                                        // search for the xref with the same xref.id
157                                                        for (UnificationXref x : new ClassFilterSet<Xref,UnificationXref>(
158                                                                        cv.getXref(), UnificationXref.class)) {
159                                                                if(id.equalsIgnoreCase(x.getId()))  {
160                                                                        noXrefsForTermNameFound = false;
161                                                                        break terms; // exit this and outer loops!
162                                                                }
163                                                        }
164                                                }
165                                                
166                                                if(noXrefsForTermNameFound)
167                                                        noXrefTerms.put(name, ots); //store terms to fix later (to generate xrefs)
168                                        }
169                                }
170                                
171                                // note: at this point, 'noXrefTerms' (valid terms only) map is defined...                              
172                                if (!noXrefTerms.isEmpty()) {           
173                                        String noXrefTermsInfo = noXrefTerms.toString();
174                                        boolean fixed = false;
175                                        
176                                        if(validation.isFix()) {
177                                        /*
178                                         * However, it's not so trivial to fix by adding the xrefs, because:
179                                         * 1) no reference to the parent Model here available
180                                         *    (thus the validator must detect and add new objects automatically! [done!])
181                                         * 2) having the chance of creating several xrefs with the same RDFId requires 
182                                         *    a special care or follow-up merging, as simply adding them to a model will 
183                                         *    throw the "already have this element" exception!); and other rules 
184                                         *    can also generate duplicates...
185                                         * 3) risk that a rule generating/adding a new element may cause  
186                                         *    other rules to interfere via AOP and prevent changes in quite 
187                                         *    unpredictable manner (...bites its own tail)
188                                         * 4) multiple terms (accession numbers) can result from searching by (synonym) name
189                                         *    
190                                         *    Well, let's try to fix, anyway (and modifying ValidatorImpl as well)!
191                                         *    That's awesome!
192                                         */
193                                                Set<OntologyTermI> validTermIs = ontologyManager.getValidTerms(this);
194                                                for (String name : noXrefTerms.keySet()) {
195                                                        //get previously saved valid ontology term beans by name
196                                                        Set<OntologyTermI> ots = noXrefTerms.get(name);
197                                                        //get only top (parent) valid terms
198                                                        Set<OntologyTermI> topvalids = new HashSet<OntologyTermI>();
199                                                        for (OntologyTermI term : ots) {
200                                                                // skip terms that are not applicable although having the same synonym name
201                                                                if(validTermIs.contains(term)) {
202                                                                        OntologyAccess ont = ((OntologyManager)ontologyManager).getOntology(term.getOntologyId());
203                                                                        //if term's parents does not contain any of these terms
204                                                                        if(Collections.disjoint(ots, ont.getAllParents(term))) {
205                                                                                topvalids.add(term);
206                                                                        }
207                                                                }
208                                                        }
209                                                        Set<String> added = new HashSet<String>();
210                                                        for (OntologyTermI term : topvalids) {                                                          
211                                                                String ontId = term.getOntologyId();
212                                                                String db = ((OntologyManager) ontologyManager).getOntology(ontId).getName();
213                                                                String id = term.getTermAccession();
214                                                                // auto-create and add the xref to the cv;
215                                                                // generate an URI in the same namespace
216                                                                String uri = Normalizer.uri(cv.getUri() + "_", db, id, UnificationXref.class);
217                                                                if(!added.contains(uri)) {
218                                                                        added.add(uri);
219                                                                        UnificationXref ux = BioPAXLevel.L3.getDefaultFactory()
220                                                                                        .create(UnificationXref.class, uri);
221                                                                        ux.setDb(db);
222                                                                        ux.setId(id);
223                                                                        cv.addXref(ux);
224                                                                        fixed = true; // 99% true ;-)
225                                                                        noXrefTermsInfo += "; " + id + " added!";
226                                                                }       
227                                                        }
228                                                }
229                                        }
230                                        
231                                        // report                                       
232                                        error(validation, thing, "no.xref.cv.terms", 
233                                                fixed, noXrefTermsInfo, cvRuleInfo);
234                                }
235                                
236                                //if in the fixing mode, 
237                                if (validation != null && validation.isFix() 
238                                                //and there were some errors found, 
239                                                && !(badTerms.isEmpty() && noXrefTerms.isEmpty())) 
240                                {
241                                        //then add/infer the valid preferred term from the unification xrefs -
242                                        Set<String> addTerms = createTermsFromUnificationXrefs(cv);
243                                        if (!addTerms.isEmpty()) {
244                                                cv.getTerm().addAll(addTerms);
245                                        }
246                                }                               
247                                
248                        }
249                }
250        }
251                
252        //discover valid terms by unification xrefs (invalid xrefs won't get you anything)
253        private Set<String> createTermsFromUnificationXrefs(
254                        ControlledVocabulary cv) 
255        {               
256                Set<String> inferred = new HashSet<String>();
257                for (UnificationXref x : new ClassFilterSet<Xref,UnificationXref>(
258                                cv.getXref(), UnificationXref.class)) 
259                {
260                        OntologyTermI ot = ((OntologyManager) ontologyManager)
261                                        .findTermByAccession(x.getId());
262                        //if found and valid
263                        if (ot != null && getValidTerms().contains(ot.getPreferredName().toLowerCase())) {
264                                inferred.add(ot.getPreferredName());
265                        } 
266                        else if(ot == null)
267                                logger.warn("No term found by the xref.id: " + x.getId());
268                        else 
269                                logger.debug("Invalid (for this CV context) term: " + x.getId());
270                }
271                
272                return inferred;
273        }
