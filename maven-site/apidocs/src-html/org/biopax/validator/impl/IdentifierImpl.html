028import org.biopax.paxtools.io.SimpleIOHandler;
029import org.biopax.paxtools.model.BioPAXElement;
030import org.biopax.paxtools.model.Model;
031import org.biopax.validator.api.beans.Validation.Identifier;
032
033/**
034 * BioPAX object domain specific implementation 
035 * of the {@link Identifier} strategy.
036 * 
037 * @author rodche
038 *
039 */
040public final class IdentifierImpl implements Identifier {
041
042        /* (non-Javadoc)
043         * @see org.biopax.validator.api.Identifier#getId(java.lang.Object)
044         */
045        public String identify(Object obj) {
046        String id = "";
047        
048                if(obj instanceof SimpleIOHandler) {
049                        SimpleIOHandler r = (SimpleIOHandler) obj;
050                        id = r.getClass().getSimpleName(); 
051                try {
052                        id = r.getId(); //current element URI
053                } catch (Throwable e) {
054                        id = r.getXmlStreamInfo(); //location
055                        }
056                } else if (obj instanceof BioPAXElement 
057                                && ((BioPAXElement)obj).getUri() != null) {
058                        id = ((BioPAXElement) obj).getUri().replaceFirst("^.+#", "");
059                        // - strictly spk., does not always get the local part (depends on xml:base) but is OK.
060                } else if(obj instanceof Model) {
061                        id = obj + "; xml:base=" + ((Model) obj).getXmlBase();
062                } else {
063                        id = "" + obj;
064                }
065                
066                return id;
067        }
068}
