025import java.io.Serializable;
026import java.util.*;
027import javax.xml.bind.annotation.*;
028
029
030@XmlRootElement
031@XmlType(name="ValidatorResponse")
032@XmlAccessorType(XmlAccessType.FIELD)
033public class ValidatorResponse implements Serializable {
034
035        private static final long serialVersionUID = 1L;
036        
037        @XmlElement
038        private List<Validation> validation;
039
040        public ValidatorResponse() {
041                validation = new ArrayList<Validation>();
042        }
043
044        public List<Validation> getValidationResult() {
045                return validation;
046        }
047
048        public void setValidationResult(List<Validation> validation) {
049                this.validation = validation;
050        }
051
052        public String toString() {
053                StringBuffer result = new StringBuffer(
054                super.toString());
055                result.append("(results count: ");
056                result.append(validation.size());
057                result.append(')');
058                return result.toString();
059        }
060        
061        public void addValidationResult (Validation result) {
062                this.validation.add(result);
063        }
064
065}
