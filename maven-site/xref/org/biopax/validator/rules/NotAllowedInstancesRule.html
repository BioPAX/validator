19  /**
20   * This class advises on instances of too general (top-level) BioPAX classes
21   * (Entity, UtilityClass, etc.)
22   *
23   * @author rodche
24   *
25   */
26  @Component
27  public class NotAllowedInstancesRule extends AbstractRule<BioPAXElement> {
28  
29     final Class[] NOT_ALLOWED = {
30          Entity.class,
31          UtilityClass.class,
32          Xref.class,
33  	xref.class,
34  	entity.class,
35  	utilityClass.class,
36  	externalReferenceUtilityClass.class
37      };
38  
39      private boolean notAllowed(BioPAXElement bp) {
40  		for (int i = 0; i < NOT_ALLOWED.length; i++) {
41  			if (bp.getModelInterface().equals(NOT_ALLOWED[i])) {
42  				return true; // found
43  			}
44  		}
45  		return false;
46  	}
47      
48  	public boolean canCheck(Object thing) {
49  		if(thing instanceof BioPAXElement) {
50  			return notAllowed((BioPAXElement) thing);
51  		} else {
52  			return false;
53  		}
54  	}
55  
56  	public void check(final Validation validation, BioPAXElement thing) {
57  		error(validation, thing, "not.allowed.element", false, thing.getModelInterface().getSimpleName());
58  	}
59      
60  }
