25  import org.biopax.paxtools.model.level3.Catalysis;
26  import org.biopax.paxtools.model.level3.Control;
27  import org.biopax.paxtools.model.level3.ControlType;
28  import org.biopax.paxtools.model.level3.TemplateReactionRegulation;
29  import org.biopax.validator.api.AbstractRule;
30  import org.biopax.validator.api.beans.Validation;
31  import org.springframework.stereotype.Component;
32  
33  /**
34   * Checks:
35   * 
36   * Catalysis.controlType - only "ACTIVATION"
37   * TemplateReactionRegulation.controlType - either "ACTIVATION" or "INHIBITION"
38   *
39   * @author rodche
40   */
41  @Component
42  public class ControlTypeRule extends AbstractRule<Control> {
43  
44  	private void fix(Control t, Object... values) {
45  		if(t instanceof Catalysis) {
46  			((Catalysis)t).setControlType(ControlType.ACTIVATION);
47  		} else if (t instanceof TemplateReactionRegulation){
48  			if(values.length >0 && values[0] instanceof ControlType) {
49  				ControlType ct = (ControlType) values[0];
50  				((TemplateReactionRegulation)t).setControlType(ct);
51  			} else {
52  				((TemplateReactionRegulation)t).setControlType(null);
53  				if(logger.isInfoEnabled()) 
54  					logger.info(t.getUri() +
55  							" - invalid controlType value deleted");
56  			}
57  		} else {
58  			logger.error("This does not auto-fix " +
59  					" controlType property of "
60  					+ t.getModelInterface().getSimpleName());
61  		}
62  	}
63  
64  	public boolean canCheck(Object thing) {
65  		return thing instanceof Control;
66  	}
67  
68  	public void check(final Validation validation, Control thing) {
69  		if(thing.getControlType() != null) 
70  		{			
71  			if(thing instanceof Catalysis) {
72  				Catalysis cat = (Catalysis) thing;
73  				if(cat.getControlType() != ControlType.ACTIVATION) {
74  					error(validation, thing, "range.violated", 
75  							validation.isFix(), "controlType",
76  							cat.getControlType().name(), "", ControlType.ACTIVATION.name()
77  							+ " (or empty)");
78  					if(validation.isFix()) {
79  						fix(thing);
80  					}
81  				}
82  			} else if(thing instanceof TemplateReactionRegulation) {
83  				TemplateReactionRegulation trr = (TemplateReactionRegulation) thing;
84  				if(! (trr.getControlType() == ControlType.ACTIVATION
85  						|| trr.getControlType() == ControlType.INHIBITION) ) 
86  				{
87  					error(validation, thing, "range.violated", 
88  							validation.isFix(), "controlType",
89  							trr.getControlType().name(), "", ControlType.ACTIVATION.name() + " or " 
90  							+ ControlType.INHIBITION.name()
91  							+ " (or empty)");
92  					if(validation.isFix()) {
93  						fix(thing);
94  					}
95  				}
96  			}
97  		}
98  	}
