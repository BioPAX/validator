25  import org.biopax.paxtools.model.level3.BindingFeature;
26  import org.biopax.paxtools.model.level3.EntityFeature;
27  import org.biopax.paxtools.model.level3.PhysicalEntity;
28  import org.biopax.validator.api.AbstractRule;
29  import org.biopax.validator.api.beans.Validation;
30  import org.springframework.stereotype.Component;
31  
32  import java.util.HashSet;
33  
34  /**
35   *
36   * A rule to check if PhysicalEntities that are both participating
37   * to a Conversion and a component of a Complex have proper BindingFeature(s).
38   * If not, it is hard to deduce whether the PE participates to the Conversion
39   * as a component of a Complex or separately?
40   *
41   */
42  @Component
43  public class PhysicalEntityAmbiguousFeatureRule extends AbstractRule<PhysicalEntity>{
44  
45      public void check(final Validation validation, PhysicalEntity thing) {
46          // Capture PEs that is both a participant of a conversion and a component of a complex
47          if(!thing.getParticipantOf().isEmpty() && !thing.getComponentOf().isEmpty()) {
48              HashSet<EntityFeature> efs = new HashSet<EntityFeature>();
49  
50              efs.addAll(thing.getFeature());
51              efs.addAll(thing.getNotFeature());
52  
53              // Do we have any information about the binding properties? If so, it's OK.
54              for(EntityFeature ef: efs) {
55                  if(ef instanceof BindingFeature) {
56                      return;
57                  }
58              }
59  
60              // TODO: Call Emek's feature resolver function explicitly if the PE is to be fixed
61  
62              error(validation, thing, "ambiguous.feature", false, thing.getName());
63          }
64      }
65  
66      public boolean canCheck(Object thing) {
67          return thing instanceof PhysicalEntity;
68      }
69  }
