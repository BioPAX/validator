31  import org.biopax.validator.api.AbstractRule;
32  import org.biopax.validator.api.beans.Validation;
33  import org.springframework.stereotype.Component;
34  
35  import java.util.Collections;
36  import java.util.Set;

38  /**
39   * This rule checks if either ComplexAssembly conversion has a complex on one side,
40   * or, at least, several participants are bound to each other on either left or right side. 
41   */
42  @Component
43  public class ComplexAssemblyHasComplexParticipantRule extends AbstractRule<ComplexAssembly>
44  {
45      public void check(final Validation validation, ComplexAssembly complexAssembly)
46      {
47  		// check if there are any complexes on either side
48      	Set<Complex> complexes = 
49  			new ClassFilterSet<Entity,Complex>(complexAssembly.getParticipant(), Complex.class);
50  		if(complexes.isEmpty()) {
51  			boolean bound = isBound(complexAssembly.getRight());
52  			if(!bound)
53  				bound = isBound(complexAssembly.getLeft());
54  			if(!bound)
55  				error(validation, complexAssembly, "complex.not.present", false);
56  		}
57      }
58  
59      /* check "whether participants acquire a binding feature" 
60       * would be much more difficult than - "several participants are bound"
61       * (e.g., because the former is subject to "the same entityReference" cond...)
62       */
63      private boolean isBound(Set<PhysicalEntity> participants) {
64      	for(PhysicalEntity pe : participants) {
65      		//if(pe instanceof Complex) continue; // it's impossible as far isBound called from where it is called
66      		if(pe.getFeature() != null 
67      				&& pe.getFeature() instanceof BindingFeature) {
68      			BindingFeature bf1 = (BindingFeature) pe.getFeature();
69      			BindingFeature bf2 = bf1.getBindsTo();
70      			if(bf2 != null) {
71      				Set<PhysicalEntity> bf2Of = bf2.getFeatureOf();
72      				//if these phys. entities are (same side) participants as well
73      				if(!Collections.disjoint(participants, bf2Of))
74      					return true;
75      			}
76      		}
77      	}
78      	return false;
79  	}
80  
81  	public boolean canCheck(Object thing) {
82          return thing instanceof ComplexAssembly;
83      }
