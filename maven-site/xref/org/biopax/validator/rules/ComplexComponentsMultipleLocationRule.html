7   import java.util.Collection;
8   import java.util.HashSet;
9   import java.util.Set;

11  import org.biopax.paxtools.model.level3.Complex;
12  import org.biopax.paxtools.model.level3.PhysicalEntity;
13  import org.biopax.validator.api.AbstractRule;
14  import org.biopax.validator.api.beans.Validation;
15  import org.springframework.stereotype.Component;
16  
17  /**
18   * Warning on complex components
19   * span multiple cellular compartments.
20   * 
21   * @author rodche
22   *
23   */
24  @Component
25  public class ComplexComponentsMultipleLocationRule extends
26  		AbstractRule<Complex> {
27  
28  	public void check(final Validation validation, Complex thing) {
29  		Set<PhysicalEntity> ents = thing.getComponent();
30  		if (ents != null) {
31  			Collection<String> diffLocs = new HashSet<String>();
32  			PhysicalEntity ref = thing;
33  			for (PhysicalEntity e : ents) {
34  				if(ref.getCellularLocation()==null 
35  						&& e.getCellularLocation() != null) {
36  					ref = e;
37  					continue;
38  				}
39  					
40  				if (e.getCellularLocation() != null
41  						&& !ref.getCellularLocation().isEquivalent(e.getCellularLocation())) {
42  					diffLocs.add(e + " " + e.getCellularLocation());		
43  				}
44  			}
45  			
46  			if(!diffLocs.isEmpty()) {
47  				error(validation, thing, 
48  						"component.different.location", false, thing.getCellularLocation() + "", diffLocs);
49  			}
50  		}
51  	}
52  
53  	public boolean canCheck(Object thing) {
54  		return thing instanceof Complex;
55  	}
56  }
