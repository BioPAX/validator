25  import org.biopax.paxtools.model.level3.Evidence;
26  import org.biopax.paxtools.model.level3.EvidenceCodeVocabulary;
27  import org.biopax.paxtools.model.level3.ExperimentalForm;
28  import org.biopax.paxtools.model.level3.Score;
29  import org.biopax.validator.api.AbstractRule;
30  import org.biopax.validator.api.beans.Validation;
31  import org.springframework.stereotype.Component;
32  
33  /**
34   * Checks:
35   * Evidence properties and cardinality constraints.
36   * 
37   * This is not a trivial "cardinality and range rule" that
38   * cannot simply extend the CardinalityAndRangeRule class.
39   * 
40   * @author rodche
41   */
42  @Component
43  public class EvidencePropertiesUsageRule extends AbstractRule<Evidence> {
44  
45      public void check(final Validation validation, Evidence ev) {
46  
47  		// check it has AT LEAST one of the things
48  		if ((ev.getEvidenceCode() == null || ev.getEvidenceCode().isEmpty())
49  				&& (ev.getExperimentalForm() == null || ev.getExperimentalForm().isEmpty())
50  				&& (ev.getConfidence() == null || ev.getConfidence().isEmpty())) {
51  			error(validation, ev,
52  					"min.cardinality.violated", false, "'evidenceCode' or 'confidence' or 'experimantalForm'", 1);
53  		} else {
54  
55  			// evidenceCode range
56  			if (ev.getEvidenceCode() != null) {
57  				for (Object cv : ev.getEvidenceCode()) {
58  					if (!EvidenceCodeVocabulary.class.isInstance(cv)) {
59  						error(validation, ev, "range.violated", 
60  								false, "evidenceCode",
61  								cv, cv.getClass().getSimpleName(), "EvidenceCodeVocabulary");
62  					}
63  				}
64  			}
65  
66  			// experimentalForm
67  			if (ev.getExperimentalForm() != null) {
68  				for (Object cv : ev.getExperimentalForm()) {
69  					if (!ExperimentalForm.class.isInstance(cv)) {
70  						error(validation, ev, "range.violated", 
71  								false, "experimentalForm", cv, cv.getClass().getSimpleName(), "ExperimentalForm");
72  					}
73  				}
74  			}
75  
76  			// confidence
77  			if (ev.getConfidence() != null) {
78  				for (Object cv : ev.getConfidence()) {
79  					if (!Score.class.isInstance(cv)) {
80  						error(validation, ev, "range.violated", 
81  								false, "confidence", cv, cv.getClass().getSimpleName(), "Score");
82  					}
83  				}
84  			}
85  		}
86  	}
87  
88  	public boolean canCheck(Object thing) {
89  		return thing instanceof Evidence;
90  	}

92  }
